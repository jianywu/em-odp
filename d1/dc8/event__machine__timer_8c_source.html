31  * Some notes about the implementation:
32  *
33  * EM timer add-on API is close to ODP timer, but there are issues
34  * making this code a bit more complex than it could be:
35  *
36  * 1) no periodic timer in ODP
37  * 2) unless using the pre-defined timeout event there is no way to access
38  * all necessary information runtime to implement a periodic timer
39  *
40  * Point 2 is solved by creating a timeout pool. When user allocates
41  * EM timeout, a new minimum size buffer is allocated to store all the needed
42  * information. Timer handle is a pointer to such buffer so all data is
43  * available via the handle (ack() is the most problematic case). This does
44  * create performance penalty, but so far it looks like the penalty is not
45  * too large and does simplify the code otherwise. Also timeouts could be
46  * pre-allocated as the API separates creation and arming.
47  * Most of the syncronization is handled by ODP timer, a ticketlock is used
48  * for high level management API.
49  *
50  */

59 static inline int is_queue_valid_type(em_timer_t tmr, const queue_elem_t *q_elem)
60 {
61  unsigned int tmridx = (unsigned int)TMR_H2I(tmr);
62 
63  /* implementation specific */
64  if (em_shm->timers.timer[tmridx].plain_q_ok && q_elem->type == EM_QUEUE_TYPE_UNSCHEDULED)
65  return 1;
66  /* EM assumes scheduled always supported */
67  return (q_elem->type == EM_QUEUE_TYPE_ATOMIC ||
68  q_elem->type == EM_QUEUE_TYPE_PARALLEL ||
69  q_elem->type == EM_QUEUE_TYPE_PARALLEL_ORDERED) ? 1 : 0;
70 
71  /* LOCAL or OUTPUT queues not supported */
72 }

74 static inline int is_timer_valid(em_timer_t tmr)
75 {
76  unsigned int i;
77  const timer_storage_t *const tmrs = &em_shm->timers;
78 
79  if (unlikely(tmr == EM_TIMER_UNDEF))
80  return 0;
81 
82  i = (unsigned int)TMR_H2I(tmr);
83  if (unlikely(i >= EM_ODP_MAX_TIMERS))
84  return 0;
85 
86  if (unlikely(tmrs->timer[i].odp_tmr_pool == ODP_TIMER_POOL_INVALID ||
87  tmrs->timer[i].tmo_pool == ODP_POOL_INVALID))
88  return 0;
89  return 1;
90 }

92 static inline em_status_t handle_ack_noskip(em_event_t next_tmo_ev,
93  event_hdr_t *ev_hdr,
94  em_queue_t queue)
95 {
96  if (esv_enabled())
97  evstate_usr2em_revert(next_tmo_ev, ev_hdr, EVSTATE__TMO_ACK__NOSKIP);
98 
99  em_status_t err = em_send(next_tmo_ev, queue);
100 
101  if (unlikely(err != EM_OK))
102  err = INTERNAL_ERROR(err, EM_ESCOPE_TMO_ACK, "Tmo ACK: noskip em_send fail");
103  return err; /* EM_OK or send-failure */
104 }

106 static inline void handle_ack_skip(em_tmo_t tmo)
107 {
108  uint64_t odpt = odp_timer_current_tick(tmo->odp_timer_pool);
109  uint64_t skips;
110 
111  if (odpt > tmo->last_tick) /* late, over next period */
112  skips = ((odpt - tmo->last_tick) / tmo->period) + 1;
113  else
114  skips = 1; /* not yet over next period, but late for setting */
115 
116  tmo->last_tick += skips * tmo->period;
117  TMR_DBG_PRINT("%lu skips * %lu ticks => new tgt %lu\n",
118  skips, tmo->period, tmo->last_tick);
119  if (EM_TIMER_TMO_STATS)
120  tmo->stats.num_period_skips += skips;
121 }

123 static inline bool check_tmo_flags(em_tmo_flag_t flags)
124 {
125  /* Check for valid tmo flags (oneshot OR periodic mainly) */
126  if (unlikely(!(flags & (EM_TMO_FLAG_ONESHOT | EM_TMO_FLAG_PERIODIC))))
127  return false;
128 
129  if (unlikely((flags & EM_TMO_FLAG_ONESHOT) && (flags & EM_TMO_FLAG_PERIODIC)))
130  return false;
131 
132  if (EM_CHECK_LEVEL > 1) {
133  em_tmo_flag_t inv_flags = ~(EM_TMO_FLAG_ONESHOT | EM_TMO_FLAG_PERIODIC |
134  EM_TMO_FLAG_NOSKIP);
135  if (unlikely(flags & inv_flags))
136  return false;
137  }
138  return true;
139 }

161 void em_timer_attr_init(em_timer_attr_t *tmr_attr)
162 {
163  if (unlikely(EM_CHECK_LEVEL > 0 && tmr_attr == NULL))
164  return; /* just ignore NULL here */
165 
166  /* strategy: first put default resolution, then validate based on that */
167  tmr_attr->resparam.res_ns = EM_ODP_TIMER_RESOL_DEF_NS;
168  tmr_attr->resparam.res_hz = 0;
169  tmr_attr->resparam.clk_src = EM_TIMER_CLKSRC_DEFAULT;
170  tmr_attr->flags = EM_TIMER_FLAG_DEFAULT;
171 
172  odp_timer_clk_src_t odp_clksrc;
173  odp_timer_capability_t odp_capa;
174  odp_timer_res_capability_t odp_res_capa;
175  int err;
176 
177  err = timer_clksrc_em2odp(tmr_attr->resparam.clk_src, &odp_clksrc);
178  if (unlikely(err)) {
179  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_TIMER_ATTR_INIT,
180  "Unsupported EM-timer clock source:%d",
181  tmr_attr->resparam.clk_src);
182  return;
183  }
184  err = odp_timer_capability(odp_clksrc, &odp_capa);
185  if (unlikely(err)) {
186  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_ATTR_INIT,
187  "Timer capability: ret %d, odp-clksrc:%d",
188  err, odp_clksrc);
189  return;
190  }
191 
192  memset(&odp_res_capa, 0, sizeof(odp_timer_res_capability_t));
193  odp_res_capa.res_ns = tmr_attr->resparam.res_ns;
194  err = odp_timer_res_capability(odp_clksrc, &odp_res_capa);
195  if (unlikely(err)) {
196  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_ATTR_INIT,
197  "Timer res capability: ret %d, odp-clksrc:%d, res %lu",
198  err, odp_clksrc, tmr_attr->resparam.res_ns);
199  return;
200  }
201 
202  TMR_DBG_PRINT("res %lu -> ODP says min %lu, max %lu\n",
203  tmr_attr->resparam.res_ns, odp_res_capa.min_tmo,
204  odp_res_capa.max_tmo);
205 
206  tmr_attr->num_tmo = EM_ODP_DEFAULT_TMOS;
207  if (odp_capa.max_timers && odp_capa.max_timers < EM_ODP_DEFAULT_TMOS)
208  tmr_attr->num_tmo = odp_capa.max_timers;
209 
210  tmr_attr->resparam.min_tmo = odp_res_capa.min_tmo;
211  tmr_attr->resparam.max_tmo = odp_res_capa.max_tmo;
212  tmr_attr->name[0] = 0; /* timer_create will add default (no index available here) */
213  tmr_attr->__internal_check = EM_CHECK_INIT_CALLED;
214 }

216 em_status_t em_timer_capability(em_timer_capability_t *capa, em_timer_clksrc_t clk_src)
217 {
218  if (EM_CHECK_LEVEL > 0 && unlikely(capa == NULL)) {
219  EM_LOG(EM_LOG_DBG, "%s(): NULL capa ptr!\n", __func__);
220  return EM_ERR_BAD_POINTER;
221  }
222 
223  odp_timer_clk_src_t odp_clksrc;
224  odp_timer_capability_t odp_capa;
225 
226  if (unlikely(timer_clksrc_em2odp(clk_src, &odp_clksrc) ||
227  odp_timer_capability(odp_clksrc, &odp_capa))) {
228  EM_LOG(EM_LOG_DBG, "%s: Not supported clk_src %d\n", __func__, clk_src);
229  return EM_ERR_BAD_ARG;
230  }
231 
232  capa->max_timers = odp_capa.max_pools < EM_ODP_MAX_TIMERS ?
233  odp_capa.max_pools : EM_ODP_MAX_TIMERS;
234  capa->max_num_tmo = odp_capa.max_timers;
235  capa->max_res.clk_src = clk_src;
236  capa->max_res.res_ns = odp_capa.max_res.res_ns;
237  capa->max_res.res_hz = odp_capa.max_res.res_hz;
238  capa->max_res.min_tmo = odp_capa.max_res.min_tmo;
239  capa->max_res.max_tmo = odp_capa.max_res.max_tmo;
240  capa->max_tmo.clk_src = clk_src;
241  capa->max_tmo.res_ns = odp_capa.max_tmo.res_ns;
242  capa->max_tmo.res_hz = odp_capa.max_tmo.res_hz;
243  capa->max_tmo.min_tmo = odp_capa.max_tmo.min_tmo;
244  capa->max_tmo.max_tmo = odp_capa.max_tmo.max_tmo;
245  return EM_OK;
246 }

249 {
250  if (EM_CHECK_LEVEL > 0 && unlikely(res == NULL)) {
251  EM_LOG(EM_LOG_DBG, "%s: NULL ptr res\n", __func__);
252  return EM_ERR_BAD_POINTER;
253  }
254 
255  odp_timer_clk_src_t odp_clksrc;
256  odp_timer_res_capability_t odp_res_capa;

259  err = timer_clksrc_em2odp(clk_src, &odp_clksrc);
260  if (unlikely(err)) {

262  return EM_ERR_BAD_ARG;
263  }
264  memset(&odp_res_capa, 0, sizeof(odp_timer_res_capability_t));
265  odp_res_capa.res_ns = res->res_ns;
266  odp_res_capa.res_hz = res->res_hz;
267  odp_res_capa.max_tmo = res->max_tmo; /* ODP will check if both were set */
268  err = odp_timer_res_capability(odp_clksrc, &odp_res_capa);
269  if (unlikely(err)) {
270  EM_LOG(EM_LOG_DBG, "%s: ODP res_capability failed (ret %d)!\n", __func__, err);
271  return EM_ERR_BAD_ARG;
272  }
273  res->min_tmo = odp_res_capa.min_tmo;
274  res->max_tmo = odp_res_capa.max_tmo;
275  res->res_ns = odp_res_capa.res_ns;
276  res->res_hz = odp_res_capa.res_hz;
277  res->clk_src = clk_src;
278  return EM_OK;
279 }

281 em_timer_t em_timer_create(const em_timer_attr_t *tmr_attr)
282 {
283  if (EM_CHECK_LEVEL > 0) {
284  if (check_timer_attr(tmr_attr) == false)
285  return EM_TIMER_UNDEF;
286  }
287 
288  odp_timer_pool_param_t odp_tpool_param;
289  odp_timer_clk_src_t odp_clksrc;
290 
291  memset(&odp_tpool_param, 0, sizeof(odp_timer_pool_param_t));
292  odp_tpool_param.res_ns = tmr_attr->resparam.res_ns;
293  odp_tpool_param.res_hz = tmr_attr->resparam.res_hz;
294  odp_tpool_param.min_tmo = tmr_attr->resparam.min_tmo;
295  odp_tpool_param.max_tmo = tmr_attr->resparam.max_tmo;
296  odp_tpool_param.num_timers = tmr_attr->num_tmo;
297  odp_tpool_param.priv = tmr_attr->flags & EM_TIMER_FLAG_PRIVATE ? 1 : 0;
298  if (unlikely(timer_clksrc_em2odp(tmr_attr->resparam.clk_src, &odp_clksrc))) {
299  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_TIMER_CREATE,
300  "Unsupported EM-timer clock source:%d",
301  tmr_attr->resparam.clk_src);
302  return EM_TIMER_UNDEF;
303  }
304  odp_tpool_param.clk_src = odp_clksrc;
305 
306  /* check queue type support */
307  odp_timer_capability_t capa;
308 
309  if (unlikely(odp_timer_capability(odp_clksrc, &capa))) {
310  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_CREATE,
311  "ODP timer capa failed for clk:%d",
312  tmr_attr->resparam.clk_src);
313  return EM_TIMER_UNDEF;
314  }
315  if (unlikely(!capa.queue_type_sched)) { /* must support scheduled queues */
316  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_CREATE,
317  "ODP does not support scheduled q for clk:%d",
318  tmr_attr->resparam.clk_src);
319  return EM_TIMER_UNDEF;
320  }
321 
322  /* buffer pool for tmos */
323  odp_pool_param_t odp_pool_param;
324 
325  odp_pool_param_init(&odp_pool_param);
326  odp_pool_param.type = ODP_POOL_BUFFER;
327  odp_pool_param.buf.size = sizeof(em_timer_timeout_t);
328  odp_pool_param.buf.align = ODP_CACHE_LINE_SIZE;
329  if (odp_pool_param.buf.cache_size > EM_ODP_TIMER_CACHE)
330  odp_pool_param.buf.cache_size = EM_ODP_TIMER_CACHE;
331  TMR_DBG_PRINT("local tmo pool cache %d\n", odp_pool_param.buf.cache_size);
332 
333  /* local pool caching may cause out of buffers situation on a core. Adjust,
334  * but not waste too much memory
335  */
336  uint32_t num = tmr_attr->num_tmo + ((em_core_count() - 1) * odp_pool_param.buf.cache_size);
337 
338  if (tmr_attr->num_tmo < num) {
339  TMR_DBG_PRINT("Adjusted pool size %d->%d due to local caching (%d)\n",
340  tmr_attr->num_tmo, num, odp_pool_param.buf.cache_size);
341  }
342  odp_pool_param.buf.num = num;
343 
344  /*
345  * Find a free timer-slot.
346  * This linear search should not be a performance problem with only a few timers
347  * available especially when these are typically created at startup.
348  */
349  int i;
350  event_timer_t *timer;
351 
352  odp_ticketlock_lock(&em_shm->timers.timer_lock);
353 
354  for (i = 0; i < EM_ODP_MAX_TIMERS; i++) {
355  timer = &em_shm->timers.timer[i];
356  if (timer->odp_tmr_pool != ODP_TIMER_POOL_INVALID) /* marks used entry */
357  continue;
358 
359  char timer_pool_name[ODP_TIMER_POOL_NAME_LEN];
360  char tmo_pool_name[ODP_POOL_NAME_LEN];
361  const char *name = tmr_attr->name;
362 
363  if (tmr_attr->name[0] == '\0') { /* replace NULL with default */
364  snprintf(timer_pool_name, ODP_TIMER_POOL_NAME_LEN,
365  "EM-timer-%d", timer->idx); /* idx initialized by timer_init */
366  name = timer_pool_name;
367  }
368 
369  TMR_DBG_PRINT("Creating ODP tmr pool: clk %d, res_ns %lu, res_hz %lu\n",
370  odp_tpool_param.clk_src, odp_tpool_param.res_ns,
371  odp_tpool_param.res_hz);
372  timer->odp_tmr_pool = odp_timer_pool_create(name, &odp_tpool_param);
373  if (unlikely(timer->odp_tmr_pool == ODP_TIMER_POOL_INVALID))
374  goto error_locked;
375  TMR_DBG_PRINT("Created timer: %s with idx: %d\n", name, timer->idx);
376 
377  snprintf(tmo_pool_name, ODP_POOL_NAME_LEN, "Tmo-pool-%d", timer->idx);
378  timer->tmo_pool = odp_pool_create(tmo_pool_name, &odp_pool_param);
379  if (unlikely(timer->tmo_pool == ODP_POOL_INVALID))
380  goto error_locked;
381  TMR_DBG_PRINT("Created ODP-pool: %s for %d timeouts\n",
382  tmo_pool_name, odp_pool_param.buf.num);
383 
384  timer->flags = tmr_attr->flags;
385  timer->plain_q_ok = capa.queue_type_plain;
386  odp_timer_pool_start();
387  break;
388  }
389 
390  odp_ticketlock_unlock(&em_shm->timers.timer_lock);
391 
392  if (unlikely(i >= EM_ODP_MAX_TIMERS)) {
393  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_TIMER_CREATE,
394  "No more timers available");
395  return EM_TIMER_UNDEF;
396  }
397  TMR_DBG_PRINT("ret %" PRI_TMR "\n", TMR_I2H(i));
398  return TMR_I2H(i);
399 
400 error_locked:
401  /* odp_ticketlock_lock(&timer_shm->tlock) */
402 
403  /* 'timer' set in loop */
404  if (timer->tmo_pool != ODP_POOL_INVALID)
405  odp_pool_destroy(timer->tmo_pool);
406  if (timer->odp_tmr_pool != ODP_TIMER_POOL_INVALID)
407  odp_timer_pool_destroy(timer->odp_tmr_pool);
408  timer->tmo_pool = ODP_POOL_INVALID;
409  timer->odp_tmr_pool = ODP_TIMER_POOL_INVALID;
410  odp_ticketlock_unlock(&em_shm->timers.timer_lock);
411 
412  TMR_DBG_PRINT("ERR odp tmr pool in: clk %u, res %lu, min %lu, max %lu, num %u\n",
413  odp_tpool_param.clk_src, odp_tpool_param.res_ns,
414  odp_tpool_param.min_tmo, odp_tpool_param.max_tmo, odp_tpool_param.num_timers);
415  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_CREATE, "Timer pool create failed");
416  return EM_TIMER_UNDEF;
417 }

419 em_status_t em_timer_delete(em_timer_t tmr)
420 {
421  timer_storage_t *const tmrs = &em_shm->timers;
422  int i = TMR_H2I(tmr);
423 
424  odp_ticketlock_lock(&tmrs->timer_lock);
425  /* take lock before checking so nothing can change */
426  if (unlikely(!is_timer_valid(tmr))) {
427  odp_ticketlock_unlock(&tmrs->timer_lock);
428  return INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TIMER_DELETE,
429  "Invalid timer:%" PRI_TMR "", tmr);
430  }
431 
432  odp_pool_destroy(tmrs->timer[i].tmo_pool);
433  tmrs->timer[i].tmo_pool = ODP_POOL_INVALID;
434  odp_timer_pool_destroy(tmrs->timer[i].odp_tmr_pool);
435  tmrs->timer[i].odp_tmr_pool = ODP_TIMER_POOL_INVALID;
436 
437  odp_ticketlock_unlock(&tmrs->timer_lock);
438  return EM_OK;
439 }

452 em_tmo_t em_tmo_create(em_timer_t tmr, em_tmo_flag_t flags, em_queue_t queue)
453 {
454  int i = TMR_H2I(tmr);
455  odp_timer_pool_t odptmr;
456  const queue_elem_t *const q_elem = queue_elem_get(queue);
457  odp_buffer_t tmo_buf;
458 
459  if (EM_CHECK_LEVEL > 0) {
460  if (unlikely(!is_timer_valid(tmr))) {
461  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TMO_CREATE,
462  "Invalid timer:%" PRI_TMR "", tmr);
463  return EM_TMO_UNDEF;
464  }
465  if (unlikely(q_elem == NULL || !queue_allocated(q_elem))) {
466  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TMO_CREATE,
467  "Tmr:%" PRI_TMR ": inv.Q:%" PRI_QUEUE "",
468  tmr, queue);
469  return EM_TMO_UNDEF;
470  }
471  if (unlikely(!is_queue_valid_type(tmr, q_elem))) {
472  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_TMO_CREATE,
473  "Tmr:%" PRI_TMR ": inv.Q (type):%" PRI_QUEUE "",
474  tmr, queue);
475  return EM_TMO_UNDEF;
476  }
477  if (unlikely(!check_tmo_flags(flags))) {
478  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_TMO_CREATE,
479  "Tmr:%" PRI_TMR ": inv. tmo-flags:0x%x",
480  tmr, flags);
481  return EM_TMO_UNDEF;
482  }
483  }
484 
485  tmo_buf = odp_buffer_alloc(em_shm->timers.timer[i].tmo_pool);
486  if (unlikely(tmo_buf == ODP_BUFFER_INVALID)) {
487  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_TMO_CREATE,
488  "Tmr:%" PRI_TMR ": tmo pool exhausted", tmr);

771 em_status_t em_tmo_ack(em_tmo_t tmo, em_event_t next_tmo_ev)
772 {
773  if (EM_CHECK_LEVEL > 0) {
774  RETURN_ERROR_IF(tmo == EM_TMO_UNDEF ||
775  next_tmo_ev == EM_EVENT_UNDEF,
776  EM_ERR_BAD_ID, EM_ESCOPE_TMO_ACK,
777  "Inv.args: tmo:%" PRI_TMO " ev:%" PRI_EVENT "",
778  tmo, next_tmo_ev);
779  RETURN_ERROR_IF(!(tmo->flags & EM_TMO_FLAG_PERIODIC),
780  EM_ERR_BAD_CONTEXT, EM_ESCOPE_TMO_ACK,
781  "Tmo ACK: Not a periodic tmo");
782  }
783 
784  if (EM_TIMER_TMO_STATS)
785  tmo->stats.num_acks++;
786 
787  em_tmo_state_t tmo_state = odp_atomic_load_acq_u32(&tmo->state);
788  /*
789  * If tmo cancelled:
790  * Return an error so the application can free the given event.
791  */
792  if (tmo_state == EM_TMO_STATE_IDLE) /* canceled, no errorhandler */
793  return EM_ERR_CANCELED;
794 
795  RETURN_ERROR_IF(tmo_state != EM_TMO_STATE_ACTIVE,
796  EM_ERR_BAD_STATE, EM_ESCOPE_TMO_ACK,
797  "Tmo ACK: invalid tmo state:%d", tmo_state);
798 
799  if (EM_CHECK_LEVEL > 1) {
800  RETURN_ERROR_IF(!odp_buffer_is_valid(tmo->odp_buffer),
801  EM_ERR_BAD_ID, EM_ESCOPE_TMO_ACK,
802  "Tmo ACK: invalid tmo buffer");
803  }
804 
805  event_hdr_t *ev_hdr = NULL;
806  odp_event_t odp_ev = event_em2odp(next_tmo_ev);
807  bool esv_ena = esv_enabled();
808 
809  if (esv_ena) {
810  ev_hdr = event_to_hdr(next_tmo_ev);
811  evstate_usr2em(next_tmo_ev, ev_hdr, EVSTATE__TMO_ACK);
812  }
813 
814  /*
815  * The periodic timer will silently stop if ack fails! Attempt to
816  * handle exceptions and if the tmo cannot be renewed, call
817  * the errorhandler so the application may recover.
818  */
819  tmo->last_tick += tmo->period; /* maintain absolute time */
820  int ret;
821  int tries = EM_TIMER_ACK_TRIES;
822  em_status_t err;
823 
824  /* try to set tmo EM_TIMER_ACK_TRIES times */
825  do {
826  /* ask new timeout for next period */
827  ret = odp_timer_set_abs(tmo->odp_timer, tmo->last_tick, &odp_ev);
828  /*
829  * Calling ack() was delayed over next period if 'ret' is
830  * ODP_TIMER_TOOEARLY, i.e. now in past. Other errors
831  * should not happen, fatal for this tmo
832  */
833  if (likely(ret != ODP_TIMER_TOOEARLY)) {
834  if (ret != ODP_TIMER_SUCCESS) {
835  TMR_DBG_PRINT("ODP return %d\n"
836  "tmo tgt/tick now %lu/%lu\n",
837  ret, tmo->last_tick,
838  odp_timer_current_tick(tmo->odp_timer_pool));
839  }
840  break;
841  }
842 
843  /* ODP_TIMER_TOOEARLY: ack() delayed beyond next time slot */
844  if (EM_TIMER_TMO_STATS)
845  tmo->stats.num_late_ack++;
846  TMR_DBG_PRINT("late, tgt/now %lu/%lu\n", tmo->last_tick,
847  odp_timer_current_tick(tmo->odp_timer_pool));
848 
849  if (tmo->flags & EM_TMO_FLAG_NOSKIP) /* not allowed to skip, next immediately */
850  return handle_ack_noskip(next_tmo_ev, ev_hdr, tmo->queue);
851 
852  /* skip already passed periods */
853  handle_ack_skip(tmo);
854 
855  tries--;
856  if (unlikely(tries < 1)) {
857  err = INTERNAL_ERROR(EM_ERR_OPERATION_FAILED,
858  EM_ESCOPE_TMO_ACK,
859  "Tmo ACK: too many retries:%u",
860  EM_TIMER_ACK_TRIES);
861  goto ack_err;
862  }
863  } while (ret != ODP_TIMER_SUCCESS);
864 
865  if (unlikely(ret != ODP_TIMER_SUCCESS)) {
866  err = INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TMO_ACK,
867  "Tmo ACK: failed to renew tmo (odp ret %d)",
868  ret);
869  goto ack_err;
870  }
871  return EM_OK;
872 
873 ack_err:
874  if (esv_ena)
875  evstate_usr2em_revert(next_tmo_ev, ev_hdr, EVSTATE__TMO_ACK__FAIL);
876  return err;
877 }

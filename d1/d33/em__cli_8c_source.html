340 static int cli_shm_setup(void)
341 {
342  if (cli_shm != NULL) {
343  EM_LOG(EM_LOG_ERR, "EM CLI shared memory ptr already set!\n");
344  return -1;
345  }
346 
347  /*
348  * Reserve the CLI shared memory once at start-up.
349  */
350  uint32_t flags = 0;
351 
352 #if ODP_VERSION_API_NUM(1, 33, 0) < ODP_VERSION_API
353  odp_shm_capability_t shm_capa;
354  int ret = odp_shm_capability(&shm_capa);
355 
356  if (ret) {
357  EM_LOG(EM_LOG_ERR, "shm capability error:%d\n", ret);
358  return -1;
359  }
360 
361  /* No huge pages needed for the CLI shm */
362  if (shm_capa.flags & ODP_SHM_NO_HP)
363  flags |= ODP_SHM_NO_HP;
364 #endif
365  odp_shm_t shm = odp_shm_reserve("em_cli", sizeof(em_cli_shm_t),
366  ODP_CACHE_LINE_SIZE, flags);
367 
368  if (shm == ODP_SHM_INVALID) {
369  EM_LOG(EM_LOG_ERR, "EM CLI shared memory reservation failed!\n");
370  return -1;
371  }
372 
373  cli_shm = odp_shm_addr(shm);
374 
375  if (cli_shm == NULL) {
376  EM_LOG(EM_LOG_ERR, "EM CLI shared memory ptr NULL!\n");
377  return -1;
378  }
379 
380  memset(cli_shm, 0, sizeof(em_cli_shm_t));
381 
382  /* Store shm handle, can be used in stop_em_cli() to free the memory */
383  cli_shm->this_shm = shm;
384 
385  return 0;
386 }

388 static int cli_shm_lookup(void)
389 {
390  odp_shm_t shm;
391  em_cli_shm_t *shm_addr;
392 
393  /* Lookup the EM shared memory on each EM-core */
394  shm = odp_shm_lookup("em_cli");
395  if (shm == ODP_SHM_INVALID) {
396  EM_LOG(EM_LOG_ERR, "Shared memory lookup failed!\n");
397  return -1;
398  }
399 
400  shm_addr = odp_shm_addr(shm);
401  if (!shm_addr) {
402  EM_LOG(EM_LOG_ERR, "Shared memory ptr NULL\n");
403  return -1;
404  }
405 
406  if (em_shm->conf.process_per_core && cli_shm == NULL)
407  cli_shm = shm_addr;
408 
409  if (shm_addr != cli_shm) {
410  EM_LOG(EM_LOG_ERR, "CLI shared memory init fails: cli_shm:%p != shm_addr:%p",
411  cli_shm, shm_addr);
412  return -1;
413  }
414 
415  return 0;
416 }

445 /**
446  * Run EM CLI server
447  *
448  * When executing this function, the CLI is accepting client connections and
449  * running commands from a client, if one is connected.
450  *
451  * @return EM_OK if successful.
452  */
453 static em_status_t run_em_cli(void)
454 {
455  /* Prepare CLI parameters */
456  odph_cli_param_t cli_param = {0};
457 
458  odph_cli_param_init(&cli_param);
459  cli_param.hostname = "EM-ODP";
460  cli_param.address = em_shm->opt.cli.ip_addr;
461  cli_param.port = em_shm->opt.cli.port;
462 
463  /* Initialize CLI helper */
464  if (odph_cli_init(&cli_param)) {
465  EM_LOG(EM_LOG_ERR, "Error: odph_cli_init() failed.\n");
466  return EM_ERR_LIB_FAILED;
467  }
468 
469  /* Register EM CLI commands */
470  if (cli_register_em_commands()) {
471  EM_LOG(EM_LOG_ERR, "Error: cli_register_em_commands() failed.\n");
472  return EM_ERR_LIB_FAILED;
473  }
474 
475  /* Create thread to run CLI server */
476  odp_cpumask_t cpumask;
477  odph_thread_common_param_t thr_common;
478  odph_thread_param_t thr_param;
479  odp_instance_t instance;
480 
481  if (odp_cpumask_default_control(&cpumask, 1) != 1) {
482  EM_LOG(EM_LOG_ERR, "Failed to get default CPU mask.\n");
483  return EM_ERR_LIB_FAILED;
484  }
485 
486  if (odp_instance(&instance)) {
487  EM_LOG(EM_LOG_ERR, "Failed to get odp instance.\n");
488  return EM_ERR_LIB_FAILED;
489  }
490 
491  odph_thread_common_param_init(&thr_common);
492  thr_common.instance = instance;
493  thr_common.cpumask = &cpumask;
494  thr_common.thread_model = 0; /* 0: Use pthread for the CLI */
495 
496  odph_thread_param_init(&thr_param);
497  thr_param.thr_type = ODP_THREAD_CONTROL;
498  thr_param.start = cli_thr_fn;
499  thr_param.arg = NULL;
500 
501  /* Set up EM CLI shared memory */
502  if (cli_shm_setup()) {
503  EM_LOG(EM_LOG_ERR, "Error: cli_shm_setup() failed.\n");
504  return EM_ERR_ALLOC_FAILED;
505  }
506 
507  EM_PRINT("Starting CLI server on %s:%d\n", cli_param.address, cli_param.port);
508 
509  /** Create EM CLI server thread and store the thread ID to be used in
510  * stop_em_cli() to wait for the thread to exit.
511  */
512  if (odph_thread_create(&cli_shm->em_cli_thread, &thr_common,
513  &thr_param, 1) != 1) {
514  EM_LOG(EM_LOG_ERR, "Failed to create CLI server thread.\n");
515  cli_shm_free();
516  return -1;
517  }
518 
519  return EM_OK;
520 }

522 /**
523  * Stop EM CLI server
524  *
525  * Stop accepting new client connections and disconnect any connected client.
526  *
527  * @return EM_OK if successful.
528  */
529 static em_status_t stop_em_cli(void)
530 {
531  if (odph_cli_stop()) {
532  EM_LOG(EM_LOG_ERR, "Failed to stop CLI.\n");
533  goto error;
534  }
535 
536  if (odph_thread_join(&cli_shm->em_cli_thread, 1) != 1) {
537  EM_LOG(EM_LOG_ERR, "Failed to join server thread.\n");
538  goto error;
539  }
540 
541  if (odph_cli_term()) {
542  EM_LOG(EM_LOG_ERR, "Failed to terminate CLI.\n");
543  goto error;
544  }
545 
546  cli_shm_free();
547  EM_PRINT("\nCLI server terminated!\n");
548 
549  return EM_OK;
550 
551 error:
552  cli_shm_free();
553  return EM_ERR_LIB_FAILED;
554 }

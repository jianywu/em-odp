43 void em_pool_cfg_init(em_pool_cfg_t *const pool_cfg)
44 {
45  odp_pool_param_t odp_pool_defaults;
46  uint32_t buf_cache_sz;
47  uint32_t pkt_cache_sz;
48  uint32_t cache_sz;
49 
50  if (unlikely(!pool_cfg)) {
51  INTERNAL_ERROR(EM_FATAL(EM_ERR_BAD_POINTER),
52  EM_ESCOPE_POOL_CFG_INIT,
53  "pool_cfg pointer NULL!");
54  return;
55  }
56 
57  odp_pool_param_init(&odp_pool_defaults);
58  memset(pool_cfg, 0, sizeof(*pool_cfg));
59 
60  pool_cfg->event_type = EM_EVENT_TYPE_UNDEF;
61 
62  buf_cache_sz = odp_pool_defaults.buf.cache_size;
63  pkt_cache_sz = odp_pool_defaults.pkt.cache_size;
64  cache_sz = MIN(buf_cache_sz, pkt_cache_sz);
65 
66  for (int i = 0; i < EM_MAX_SUBPOOLS; i++)
67  pool_cfg->subpool[i].cache_size = cache_sz;
68 
69  pool_cfg->__internal_check = EM_CHECK_INIT_CALLED;
70 }

74 {

85  return EM_POOL_UNDEF;

87 
88  pool_created = pool_create(name, pool, pool_cfg);
89 
90  if (unlikely(pool_created == EM_POOL_UNDEF ||
91  (pool != EM_POOL_UNDEF && pool != pool_created))) {
92  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_POOL_CREATE,
93  "Pool create failed,\n"
94  "requested:%" PRI_POOL " created:%" PRI_POOL "",
95  pool, pool_created);
96  return EM_POOL_UNDEF;
97  }

123 size_t
124 em_pool_get_name(em_pool_t pool, char *name /*out*/, size_t maxlen)
125 {
126  const mpool_elem_t *mpool_elem = pool_elem_get(pool);
127  size_t len = 0;
128 
129  if (unlikely(name == NULL || maxlen == 0)) {
130  INTERNAL_ERROR(EM_ERR_BAD_POINTER, EM_ESCOPE_POOL_GET_NAME,
131  "Invalid args: name=0x%" PRIx64 ", maxlen=%zu",
132  name, maxlen);
133  return 0;
134  }
135 
136  if (unlikely(mpool_elem == NULL || !pool_allocated(mpool_elem))) {
137  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_POOL_GET_NAME,
138  "Invalid Pool:%" PRI_POOL "", pool);
139  name[0] = '\0';
140  return 0;
141  }
142 
143  len = strnlen(mpool_elem->name, sizeof(mpool_elem->name) - 1);
144  if (maxlen - 1 < len)
145  len = maxlen - 1;
146 
147  memcpy(name, mpool_elem->name, len);
148  name[len] = '\0';
149 
150  return len;
151 }

153 em_pool_t
154 em_pool_get_first(unsigned int *num)
155 {
156  const mpool_elem_t *const mpool_elem_tbl = em_shm->mpool_tbl.pool;
157  const mpool_elem_t *mpool_elem = &mpool_elem_tbl[0];
158  const unsigned int pool_cnt = pool_count();
159 
160  _pool_tbl_iter_idx = 0; /* reset iteration */
161 
162  if (num)
163  *num = pool_cnt;
164 
165  if (pool_cnt == 0) {
166  _pool_tbl_iter_idx = EM_CONFIG_POOLS; /* UNDEF = _get_next() */
167  return EM_POOL_UNDEF;
168  }
169 
170  /* find first */
171  while (!pool_allocated(mpool_elem)) {
172  _pool_tbl_iter_idx++;
173  if (_pool_tbl_iter_idx >= EM_CONFIG_POOLS)
174  return EM_POOL_UNDEF;
175  mpool_elem = &mpool_elem_tbl[_pool_tbl_iter_idx];
176  }
177 
178  return pool_idx2hdl(_pool_tbl_iter_idx);
179 }

181 em_pool_t
182 em_pool_get_next(void)
183 {
184  if (_pool_tbl_iter_idx >= EM_CONFIG_POOLS - 1)
185  return EM_POOL_UNDEF;
186 
187  _pool_tbl_iter_idx++;
188 
189  const mpool_elem_t *const mpool_elem_tbl = em_shm->mpool_tbl.pool;
190  const mpool_elem_t *mpool_elem = &mpool_elem_tbl[_pool_tbl_iter_idx];
191 
192  /* find next */
193  while (!pool_allocated(mpool_elem)) {
194  _pool_tbl_iter_idx++;
195  if (_pool_tbl_iter_idx >= EM_CONFIG_POOLS)
196  return EM_POOL_UNDEF;
197  mpool_elem = &mpool_elem_tbl[_pool_tbl_iter_idx];
198  }
199 
200  return pool_idx2hdl(_pool_tbl_iter_idx);
201 }

203 em_status_t
204 em_pool_info(em_pool_t pool, em_pool_info_t *pool_info /*out*/)
205 {
206  const mpool_elem_t *pool_elem;
207 
208  RETURN_ERROR_IF(pool_info == NULL,
209  EM_ERR_BAD_POINTER, EM_ESCOPE_POOL_INFO,
210  "arg 'pool_info' invalid");
211 
212  pool_elem = pool_elem_get(pool);
213  RETURN_ERROR_IF(pool_elem == NULL || !pool_allocated(pool_elem),
214  EM_ERR_BAD_ID, EM_ESCOPE_POOL_INFO,
215  "EM-pool:%" PRI_POOL " invalid", pool);
216 
217  memset(pool_info, 0, sizeof(*pool_info));
218  /* copy pool info into the user provided 'pool_info' */
219  strncpy(pool_info->name, pool_elem->name, sizeof(pool_info->name));
220  pool_info->name[sizeof(pool_info->name) - 1] = '\0';
221  pool_info->em_pool = pool_elem->em_pool;
222  pool_info->event_type = pool_elem->event_type;
223  pool_info->align_offset = pool_elem->align_offset;
224  pool_info->user_area_size = pool_elem->user_area.req_size;
225  pool_info->num_subpools = pool_elem->num_subpools;
226 
227  for (int i = 0; i < pool_elem->num_subpools; i++) {
228  pool_info->subpool[i].size = pool_elem->size[i]; /*sorted sz*/
229  pool_info->subpool[i].num = pool_elem->pool_cfg.subpool[i].num;
230  pool_info->subpool[i].cache_size = pool_elem->pool_cfg.subpool[i].cache_size;
231  }
232 
233  /*
234  * EM pool usage statistics only collected if
235  * EM config file: pool.statistics_enable=true.
236  */
237  if (!em_shm->opt.pool.statistics_enable)
238  return EM_OK; /* no statistics, return */
239 
240  /* EM pool usage statistics _enabled_ - collect it: */
241  const int cores = em_core_count();
242  const int pool_idx = pool_hdl2idx(pool);
243  mpool_statistics_t pool_stat[cores];
244 
245  /* copy pool-statistics from all cores and work on a local snapshot */
246  memcpy(pool_stat, em_shm->mpool_tbl.pool_stat_core,
247  sizeof(pool_stat));
248 
249  for (int i = 0; i < pool_elem->num_subpools; i++) {
250  const uint64_t num = pool_elem->pool_cfg.subpool[i].num;
251  uint64_t used = 0;
252  uint64_t free = 0;
253  uint64_t alloc_sum = 0;
254  uint64_t free_sum = 0;
255 
256  for (int j = 0; j < cores; j++) {
257  free_sum += pool_stat[j].stat[pool_idx][i].free;
258  alloc_sum += pool_stat[j].stat[pool_idx][i].alloc;
259  }
260 
261  used = alloc_sum - free_sum;
262  if (unlikely(free_sum > alloc_sum)) {
263  /*
264  * free-increments seen by this core before
265  * alloc increments or unlikely wrap-around.
266  */
267  uint64_t diff = free_sum - alloc_sum;
268 
269  if (diff <= num) /* counts close so set '0' */
270  used = 0;
271  else /* wrap, should not happen very soon... */
272  used = UINT64_MAX - diff + 1;
273  }
274  /* Sanity check */
275  if (used > num)
276  used = num;
277 
278  free = num - used;
279 
280  pool_info->subpool[i].used = used;
281  pool_info->subpool[i].free = free;
282  }
283 
284  return EM_OK;
285 }

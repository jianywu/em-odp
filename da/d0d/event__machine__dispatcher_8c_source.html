33 /*
34  * em_dispatch() helper: dispatch and call the user provided callback functions
35  * 'input_poll' and 'output_drain'
36  */
37 static inline uint64_t
38 dispatch_with_userfn(uint64_t rounds, bool do_input_poll, bool do_output_drain)
39 {
40  const bool do_forever = rounds == 0 ? true : false;
41  const em_input_poll_func_t input_poll =
42  em_shm->conf.input.input_poll_fn;
43  const em_output_drain_func_t output_drain =
44  em_shm->conf.output.output_drain_fn;
45  int rx_events = 0;
46  int dispatched_events;
47  int round_events;
48  uint64_t events = 0;
49  uint64_t i;
50 
51  for (i = 0; do_forever || i < rounds;) {
52  dispatched_events = 0;
53  if (do_input_poll)
54  rx_events = input_poll();
55 
56  do {
57  round_events = dispatch_round();
58  dispatched_events += round_events;
59  i++; /* inc rounds */
60  } while (dispatched_events < rx_events &&
61  round_events > 0 && (do_forever || i < rounds));
62 
63  events += dispatched_events; /* inc ret value*/
64  if (do_output_drain)
65  (void)output_drain();
66  }
67 
68  return events;
69 }

71 /*
72  * em_dispatch() helper: dispatch without calling any user provided callbacks
73  */
74 static inline uint64_t
75 dispatch_no_userfn(uint64_t rounds)
76 {
77  const bool do_forever = rounds == 0 ? true : false;
78  uint64_t events = 0;
79  uint64_t i;
80 
81  if (do_forever) {
82  for (;/*ever*/;)
83  dispatch_round();
84  } else {
85  for (i = 0; i < rounds; i++)
86  events += dispatch_round();
87  }
88 
89  return events;
90 }

92 uint64_t
93 em_dispatch(uint64_t rounds /* 0 = forever */)
94 {
95  uint64_t events = 0;
96  int round_events;
97 
98  const em_locm_t *const locm = &em_locm;
99  const bool do_input_poll = locm->do_input_poll;
100  const bool do_output_drain = locm->do_output_drain;
101 
102  odp_schedule_resume();
103 
104  if (do_input_poll || do_output_drain)
105  events = dispatch_with_userfn(rounds, do_input_poll,
106  do_output_drain);
107  else
108  events = dispatch_no_userfn(rounds);
109 
110  /* pause scheduling before exiting the dispatch loop */
111  odp_schedule_pause();
112  /* empty the locally pre-scheduled events (if any) */
113  do {
114  round_events = dispatch_round();
115  events += round_events;
116  } while (round_events > 0);
117 
118  return events;
119 }

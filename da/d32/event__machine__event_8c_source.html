33 em_event_t
34 em_alloc(size_t size, em_event_type_t type, em_pool_t pool)
35 {
36  mpool_elem_t *pool_elem;
37 
38  pool_elem = pool_elem_get(pool);
39  if (unlikely(size == 0 ||
40  pool_elem == NULL || !pool_allocated(pool_elem))) {
41  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_ALLOC,
42  "Invalid args: size:%zu type:%u pool:%" PRI_POOL "",
43  size, type, pool);
44  return EM_EVENT_UNDEF;
45  }
46 
47  /*
48  * EM event pools created with type=SW can not support pkt events.
49  */
50  if (unlikely(pool_elem->event_type == EM_EVENT_TYPE_SW &&
51  em_get_type_major(type) == EM_EVENT_TYPE_PACKET)) {
52  INTERNAL_ERROR(EM_ERR_NOT_IMPLEMENTED, EM_ESCOPE_ALLOC,
53  "EM-pool:%s(%" PRI_POOL "):\n"
54  "Invalid event type:0x%" PRIx32 " for buf",
55  pool_elem->name, pool_elem->em_pool, type);
56  return EM_EVENT_UNDEF;
57  }
58 
59  event_hdr_t *ev_hdr = event_alloc(pool_elem, size, type);
60 
61  if (unlikely(!ev_hdr)) {
62  em_status_t err =
63  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_ALLOC,
64  "EM-pool:'%s': sz:%zu type:0x%x pool:%" PRI_POOL "",
65  pool_elem->name, size, type, pool);
66  if (EM_CHECK_LEVEL > 1 && err != EM_OK &&
67  em_shm->opt.pool.statistics_enable) {
68  em_pool_info_print(pool);
69  }
70  return EM_EVENT_UNDEF;
71  }
72 
73  em_event_t event = ev_hdr->event;
74 
75  /* Update event ESV state for alloc */
76  if (esv_enabled())
77  event = evstate_alloc(event, ev_hdr);
78 
79  if (EM_API_HOOKS_ENABLE)
80  call_api_hooks_alloc(&event, 1, 1, size, type, pool);
81 
82  return event;
83 }

85 int
86 em_alloc_multi(em_event_t events[/*out*/], int num,
87  size_t size, em_event_type_t type, em_pool_t pool)
88 {
89  if (unlikely(num <= 0)) {
90  if (num < 0)
91  INTERNAL_ERROR(EM_ERR_TOO_SMALL, EM_ESCOPE_ALLOC_MULTI,
92  "Invalid arg: num:%d", num);
93  return 0;
94  }
95 
96  mpool_elem_t *const pool_elem = pool_elem_get(pool);
97  int ret;
98 
99  if (unlikely(size == 0 ||
100  pool_elem == NULL || !pool_allocated(pool_elem))) {
101  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_ALLOC_MULTI,
102  "Invalid args: size:%zu type:%u pool:%" PRI_POOL "",
103  size, type, pool);
104  return 0;
105  }
106 
107  if (pool_elem->event_type == EM_EVENT_TYPE_PACKET) {
108  /*
109  * EM event pools created with type=PKT can support SW events
110  * as well as pkt events.
111  */
112  ret = event_alloc_pkt_multi(events, num, pool_elem, size, type);
113  } else { /* pool_elem->event_type == EM_EVENT_TYPE_SW */
114  /*
115  * EM event pools created with type=SW can not support
116  * pkt events.
117  */
118  if (unlikely(em_get_type_major(type) == EM_EVENT_TYPE_PACKET)) {
119  INTERNAL_ERROR(EM_ERR_NOT_IMPLEMENTED, EM_ESCOPE_ALLOC_MULTI,
120  "EM-pool:%s(%" PRI_POOL "): Invalid event type:%u for buf",
121  pool_elem->name, pool, type);
122  return 0;
123  }
124 
125  ret = event_alloc_buf_multi(events, num, pool_elem, size, type);
126  }
127 
128  if (unlikely(ret != num)) {
129  em_status_t err =
130  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_ALLOC_MULTI,
131  "Requested num:%d events, allocated:%d\n"
132  "EM-pool:'%s': sz:%zu type:0x%x pool:%" PRI_POOL "",
133  num, ret,
134  pool_elem->name, size, type, pool);
135  if (EM_CHECK_LEVEL > 1 && err != EM_OK &&
136  em_shm->opt.pool.statistics_enable) {
137  em_pool_info_print(pool);
138  }
139  }
140 
141  if (EM_API_HOOKS_ENABLE)
142  call_api_hooks_alloc(events, ret, num, size, type, pool);
143 
144  return ret;
145 }

147 void
148 em_free(em_event_t event)
149 {
150  bool esv_ena = esv_enabled();
151  odp_event_t odp_event;
152 
153  if (unlikely(event == EM_EVENT_UNDEF)) {
154  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_FREE,
155  "event undefined!");
156  return;
157  }
158 
159  if (EM_API_HOOKS_ENABLE)
160  call_api_hooks_free(&event, 1);
161 
162  if (esv_ena || em_shm->opt.pool.statistics_enable) {
163  event_hdr_t *const ev_hdr = event_to_hdr(event);
164 
165  if (esv_ena)
166  evstate_free(event, ev_hdr, EVSTATE__FREE);
167  if (em_shm->opt.pool.statistics_enable)
168  poolstat_dec_evhdr(ev_hdr);
169  }
170 
171  odp_event = event_em2odp(event);
172  odp_event_free(odp_event);
173 }

175 void em_free_multi(const em_event_t events[], int num)
176 {
177  if (unlikely(!events || num < 0)) {
178  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_FREE_MULTI,
179  "Inv.args: events[]:%p num:%d", events, num);
180  return;
181  }
182  if (unlikely(num == 0))
183  return;
184 
185  if (EM_CHECK_LEVEL > 1) {
186  int i;
187 
188  for (i = 0; i < num && events[i] != EM_EVENT_UNDEF; i++)
189  ;
190  if (unlikely(i != num)) {
191  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_FREE_MULTI,
192  "events[%d] undefined!", i);
193  return;
194  }
195  }
196 
197  if (EM_API_HOOKS_ENABLE)
198  call_api_hooks_free(events, num);
199 
200  bool esv_ena = esv_enabled();
201  odp_event_t odp_events[num];
202 
203  if (esv_ena || em_shm->opt.pool.statistics_enable) {
204  event_hdr_t *ev_hdrs[num];
205 
206  event_to_hdr_multi(events, ev_hdrs, num);
207  if (esv_ena)
208  evstate_free_multi(events, ev_hdrs, num,
209  EVSTATE__FREE_MULTI);
210  if (em_shm->opt.pool.statistics_enable)
211  poolstat_dec_evhdr_multi(ev_hdrs, num);
212  }
213 
214  events_em2odp(events, odp_events/*out*/, num);
215  odp_event_free_multi(odp_events, num);
216 }

218 em_status_t
219 em_send(em_event_t event, em_queue_t queue)
220 {
221  const bool is_external = queue_external(queue);
222  queue_elem_t *q_elem = NULL;
223  event_hdr_t *ev_hdr;
224  int num_sent;
225  em_status_t stat;
226 
227  /*
228  * Check all args.
229  */
230  RETURN_ERROR_IF(EM_CHECK_LEVEL > 0 && event == EM_EVENT_UNDEF,
231  EM_ERR_BAD_ID, EM_ESCOPE_SEND, "Invalid event");
232 
233  ev_hdr = event_to_hdr(event);
234  ev_hdr->egrp = EM_EVENT_GROUP_UNDEF;
235 
236  if (!is_external) {
237  /* queue belongs to this EM instance */
238  q_elem = queue_elem_get(queue);
239  RETURN_ERROR_IF(EM_CHECK_LEVEL > 0 && !q_elem,
240  EM_ERR_BAD_ID, EM_ESCOPE_SEND,
241  "Invalid queue:%" PRI_QUEUE "", queue);
242  RETURN_ERROR_IF(EM_CHECK_LEVEL > 1 && !queue_allocated(q_elem),
243  EM_ERR_BAD_STATE, EM_ESCOPE_SEND,
244  "Invalid queue:%" PRI_QUEUE "", queue);
245  }
246 
247  /* Buffer events from EO-start sent to scheduled queues */
248  if (unlikely(!is_external &&
249  q_elem->scheduled && em_locm.start_eo_elem)) {
250  /*
251  * em_send() called from within an EO-start function:
252  * all events sent to scheduled queues will be buffered
253  * and sent when the EO-start operation completes.
254  */
255  num_sent = eo_start_buffer_events(&event, 1, queue,
256  EM_EVENT_GROUP_UNDEF);
257  stat = num_sent == 1 ? EM_OK : EM_ERR_OPERATION_FAILED;
258  if (EM_CHECK_LEVEL == 0)
259  return stat;
260  RETURN_ERROR_IF(stat != EM_OK, stat, EM_ESCOPE_SEND,
261  "send from EO-start failed");
262  return EM_OK;
263  }
264 
265  if (EM_API_HOOKS_ENABLE)
266  call_api_hooks_send(&event, 1, queue, EM_EVENT_GROUP_UNDEF);
267 
268  if (esv_enabled())
269  evstate_usr2em(event, ev_hdr, EVSTATE__SEND);
270 
271  if (is_external) {
272  /*
273  * Send out of EM to another device via event-chaining and a
274  * user-provided function 'event_send_device()'
275  */
276  stat = send_chaining(event, ev_hdr, queue);
277  if (EM_CHECK_LEVEL == 0)
278  return stat;
279  if (unlikely(stat != EM_OK)) {
280  stat = INTERNAL_ERROR(stat, EM_ESCOPE_SEND,
281  "send_chaining failed: Q:%" PRI_QUEUE "",
282  queue);
283  goto send_err;
284  }
285  return EM_OK;
286  }
287 
288  /*
289  * Normal send to a queue on this device
290  */
291  switch (q_elem->type) {
292  case EM_QUEUE_TYPE_ATOMIC:
293  case EM_QUEUE_TYPE_PARALLEL:
294  case EM_QUEUE_TYPE_PARALLEL_ORDERED:
295  stat = send_event(event, q_elem);
296  break;
297  case EM_QUEUE_TYPE_UNSCHEDULED:
298  stat = queue_unsched_enqueue(event, q_elem);
299  break;
300  case EM_QUEUE_TYPE_LOCAL:
301  stat = send_local(event, ev_hdr, q_elem);
302  break;
303  case EM_QUEUE_TYPE_OUTPUT:
304  stat = send_output(event, ev_hdr, q_elem);
305  break;
306  default:
307  stat = EM_ERR_NOT_FOUND;
308  break;
309  }
310 
311  if (EM_CHECK_LEVEL == 0)
312  return stat;
313 
314  if (unlikely(stat != EM_OK)) {
315  stat =
316  INTERNAL_ERROR(stat, EM_ESCOPE_SEND,
317  "send failed: Q:%" PRI_QUEUE " type:%" PRI_QTYPE "",
318  queue, q_elem->type);
319  goto send_err;
320  }
321 
322  return EM_OK;
323 
324 send_err:
325  if (esv_enabled())
326  evstate_usr2em_revert(event, ev_hdr, EVSTATE__SEND__FAIL);
327  return stat;
328 }

330 /*
331  * em_send_group_multi() helper: check function arguments
332  */
333 static inline em_status_t
334 send_multi_check_args(const em_event_t events[], int num, em_queue_t queue,
335  bool *is_external__out /*out if EM_OK*/,
336  queue_elem_t **q_elem__out /*out if EM_OK*/)
337 {
338  const bool is_external = queue_external(queue);
339  queue_elem_t *q_elem = NULL;
340  int i;
341 
342  if (EM_CHECK_LEVEL > 0 && unlikely(!events || num <= 0))
343  return EM_ERR_BAD_ARG;
344 
345  if (EM_CHECK_LEVEL > 2) {
346  for (i = 0; i < num && events[i] != EM_EVENT_UNDEF; i++)
347  ;
348  if (unlikely(i != num))
349  return EM_ERR_BAD_POINTER;
350  }
351 
352  if (!is_external) {
353  /* queue belongs to this EM instance */
354  q_elem = queue_elem_get(queue);
355 
356  if (EM_CHECK_LEVEL > 0 && unlikely(!q_elem))
357  return EM_ERR_BAD_ARG;
358  if (EM_CHECK_LEVEL > 1 && unlikely(!queue_allocated(q_elem)))
359  return EM_ERR_BAD_STATE;
360  }
361 
362  *is_external__out = is_external;
363  *q_elem__out = q_elem; /* NULL if is_external */
364  return EM_OK;
365 }

367 int
368 em_send_multi(const em_event_t events[], int num, em_queue_t queue)
369 {
370  bool is_external = false; /* set by check_args */
371  queue_elem_t *q_elem = NULL; /* set by check_args */
372  int num_sent;
373  int i;
374 
375  /*
376  * Check all args.
377  */
378  em_status_t err =
379  send_multi_check_args(events, num, queue,
380  /*out if EM_OK:*/ &is_external, &q_elem);
381  if (unlikely(err != EM_OK)) {
382  INTERNAL_ERROR(err, EM_ESCOPE_SEND_MULTI,
383  "Invalid args: events:%p num:%d Q:%" PRI_QUEUE "",
384  events, num, queue);
385  return 0;
386  }
387 
388  /* Buffer events from EO-start sent to scheduled queues */
389  if (unlikely(!is_external &&
390  q_elem->scheduled && em_locm.start_eo_elem)) {
391  /*
392  * em_send_multi() called from within an EO-start function:
393  * all events sent to scheduled queues will be buffered
394  * and sent when the EO-start operation completes.
395  */
396  num_sent = eo_start_buffer_events(events, num, queue,
397  EM_EVENT_GROUP_UNDEF);
398  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num))
399  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_SEND_MULTI,
400  "send-multi EO-start: req:%d, sent:%d",
401  num, num_sent);
402  return num_sent;
403  }
404 
405  event_hdr_t *ev_hdrs[num];
406 
407  event_to_hdr_multi(events, ev_hdrs, num);
408  for (i = 0; i < num; i++)
409  ev_hdrs[i]->egrp = EM_EVENT_GROUP_UNDEF;
410 
411  if (EM_API_HOOKS_ENABLE)
412  call_api_hooks_send(events, num, queue, EM_EVENT_GROUP_UNDEF);
413 
414  if (esv_enabled())
415  evstate_usr2em_multi(events, ev_hdrs, num, EVSTATE__SEND_MULTI);
416 
417  if (is_external) {
418  /*
419  * Send out of EM to another device via event-chaining and a
420  * user-provided function 'event_send_device_multi()'
421  */
422  num_sent = send_chaining_multi(events, ev_hdrs, num, queue);
423  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num)) {
424  INTERNAL_ERROR(EM_ERR_OPERATION_FAILED,
425  EM_ESCOPE_SEND_MULTI,
426  "send_chaining_multi: req:%d, sent:%d",
427  num, num_sent);
428  goto send_multi_err;
429  }
430  return num_sent;
431  }
432 
433  /*
434  * Normal send to a queue on this device
435  */
436  switch (q_elem->type) {
437  case EM_QUEUE_TYPE_ATOMIC:
438  case EM_QUEUE_TYPE_PARALLEL:
439  case EM_QUEUE_TYPE_PARALLEL_ORDERED:
440  num_sent = send_event_multi(events, num, q_elem);
441  break;
442  case EM_QUEUE_TYPE_UNSCHEDULED:
443  num_sent = queue_unsched_enqueue_multi(events, num, q_elem);
444  break;
445  case EM_QUEUE_TYPE_LOCAL:
446  num_sent = send_local_multi(events, ev_hdrs, num, q_elem);
447  break;
448  case EM_QUEUE_TYPE_OUTPUT:
449  num_sent = send_output_multi(events, ev_hdrs, num, q_elem);
450  break;
451  default:
452  num_sent = 0;
453  break;
454  }
455 
456  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num)) {
457  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_SEND_MULTI,
458  "send-multi failed: req:%d, sent:%d",
459  num, num_sent);
460  goto send_multi_err;
461  }
462 
463  return num_sent;
464 
465 send_multi_err:
466  if (esv_enabled()) {
467  evstate_usr2em_revert_multi(&events[num_sent], &ev_hdrs[num_sent],
468  num - num_sent,
469  EVSTATE__SEND_MULTI__FAIL);
470  }
471  return num_sent;
472 }

492 size_t
493 em_event_get_size(em_event_t event)
494 {
495  odp_event_t odp_event;
496  odp_event_type_t odp_etype;
497 
498  if (unlikely(event == EM_EVENT_UNDEF)) {
499  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_EVENT_GET_SIZE,
500  "event undefined!");
501  return 0;
502  }
503 
504  odp_event = event_em2odp(event);
505  odp_etype = odp_event_type(odp_event);
506 
507  if (odp_etype == ODP_EVENT_PACKET) {
508  odp_packet_t odp_pkt = odp_packet_from_event(odp_event);
509 
510  return odp_packet_seg_len(odp_pkt);
511  } else if (odp_etype == ODP_EVENT_BUFFER) {
512  const event_hdr_t *ev_hdr = event_to_hdr(event);
513 
514  return ev_hdr->event_size;
515  }
516 
517  INTERNAL_ERROR(EM_ERR_NOT_FOUND, EM_ESCOPE_EVENT_GET_SIZE,
518  "Unexpected odp event type:%u", odp_etype);
519  return 0;
520 }

522 em_pool_t em_event_get_pool(em_event_t event)
523 {
524  if (unlikely(event == EM_EVENT_UNDEF)) {
525  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_EVENT_GET_POOL,
526  "event undefined!");
527  return EM_POOL_UNDEF;
528  }
529 
530  odp_event_t odp_event = event_em2odp(event);
531  odp_event_type_t type = odp_event_type(odp_event);
532  odp_pool_t odp_pool = ODP_POOL_INVALID;
533 
534  if (type == ODP_EVENT_PACKET) {
535  odp_packet_t pkt = odp_packet_from_event(odp_event);
536 
537  odp_pool = odp_packet_pool(pkt);
538  } else if (type == ODP_EVENT_BUFFER) {
539  odp_buffer_t buf = odp_buffer_from_event(odp_event);
540 
541  odp_pool = odp_buffer_pool(buf);
542  }
543 
544  if (unlikely(odp_pool == ODP_POOL_INVALID))
545  return EM_POOL_UNDEF;
546 
547  em_pool_t pool = pool_odp2em(odp_pool);
548 
549  /*
550  * Don't report an error if 'pool == EM_POOL_UNDEF' since that might
551  * happen if the event is input from pktio that is using external
552  * (to EM) odp pools.
553  */
554  return pool;
555 }

838 em_event_t em_event_clone(em_event_t event, em_pool_t pool/*or EM_POOL_UNDEF*/)
839 {
840  const mpool_elem_t *pool_elem = pool_elem_get(pool);
841 
842  /* Check all args */
843  if (EM_CHECK_LEVEL >= 1 &&
844  unlikely(event == EM_EVENT_UNDEF ||
845  (pool != EM_POOL_UNDEF &&
846  (pool_elem == NULL || !pool_allocated(pool_elem))))) {
847  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_EVENT_CLONE,
848  "Inv.args: event:%" PRI_EVENT " pool:%" PRI_POOL "",
849  event, pool);
850  return EM_EVENT_UNDEF;
851  }
852 
853  odp_event_t odp_event = event_em2odp(event);
854  odp_event_type_t odp_evtype = odp_event_type(odp_event);
855  odp_pool_t odp_pool = ODP_POOL_INVALID;
856  odp_packet_t pkt = ODP_PACKET_INVALID;
857  odp_buffer_t buf = ODP_BUFFER_INVALID;
858 
859  if (unlikely(odp_evtype != ODP_EVENT_PACKET &&
860  odp_evtype != ODP_EVENT_BUFFER)) {
861  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_EVENT_CLONE,
862  "Inv. odp-event-type:%d", odp_evtype);
863  return EM_EVENT_UNDEF;
864  }
865 
866  /* Obtain the event-hdr, event-size and the pool to use */
867  const event_hdr_t *ev_hdr;
868  size_t size;
869  em_event_type_t type;
870  em_pool_t em_pool = pool;
871  em_event_t clone_event; /* return value */
872 
873  if (odp_evtype == ODP_EVENT_PACKET) {
874  pkt = odp_packet_from_event(odp_event);
875  ev_hdr = odp_packet_user_area(pkt);
876  size = odp_packet_seg_len(pkt);
877  if (pool == EM_POOL_UNDEF) {
878  odp_pool = odp_packet_pool(pkt);
879  em_pool = pool_odp2em(odp_pool);
880  }
881  } else /* ODP_EVENT_BUFFER */ {
882  buf = odp_buffer_from_event(odp_event);
883  ev_hdr = odp_buffer_addr(buf);
884  size = ev_hdr->event_size;
885  if (pool == EM_POOL_UNDEF) {
886  odp_pool = odp_buffer_pool(buf);
887  em_pool = pool_odp2em(odp_pool);
888  }
889  }
890 
891  /* No EM-pool found */
892  if (em_pool == EM_POOL_UNDEF) {
893  if (unlikely(odp_evtype == ODP_EVENT_BUFFER)) {
894  INTERNAL_ERROR(EM_ERR_NOT_FOUND, EM_ESCOPE_EVENT_CLONE,
895  "No suitable event-pool found");
896  return EM_EVENT_UNDEF;
897  }
898  /* odp_evtype == ODP_EVENT_PACKET:
899  * Not an EM-pool, e.g. event from external pktio odp-pool.
900  * Allocate and clone pkt via ODP directly.
901  */
902  clone_event = pkt_clone_odp(pkt, odp_pool);
903  if (unlikely(clone_event == EM_EVENT_UNDEF)) {
904  INTERNAL_ERROR(EM_ERR_OPERATION_FAILED, EM_ESCOPE_EVENT_CLONE,
905  "Cloning from ext odp-pool:%" PRIu64 " failed",
906  odp_pool_to_u64(odp_pool));
907  }
908  return clone_event;
909  }
910 
911  /*
912  * Clone the event from an EM-pool:
913  */
914  pool_elem = pool_elem_get(em_pool);
915  type = ev_hdr->event_type;
916 
917  /* EM event pools created with type=SW can not support pkt events */
918  if (unlikely(pool_elem->event_type == EM_EVENT_TYPE_SW &&
919  em_get_type_major(type) == EM_EVENT_TYPE_PACKET)) {
920  INTERNAL_ERROR(EM_ERR_NOT_IMPLEMENTED, EM_ESCOPE_EVENT_CLONE,
921  "EM-pool:%s(%" PRI_POOL "):\n"
922  "Invalid event type:0x%" PRIx32 " for buf",
923  pool_elem->name, em_pool, type);
924  return EM_EVENT_UNDEF;
925  }
926 
927  event_hdr_t *clone_hdr = event_alloc(pool_elem, size, type);
928 
929  if (unlikely(!clone_hdr)) {
930  em_status_t err =
931  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_EVENT_CLONE,
932  "EM-pool:'%s': sz:%zu type:0x%x pool:%" PRI_POOL "",
933  pool_elem->name, size, type, em_pool);
934  if (EM_CHECK_LEVEL > 1 && err != EM_OK &&
935  em_shm->opt.pool.statistics_enable)
936  em_pool_info_print(em_pool);
937  return EM_EVENT_UNDEF;
938  }
939 
940  clone_event = clone_hdr->event;
941  /* Update clone_event ESV state for the clone-alloc */
942  if (esv_enabled())
943  clone_event = evstate_clone(clone_event, clone_hdr);
944 
945  /* Call the 'alloc' API hook function also for event-clone */
946  if (EM_API_HOOKS_ENABLE)
947  call_api_hooks_alloc(&clone_event, 1, 1, size, type, pool);
948 
949  /* Copy event payload from the parent event into the clone event */
950  const void *src = event_pointer(event);
951  void *dst = event_pointer(clone_event);
952 
953  memcpy(dst, src, size);
954 
955  return clone_event;

958 static int event_uarea_init(em_event_t event, event_hdr_t **ev_hdr/*out*/)
959 {
960  odp_event_t odp_event = event_em2odp(event);
961  odp_event_type_t odp_evtype = odp_event_type(odp_event);
962  odp_pool_t odp_pool = ODP_POOL_INVALID;
963  odp_packet_t odp_pkt;
964  odp_buffer_t odp_buf;
965  event_hdr_t *hdr;
966  bool is_init;
967 
968  switch (odp_evtype) {
969  case ODP_EVENT_PACKET:
970  odp_pkt = odp_packet_from_event(odp_event);
971  hdr = odp_packet_user_area(odp_pkt);
972  is_init = hdr->user_area.isinit;
973  if (!is_init)
974  odp_pool = odp_packet_pool(odp_pkt);
975  break;
976  case ODP_EVENT_BUFFER:
977  odp_buf = odp_buffer_from_event(odp_event);
978  hdr = odp_buffer_addr(odp_buf);
979  is_init = hdr->user_area.isinit;
980  if (!is_init)
981  odp_pool = odp_buffer_pool(odp_buf);
982  break;
983  default:
984  return -1;
985  }
986 
987  *ev_hdr = hdr;
988 
989  if (!is_init) {
990  /*
991  * Event user area metadata is not initialized in
992  * the event header - initialize it:
993  */
994  hdr->user_area.all = 0; /* user_area.{} = all zero (.sizes=0) */
995  hdr->user_area.isinit = 1;
996 
997  em_pool_t pool = pool_odp2em(odp_pool);
998 
999  if (pool == EM_POOL_UNDEF)
1000  return 0; /* ext ODP pool: OK, no user area, sz=0 */
1001 
1002  /* Event from an EM event pool, can init event user area */
1003  const mpool_elem_t *pool_elem = pool_elem_get(pool);
1004 
1005  if (unlikely(!pool_elem))
1006  return -2; /* invalid pool_elem */
1007 
1008  hdr->user_area.req_size = pool_elem->user_area.req_size;
1009  hdr->user_area.pad_size = pool_elem->user_area.pad_size;
1010  }
1011 
1012  return 0;
1013 }

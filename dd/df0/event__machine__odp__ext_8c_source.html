101 int em_odp_pool2odp(em_pool_t pool, odp_pool_t odp_pools[/*out*/], int num)
102 {
103  mpool_elem_t *pool_elem = pool_elem_get(pool);
104 
105  if (unlikely(!pool_elem || !pool_allocated(pool_elem) ||
106  !odp_pools || num <= 0)) {
107  INTERNAL_ERROR(EM_ERR_BAD_ARG, EM_ESCOPE_ODP_EXT,
108  "Inv.args: pool:%" PRI_POOL " odp_pools:%p num:%d",
109  pool, odp_pools, num);
110  return 0;
111  }
112 
113  int num_subpools = MIN(num, pool_elem->num_subpools);
114  int i;
115 
116  for (i = 0; i < num_subpools; i++)
117  odp_pools[i] = pool_elem->odp_pool[i];
118 
119  /* return the number of odp-pools filled into 'odp_pools[]' */
120  return num_subpools;
121 }

134 int pkt_enqueue(const odp_packet_t pkt_tbl[/*num*/], int num, em_queue_t queue)
135 {
136  if (unlikely(!pkt_tbl || num <= 0))
137  return 0;
138 
139  queue_elem_t *const q_elem = queue_elem_get(queue);
140  odp_event_t odp_event_tbl[num];
141  int sent = 0;
142 
143  odp_packet_to_event_multi(pkt_tbl, odp_event_tbl/*out*/, num);
144 
145  if (q_elem != NULL && q_elem->scheduled) {
146  /*
147  * Enqueue the events into a scheduled em-odp queue.
148  * No need to init the ev-hdrs - init is done in dispatch.
149  */
150  sent = odp_queue_enq_multi(q_elem->odp_queue,
151  odp_event_tbl, num);
152  } else {
153  em_event_t event_tbl[num];
154  event_hdr_t *evhdr_tbl[num];
155 
156  events_odp2em(odp_event_tbl, event_tbl/*out*/, num);
157 
158  /* Init the event-hdrs for incoming non-scheduled pkts */
159  event_init_pkt_multi(pkt_tbl, event_tbl/*in/out*/,
160  evhdr_tbl/*out*/, num, true /*is_extev*/);
161 
162  if (q_elem == NULL) {
163  /* Send directly out via event chaining */
164  if (likely(queue_external(queue)))
165  sent = send_chaining_multi(event_tbl, evhdr_tbl,
166  num, queue);
167  } else if (q_elem->type == EM_QUEUE_TYPE_UNSCHEDULED) {
168  /* Enqueue into an unscheduled em-odp queue */
169  sent = odp_queue_enq_multi(q_elem->odp_queue,
170  odp_event_tbl, num);
171  } else if (q_elem->type == EM_QUEUE_TYPE_LOCAL) {
172  /* Send into an local em-odp queue */
173  sent = send_local_multi(event_tbl, evhdr_tbl,
174  num, q_elem);
175  } else if (q_elem->type == EM_QUEUE_TYPE_OUTPUT) {
176  /* Send directly out via an output em-odp queue */
177  sent = send_output_multi(event_tbl, evhdr_tbl,
178  num, q_elem);
179  }
180  }
181 
182  if (unlikely(sent < num)) {
183  sent = unlikely(sent < 0) ? 0 : sent;
184  odp_packet_free_multi(&pkt_tbl[sent], num - sent);
185  /*
186  * Event state checking: No need to adjust the event state
187  * since the events were never enqueued into EM.
188  */
189  }
190 
191  return sent;
192 }

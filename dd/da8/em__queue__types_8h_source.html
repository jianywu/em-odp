86 /**
87  * Internal represenation of the EM queue handle
88  * The EM queue handle contains a 16-bit queue-id and a 16-bit device-id.
89  */
90 typedef union {
91  em_queue_t queue;
92  struct {
93 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
94  uint16_t queue_id;
95  uint16_t device_id;
96 #ifdef EM_64_BIT
97  uint32_t unused;
98 #endif
99 #elif __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
100 #ifdef EM_64_BIT
101  uint32_t unused;
102 #endif
103  uint16_t device_id;
104  uint16_t queue_id;
105 #endif
106  };
107 } internal_queue_t;
108 
109 /* Assert that all EM queues can fit into the .queue_id field */
110 COMPILE_TIME_ASSERT(UINT16_MAX >= EM_MAX_QUEUES,
111  INTERNAL_QUEUE_ID_MAX_ERROR);
112 /* Verify size of struct, i.e. accept no padding */
113 COMPILE_TIME_ASSERT(sizeof(internal_queue_t) == sizeof(em_queue_t),
114  INTERNAL_QUEUE_T_SIZE_ERROR);

116 /*
117  * Queue state
118  */
119 typedef enum queue_state {
120  /** Invalid queue state, queue not created/allocated */
121  EM_QUEUE_STATE_INVALID = 0,
122 
123  /*
124  * Scheduled queue (ATOMIC, PARALLEL, ORDERED) states:
125  * (keep state values consecutive: ...n-1,n,n+1...)
126  */
127  /** Queue initialization, allocated and being set up */
128  EM_QUEUE_STATE_INIT = 1,
129  /** Queue added/bound to an EO, but EO-start not yet complete */
130  EM_QUEUE_STATE_BIND = 2,
131  /** Queue ready, related EO started */
132  EM_QUEUE_STATE_READY = 3,
133 
134  /*
135  * Non-scheduled queue (UNSCHED, LOCAL, OUTPUT) state use the
136  * UNSCHEDULED-state.
137  */
138  /* Use separete value for unscheduled queues to catch illegal usage */
139  EM_QUEUE_STATE_UNSCHEDULED = 999
140 } queue_state_t;

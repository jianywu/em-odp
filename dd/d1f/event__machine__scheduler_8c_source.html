33 void
34 em_atomic_processing_end(void)
35 {
36  em_locm_t *const locm = &em_locm;
37 
38  if (locm->current.sched_context_type != EM_SCHED_CONTEXT_TYPE_ATOMIC)
39  return;
40 
41  if (locm->event_burst_cnt != 0)
42  return;
43 
44  const queue_elem_t *q_elem = locm->current.sched_q_elem;
45 
46  /*
47  * Do nothing for non-atomic queues or if the func has already
48  * been called.
49  */
50  if (unlikely(q_elem == NULL ||
51  q_elem->type != EM_QUEUE_TYPE_ATOMIC))
52  return;
53 
54  if (q_elem->atomic_group == EM_ATOMIC_GROUP_UNDEF)
55  odp_schedule_release_atomic();
56  else
57  atomic_group_release();
58 
59  /*
60  * Mark that em_atomic_processing_end() has been called
61  * for the current queue.
62  */
63  locm->current.sched_context_type = EM_SCHED_CONTEXT_TYPE_NONE;
64 }

66 void
67 em_ordered_processing_end(void)
68 {
69  em_locm_t *const locm = &em_locm;
70 
71  if (locm->event_burst_cnt != 0)
72  return;
73 
74  const queue_elem_t *q_elem = locm->current.q_elem;
75  em_queue_type_t qtype;
76 
77  if (unlikely(q_elem == NULL))
78  return;
79 
80  qtype = em_queue_get_type(q_elem->queue);
81  if (unlikely(qtype != EM_QUEUE_TYPE_PARALLEL_ORDERED))
82  return;
83 
84  odp_schedule_release_ordered();
85  /*
86  * ODP might not actually release the ordered context here. From an EM
87  * point of view the context needs to be ended since the ODP result is
88  * unknown.
89  */
90  locm->current.sched_context_type = EM_SCHED_CONTEXT_TYPE_NONE;
91 }

99 em_sched_context_type_t
100 em_sched_context_type_current(em_queue_t *queue)
101 {
102  const em_locm_t *const locm = &em_locm;
103 
104  if (locm->current.sched_q_elem == NULL) {
105  if (queue != NULL)
106  *queue = EM_QUEUE_UNDEF;
107  return EM_SCHED_CONTEXT_TYPE_NONE;
108  }
109 
110  if (queue != NULL)
111  *queue = locm->current.sched_q_elem->queue;
112 
113  return locm->current.sched_context_type;
114 }

54 /**
55  * Helper: Remove undef-event entries from ev_tbl[]
56  */
57 static inline int pack_ev_tbl(em_event_t ev_tbl[/*in,out*/], const int num)
58 {
59  if (num == 1) {
60  if (ev_tbl[0] != EM_EVENT_UNDEF)
61  return 1;
62  else
63  return 0;
64  }
65 
66  int pack = 0;
67 
68  for (int i = 0; i < num; i++) {
69  if (ev_tbl[i] != EM_EVENT_UNDEF) {
70  if (pack < i)
71  ev_tbl[pack] = ev_tbl[i];
72  pack++;
73  }
74  }
75 
76  return pack;
77 }

79 /**
80  * Run all dispatch enter-callback functions.
81  *
82  * @note Neither EO-receive nor any further enter-callbacks will be called if
83  * all events have been dropped by the callbacks already run, i.e.
84  * no callback or EO-receive will be called with 'num=0'.
85  *
86  * @param eo EO handle
87  * @param eo_ctx EO context data
88  * @param[in,out] ev_tbl Event table
89  * @param num_events Number of events in the event table
90  * @param queue Queue from which this event came from
91  * @param q_ctx Queue context data
92  *
93  * @return The number of events in ev_tbl[] after all dispatch enter callbacks
94  */
95 static inline int
96 dispatch_enter_cb(em_eo_t eo, void **eo_ctx,
97  em_event_t ev_tbl[/*in,out*/], const int num_events,
98  em_queue_t *queue, void **q_ctx)
99 {
100  const hook_tbl_t *cb_tbl = em_shm->dispatch_enter_cb_tbl;
101  em_dispatch_enter_func_t dispatch_enter_fn;
102  int num = num_events;
103 
104  for (int i = 0; i < EM_CALLBACKS_MAX && num > 0; i++) {
105  dispatch_enter_fn = cb_tbl->tbl[i].disp_enter;
106  if (dispatch_enter_fn == NULL)
107  break;
108  dispatch_enter_fn(eo, eo_ctx, ev_tbl, num, queue, q_ctx);
109  num = pack_ev_tbl(ev_tbl, num);
110  }
111 
112  return num;
113 }

135 static inline void
136 call_eo_receive_fn(const em_eo_t eo, const em_receive_func_t eo_receive_func,
137  event_hdr_t *ev_hdr, queue_elem_t *const q_elem)
138 {
139  em_locm_t *const locm = &em_locm;
140  em_queue_t queue = q_elem->queue;
141  void *queue_ctx = q_elem->context;
142  void *eo_ctx = q_elem->eo_ctx;
143  em_event_t event = event_hdr_to_event(ev_hdr);
144  int num = 1;
145 
146  locm->current.q_elem = q_elem;
147  locm->current.rcv_multi_cnt = 1;
148  /* Check and set core local event group (before dispatch callback(s)) */
149  event_group_set_local(ev_hdr->egrp, ev_hdr->egrp_gen, 1);
150 
151  if (esv_enabled())
152  event = evstate_em2usr(event, ev_hdr, EVSTATE__DISPATCH);
153 
154  if (EM_DISPATCH_CALLBACKS_ENABLE) {
155  em_event_t ev_tbl[1] = {event};
156 
157  num = dispatch_enter_cb(eo, &eo_ctx, ev_tbl/*in,out*/, 1,
158  &queue, &queue_ctx);
159  if (num && ev_tbl[0] != event) {
160  /* user-callback changed event: update event & hdr */
161  event = ev_tbl[0];
162  ev_hdr = event_to_hdr(event);
163  }
164  }
165 
166  if (likely(num == 1)) {
167  em_event_type_t event_type = ev_hdr->event_type;
168  /*
169  * Call the EO receive function
170  * (only if the dispatch callback(s) did not free the event)
171  */
172  eo_receive_func(eo_ctx, event, event_type,
173  queue, queue_ctx);
174  }
175 
176  if (EM_DISPATCH_CALLBACKS_ENABLE)
177  dispatch_exit_cb(eo);
178 
179  /*
180  * Event belongs to an event_group, update the count and
181  * if requested send notifications
182  */
183  if (locm->current.egrp != EM_EVENT_GROUP_UNDEF) {
184  /*
185  * Atomically decrease the event group count.
186  * If the new count is zero, send notification events.
187  */
188  event_group_count_decrement(1);
189  }
190  locm->current.egrp = EM_EVENT_GROUP_UNDEF;
191 }

193 /**
194  * @note All events belong to the same event group
195  * @note Event type dropped from multi-event receive - use em_event_get_type()
196  */
197 static inline void
198 call_eo_receive_multi_fn(const em_eo_t eo,
199  const em_receive_multi_func_t eo_receive_multi_func,
200  em_event_t ev_tbl[], event_hdr_t *ev_hdr_tbl[],
201  const int num_events, queue_elem_t *const q_elem)
202 {
203  em_locm_t *const locm = &em_locm;
204  em_queue_t queue = q_elem->queue;
205  void *queue_ctx = q_elem->context;
206  void *eo_ctx = q_elem->eo_ctx;
207  int num = num_events;
208 
209  locm->current.q_elem = q_elem;
210  locm->current.rcv_multi_cnt = num_events;
211  /* Check and set core local event group (before dispatch callback(s)) */
212  event_group_set_local(ev_hdr_tbl[0]->egrp, ev_hdr_tbl[0]->egrp_gen,
213  num_events);
214 
215  if (esv_enabled())
216  evstate_em2usr_multi(ev_tbl, ev_hdr_tbl, num_events,
217  EVSTATE__DISPATCH_MULTI);
218 
219  if (EM_DISPATCH_CALLBACKS_ENABLE)
220  num = dispatch_enter_cb(eo, &eo_ctx,
221  ev_tbl/*in,out*/, num_events,
222  &queue, &queue_ctx);
223  if (likely(num > 0)) {
224  /*
225  * Call the EO multi-event receive function
226  * (only if the dispatch callback(s) did not free all events)
227  */
228  eo_receive_multi_func(eo_ctx, ev_tbl, num, queue, queue_ctx);
229  }
230 
231  if (EM_DISPATCH_CALLBACKS_ENABLE)
232  dispatch_exit_cb(eo);
233 
234  /*
235  * Event belongs to an event_group, update the count and
236  * if requested send notifications
237  */
238  if (locm->current.egrp != EM_EVENT_GROUP_UNDEF) {
239  /*
240  * Atomically decrease the event group count.
241  * If the new count is zero, send notification events.
242  */
243  event_group_count_decrement(num_events);
244  }
245  locm->current.egrp = EM_EVENT_GROUP_UNDEF;
246 }

248 static inline void
249 dispatch_local_queues(em_event_t ev_tbl[], event_hdr_t *ev_hdr_tbl[],
250  const int num)
251 {
252  const bool esv_ena = esv_enabled();
253  queue_elem_t *q_elem;
254  int idx = 0; /* index into ev_tbl[] & ev_hdr_tbl[] */
255  int ev_cnt; /* number of events to the same local-queue */
256  int i;
257 
258  /* Loop through 'num' events and dispatch in batches to local queues */
259  do {
260  /* dst local queue */
261  q_elem = ev_hdr_tbl[idx]->q_elem;
262  /* count events sent to the same local queue */
263  for (i = idx + 1; i < num && q_elem == ev_hdr_tbl[i]->q_elem; i++)
264  ;
265  ev_cnt = i - idx; /* '1 to num' events */
266 
267  if (unlikely(q_elem == NULL || q_elem->state != EM_QUEUE_STATE_READY)) {
268  if (esv_ena)
269  evstate_em2usr_multi(&ev_tbl[idx],
270  &ev_hdr_tbl[idx], ev_cnt,
271  EVSTATE__DISPATCH_LOCAL__FAIL);
272  em_free_multi(&ev_tbl[idx], ev_cnt);
273  /* Consider removing the logging */
274  EM_LOG(EM_LOG_PRINT,
275  "EM info: %s(): localQ:%" PRI_QUEUE ":\n"
276  "Not ready - state:%d drop:%d events\n",
277  __func__, q_elem->queue,
278  q_elem->state, ev_cnt);
279  idx += ev_cnt;
280  continue;
281  }
282 
283  if (q_elem->use_multi_rcv)
284  dispatch_multi_receive(&ev_tbl[idx],
285  &ev_hdr_tbl[idx],
286  ev_cnt, q_elem, false);
287  else
288  dispatch_single_receive(&ev_hdr_tbl[idx],
289  ev_cnt, q_elem, false);
290  idx += ev_cnt;
291  } while (idx < num);
292 }

294 static inline void
295 check_local_queues(void)
296 {
297  em_event_t ev_tbl[EM_SCHED_MULTI_MAX_BURST];
298  event_hdr_t *ev_hdr_tbl[EM_SCHED_MULTI_MAX_BURST];
299  int num;
300 
301  if (em_locm.local_queues.empty)
302  return;
303 
304  /*
305  * Check if the previous EO receive function sent events to a
306  * local queue ('EM_QUEUE_TYPE_LOCAL') - and if so, dispatch
307  * those events immediately.
308  */
309  for (;;) {
310  num = next_local_queue_events(ev_tbl, EM_SCHED_MULTI_MAX_BURST);
311  if (num <= 0)
312  return;
313 
314  event_to_hdr_multi(ev_tbl, ev_hdr_tbl, num);
315 
316  dispatch_local_queues(ev_tbl, ev_hdr_tbl, num);
317  }
318 }

320 /**
321  * Count events (hdrs) sent/tagged with the same event group
322  */
323 static inline int
324 count_same_evgroup(event_hdr_t *ev_hdr_tbl[], const unsigned int num)
325 {
326  if (unlikely(num < 2))
327  return num;
328 
329  const em_event_group_t egrp = ev_hdr_tbl[0]->egrp;
330  unsigned int i = 1; /* 2nd hdr */
331 
332  if (EM_EVENT_GROUP_SAFE_MODE) {
333  const int32_t egrp_gen = ev_hdr_tbl[0]->egrp_gen;
334 
335  for (; i < num &&
336  egrp == ev_hdr_tbl[i]->egrp &&
337  egrp_gen == ev_hdr_tbl[i]->egrp_gen; i++)
338  ;
339  } else {
340  for (; i < num &&
341  egrp == ev_hdr_tbl[i]->egrp; i++)
342  ;
343  }
344 
345  return i;
346 }

348 static inline void
349 dispatch_multi_receive(em_event_t ev_tbl[], event_hdr_t *ev_hdr_tbl[],
350  const int num_events, queue_elem_t *const q_elem,
351  const bool check_local_qs)
352 {
353  const em_eo_t eo = q_elem->eo;
354  const em_receive_multi_func_t eo_rcv_multi_fn =
355  q_elem->receive_multi_func;
356  int idx = 0; /* index into ev_hdr_tbl[] */
357  int i;
358  int j;
359 
360  do {
361  /* count same event groups: 1 to num_events */
362  const int egrp_cnt = count_same_evgroup(&ev_hdr_tbl[idx],
363  num_events - idx);
364  const int max = q_elem->max_events;
365  const int num = MIN(egrp_cnt, max);
366  const int rounds = egrp_cnt / num;
367  const int left_over = egrp_cnt % num;
368 
369  if (check_local_qs) {
370  em_locm_t *const locm = &em_locm;
371 
372  j = idx;
373  for (i = 0; i < rounds; i++) {
374  locm->event_burst_cnt -= num;
375  call_eo_receive_multi_fn(eo, eo_rcv_multi_fn,
376  &ev_tbl[j],
377  &ev_hdr_tbl[j],
378  num, q_elem);
379  check_local_queues();
380  j += num;
381  }
382  if (left_over) {
383  locm->event_burst_cnt = 0;
384  call_eo_receive_multi_fn(eo, eo_rcv_multi_fn,
385  &ev_tbl[j],
386  &ev_hdr_tbl[j],
387  left_over, q_elem);
388  check_local_queues();
389  }
390  } else {
391  j = idx;
392  for (i = 0; i < rounds; i++) {
393  call_eo_receive_multi_fn(eo, eo_rcv_multi_fn,
394  &ev_tbl[j],
395  &ev_hdr_tbl[j],
396  num, q_elem);
397  j += num;
398  }
399  if (left_over) {
400  call_eo_receive_multi_fn(eo, eo_rcv_multi_fn,
401  &ev_tbl[j],
402  &ev_hdr_tbl[j],
403  left_over, q_elem);
404  }
405  }
406 
407  idx += egrp_cnt;
408  } while (idx < num_events);
409 }

411 static inline void
412 dispatch_single_receive(event_hdr_t *ev_hdr_tbl[], const int num_events,
413  queue_elem_t *const q_elem, const bool check_local_qs)
414 {
415  const em_eo_t eo = q_elem->eo;
416  const em_receive_func_t eo_rcv_fn = q_elem->receive_func;
417  int i;
418 
419  if (check_local_qs) {
420  for (i = 0; i < num_events; i++) {
421  em_locm.event_burst_cnt--;
422  call_eo_receive_fn(eo, eo_rcv_fn,
423  ev_hdr_tbl[i], q_elem);
424  check_local_queues();
425  }
426  } else {
427  for (i = 0; i < num_events; i++)
428  call_eo_receive_fn(eo, eo_rcv_fn,
429  ev_hdr_tbl[i], q_elem);
430  }
431 }

433 /**
434  * Dispatch events - call the EO-receive functions and pass the
435  * events for processing
436  */
437 static inline void
438 dispatch_events(em_event_t ev_tbl[], event_hdr_t *ev_hdr_tbl[],
439  const int num_events, queue_elem_t *const q_elem)
440 {
441  em_locm_t *const locm = &em_locm;
442  const em_queue_type_t q_type = q_elem->type;
443  em_sched_context_type_t sched_ctx_type = EM_SCHED_CONTEXT_TYPE_NONE;
444 
445  if (q_type == EM_QUEUE_TYPE_ATOMIC)
446  sched_ctx_type = EM_SCHED_CONTEXT_TYPE_ATOMIC;
447  else if (q_type == EM_QUEUE_TYPE_PARALLEL_ORDERED)
448  sched_ctx_type = EM_SCHED_CONTEXT_TYPE_ORDERED;
449 
450  locm->current.sched_context_type = sched_ctx_type;
451  locm->current.sched_q_elem = q_elem;
452  /* here: locm->current.egrp == EM_EVENT_GROUP_UNDEF */
453 
454  /*
455  * Call the Execution Object (EO) receive function.
456  * Scheduling context may be released during this.
457  */
458  if (q_elem->use_multi_rcv)
459  dispatch_multi_receive(ev_tbl, ev_hdr_tbl, num_events,
460  q_elem, true);
461  else
462  dispatch_single_receive(ev_hdr_tbl, num_events, q_elem, true);
463 
464  /*
465  * Check for buffered events sent to output queues during the previous
466  * dispatch rounds
467  */
468  if (!EM_OUTPUT_QUEUE_IMMEDIATE &&
469  locm->output_queue_track.idx_cnt > 0)
470  output_queue_buffering_drain();
471 
472  locm->current.q_elem = NULL;
473  locm->current.sched_q_elem = NULL;
474  locm->current.sched_context_type = EM_SCHED_CONTEXT_TYPE_NONE;
475 }

477 static inline void
478 dispatch_poll_ctrl_queue(void)
479 {
480  const unsigned int poll_interval = em_shm->opt.dispatch.poll_ctrl_interval;
481 
482  /*
483  * Rate limit how often this core checks the unsched ctrl queue.
484  */
485 
486  if (poll_interval > 1) {
487  em_locm_t *const locm = &em_locm;
488 
489  locm->dispatch_cnt--;
490  if (locm->dispatch_cnt > 0)
491  return;
492  locm->dispatch_cnt = poll_interval;
493 
494  odp_time_t now = odp_time_global();
495  odp_time_t period = odp_time_diff(now, locm->dispatch_last_run);
496  odp_time_t poll_period = em_shm->opt.dispatch.poll_ctrl_interval_time;
497 
498  if (odp_time_cmp(period, poll_period) < 0)
499  return;
500  locm->dispatch_last_run = now;
501  }
502 
503  /* Poll internal unscheduled ctrl queues */
504  poll_unsched_ctrl_queue();
505 }

507 /*
508  * Run a dispatch round - query the scheduler for events and dispatch
509  */
510 static inline int
511 dispatch_round(void)
512 {
513  odp_queue_t odp_queue;
514  odp_event_t odp_ev_tbl[EM_SCHED_MULTI_MAX_BURST];
515  event_hdr_t *ev_hdr_tbl[EM_SCHED_MULTI_MAX_BURST];
516  em_event_t ev_tbl[EM_SCHED_MULTI_MAX_BURST];
517  queue_elem_t *queue_elem;
518 
519  dispatch_poll_ctrl_queue();
520 
521  /*
522  * Schedule events to the core from queues
523  */
524  const int num_events =
525  odp_schedule_multi_no_wait(&odp_queue, odp_ev_tbl,
526  EM_SCHED_MULTI_MAX_BURST);
527  if (unlikely(num_events <= 0)) {
528  /*
529  * No scheduled events available, check if the local queues
530  * contain anything on this core - e.g. pktio or something
531  * outside the dispatch-context might have sent to a local queue
532  */
533  check_local_queues();
534  return 0;
535  }
536 
537  queue_elem = odp_queue_context(odp_queue);
538 
539  /* Events might originate from outside of EM and need init */
540  event_init_odp_multi(odp_ev_tbl, ev_tbl/*out*/, ev_hdr_tbl/*out*/,
541  num_events, true/*is_extev*/);
542 
543  if (unlikely(queue_elem == NULL ||
544  queue_elem->state != EM_QUEUE_STATE_READY)) {
545  if (esv_enabled())
546  evstate_em2usr_multi(ev_tbl/*in/out*/, ev_hdr_tbl, num_events,
547  EVSTATE__DISPATCH_SCHED__FAIL);
548  /* Drop all events dequeued from this queue */
549  em_free_multi(ev_tbl, num_events);
550 
551  if (queue_elem == NULL)
552  INTERNAL_ERROR(EM_ERR_BAD_POINTER, EM_ESCOPE_DISPATCH,
553  "Event(s) from non-EM Q, drop %d events",
554  num_events);
555  else
556  INTERNAL_ERROR(EM_ERR_BAD_STATE, EM_ESCOPE_DISPATCH,
557  "Q:%" PRI_QUEUE " not ready, state=%d\n"
558  " drop:%d event(s)\n",
559  queue_elem->queue, queue_elem->state,
560  num_events);
561  return 0;
562  }
563 
564  if (queue_elem->atomic_group == EM_ATOMIC_GROUP_UNDEF) {
565  em_locm.event_burst_cnt = num_events;
566  dispatch_events(ev_tbl, ev_hdr_tbl, num_events, queue_elem);
567  } else {
568  atomic_group_dispatch(ev_tbl, ev_hdr_tbl,
569  num_events, queue_elem);
570  }
571 
572  return num_events;
573 }

33 static int read_config_file(void)
34 {
35  const char *conf_str;
36  int val = 0;
37  int64_t val64 = 0;
38  int ret;
39 
40  /*
41  * Option: dispatch.poll_ctrl_interval
42  */
43  conf_str = "dispatch.poll_ctrl_interval";
44  ret = em_libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
45  if (unlikely(!ret)) {
46  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
47  return -1;
48  }
49 
50  if (val < 0) {
51  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %d'\n",
52  conf_str, val);
53  return -1;
54  }
55  /* store & print the value */
56  em_shm->opt.dispatch.poll_ctrl_interval = val;
57  EM_PRINT(" %s: %d\n", conf_str, val);
58 
59  /*
60  * Option: dispatch.poll_ctrl_interval_ns
61  */
62  conf_str = "dispatch.poll_ctrl_interval_ns";
63  ret = em_libconfig_lookup_int64(&em_shm->libconfig, conf_str, &val64);
64  if (unlikely(!ret)) {
65  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
66  return -1;
67  }
68 
69  if (val64 < 0) {
70  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %" PRId64 "'\n",
71  conf_str, val64);
72  return -1;
73  }
74  /* store & print the value */
75  em_shm->opt.dispatch.poll_ctrl_interval_ns = val64;
76  long double sec = (long double)val64 / 1000000000.0;
77 
78  EM_PRINT(" %s: %" PRId64 "ns (%Lfs)\n", conf_str, val64, sec);
79 
80  /* Store ns value as odp_time_t */
81  em_shm->opt.dispatch.poll_ctrl_interval_time = odp_time_global_from_ns(val64);
82 
83  return 0;
84 }

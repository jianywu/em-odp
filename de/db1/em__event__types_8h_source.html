47 /**
48  * @def PKT_USERPTR_MAGIC_NBR
49  *
50  * Magic number used to detect whether the EM event-header has been initialized
51  * by EM in events based on odp-pkt-buffers.
52  *
53  * Set the odp-pkt user-ptr to this magic number to be able to recognize
54  * pkt-events that EM has created vs. pkts from pkt-input that needs their
55  * ev-hdrs to be initialized before further EM processing.
56  *
57  * if (odp_packet_user_ptr(odp_pkt) != PKT_USERPTR_MAGIC_NBR) {
58  * // Pkt from outside of EM, need to init ev_hdr
59  * odp_packet_user_ptr_set(odp_pkt, PKT_USERPTR_MAGIC_NBR);
60  * init_ev_hdr('ev_hdr in the user-area of the odp-pkt');
61  * ...
62  * }
63  */
64 #define PKT_USERPTR_MAGIC_NBR ((void *)(intptr_t)0xA5A5)

66 /**
67  * Internal representation of the event handle (em_event_t) when using
68  * Event State Verification (ESV)
69  *
70  * An event-generation-count is encoded into the high bits of the event handle
71  * to catch illegal usage after the event ownership has been transferred.
72  * Each user-to-EM event state transition increments the .evgen and thus
73  * obsoletes any further use of the handle by that user.
74  */
75 typedef union {
76  em_event_t event;
77  struct {
78 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
79  uint64_t evptr : 48;
80  uint64_t evgen : 16;
81 #elif __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
82  uint64_t evgen : 16;
83  uint64_t evptr : 48;
84 #endif
85  };
86 } evhdl_t;
87 
88 COMPILE_TIME_ASSERT(sizeof(evhdl_t) == sizeof(em_event_t), EVHDL_T_SIZE_ERROR);
89 
90 /**
91  * Event-state counters: 'evgen', 'free_cnt' and 'send_cnt'.
92  *
93  * Updated as one single atomic var via 'evstate_cnt_t::u64'.
94  */
95 typedef union ODP_ALIGNED(sizeof(uint64_t)) {
96  uint64_t u64; /* updated atomically in the event-hdr */
97  struct {
98  uint16_t evgen;
99  uint16_t rsvd;
100  union {
101  struct {
102  uint16_t free_cnt;
103  uint16_t send_cnt;
104  };
105  uint32_t free_send_cnt;
106  };
107  };
108 } evstate_cnt_t;
109 
110 /* Verify size of struct, i.e. accept no padding */
111 COMPILE_TIME_ASSERT(sizeof(evstate_cnt_t) == sizeof(uint64_t),
112  EVSTATE_CNT_T_SIZE_ERROR);
113 
114 /**
115  * Event-state information.
116  * Not atomically updated (but evstate_cnt_t is updated atomically)
117  */
118 typedef struct {
119  /**
120  * Event state, updated on valid state trasitions.
121  * "Best effort" update, i.e. atomic update of state not
122  * guaranteed in invalid simultaneous state updates.
123  *
124  * Contains the previously known good state and will be
125  * printed when detecting an invalid state transition.
126  */
127  em_eo_t eo;
128  em_queue_t queue;
129  /**
130  * EM API operation ID.
131  * Identifies the previously called API func that altered state
132  */
133  uint16_t api_op;
134  /** EM core that called API('api_op') */
135  uint16_t core;
136  /**
137  * First 'word' of the event payload as seen
138  * at the time of the previous state update.
139  */
140  uint32_t payload_first;
141 } ev_hdr_state_t;
142 
143 /**
144  * Event header
145  *
146  * SW & I/O originated events.
147  */
148 typedef struct {
149  /**
150  * Event State Verification (ESV): event state data
151  */
152  union {
153  uint8_t u8[32];
154  struct {
155  /**
156  * Together the evstate_cnt_t counters (evgen, free_cnt
157  * and send_cnt) can be used to detect invalid states
158  * and operations on the event, e.g.:
159  * double-free, double-send, send-after-free,
160  * free-after-send, usage-after-output,
161  * usage-after-timer-tmo-set/ack/cancel/delete etc.
162  */
163  evstate_cnt_t state_cnt;
164 
165  /**
166  * Event state, updated on valid state trasitions.
167  * "Best effort" update, i.e. atomic update not
168  * guaranteed in invalid simultaneous state-updates.
169  *
170  * Contains the previously known good state and will be
171  * printed when detecting an invalid state transition.
172  */
173  ev_hdr_state_t state;
174  };
175  };

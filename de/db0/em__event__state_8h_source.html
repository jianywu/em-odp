99 /**
100  * Init values for the event-state counters 'free_cnt' and 'send_cnt'.
101  *
102  * The counters are 32-bit but are updated as one combined 64-bit atomic var,
103  * thus the init values are in the middle of the u32-range to avoid wraparounds
104  * when decrementing below '0'.
105  */
106 #define FREE_CNT_INIT ((uint16_t)0x0100) /* = 0 + 'offset' */
107 #define SEND_CNT_INIT ((uint16_t)0x0100) /* = 0 + 'offset' */
108 /** Initial event generation value */
109 #define EVGEN_INIT ((uint16_t)1)
110 /** Max evgen value before resetting to 'EVGEN_INIT' to avoid wrap */
111 #define EVGEN_MAX ((uint16_t)UINT16_MAX - 0x1000)

113 /**

115  *
116  * - EM_ESV_ENABLE is set via the 'configure' script: --enable/disable-esv

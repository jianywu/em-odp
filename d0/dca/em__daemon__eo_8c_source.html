41 void daemon_eo_create(void)
42 {
43  em_eo_t eo;
44  em_status_t stat;
45  em_status_t stat_eo_start = EM_ERROR;
46 
47  eo = em_eo_create("daemon-eo", daemon_eo_start, NULL /*start_local*/,
48  daemon_eo_stop, NULL /*stop_local*/,
49  daemon_eo_receive, NULL);
50  if (eo == EM_EO_UNDEF)
51  DAEMON_ERROR(EM_FATAL(EM_ERR_BAD_ID), "daemon-eo create fail");
52 
53  /* Store the EO in shared memory */
54  em_shm->daemon.eo = eo;
55 
56  stat = em_eo_start(eo, &stat_eo_start, NULL, 0, NULL);
57  if (stat != EM_OK || stat_eo_start != EM_OK)
58  DAEMON_ERROR(EM_FATAL(EM_ERR_LIB_FAILED),
59  "daemon-eo start failed!");
60 }

62 void daemon_eo_shutdown(void)
63 {
64  const int core = em_core_id();
65  const em_eo_t eo = em_shm->daemon.eo;
66  eo_elem_t *const eo_elem = eo_elem_get(eo);
67  em_status_t stat;
68 
69  EM_PRINT("%s() on EM-core %d\n", __func__, core);
70 
71  if (unlikely(eo_elem == NULL)) {
72  DAEMON_ERROR(EM_FATAL(EM_ERR_BAD_ID),
73  "daemon-eo handle:%" PRI_EO " invalid!", eo);
74  return;
75  }
76 
77  /*
78  * Stop the daemon-EO, i.e. call the daemon-EO global stop func.
79  * Note: cannot call normal API func em_eo_stop() since that would use
80  * internal ctrl events that might not be dispatched during shutdown.
81  */
82  /* Change state here to allow em_eo_delete() from EO stop func */
83  eo_elem->state = EM_EO_STATE_CREATED; /* == EO_STATE_STOPPED */
84  stat = eo_elem->stop_func(eo_elem->eo_ctx, eo);
85  if (stat != EM_OK)
86  DAEMON_ERROR(EM_FATAL(EM_ERR_LIB_FAILED),
87  "daemon-eo stop/delete failed!");
88 }

90 em_status_t daemon_eo_queues_create(void)
91 {
92  const int num_cores = em_core_count();
93  char q_name[EM_QUEUE_NAME_LEN];
94  em_queue_t shared_unsched_queue;
95  em_queue_t queue;
96  em_queue_conf_t unsch_conf;
97 
98  const char *err_str = "";
99  int i;
100 
101  EM_DBG("%s()\n", __func__);
102 
103  /*
104  * Create shared internal unsched queue used for internal EM messaging.
105  * Cannot use em_queue_create_static() here since the requested handle
106  * 'SHARED_INTERNAL_UNSCHED_QUEUE' lies outside of the normal static
107  * range.
108  */
109  shared_unsched_queue = queue_id2hdl(SHARED_INTERNAL_UNSCHED_QUEUE);
110  queue = queue_create("EMctrl-unschedQ-shared", EM_QUEUE_TYPE_UNSCHEDULED,
111  EM_QUEUE_PRIO_UNDEF, EM_QUEUE_GROUP_UNDEF,
112  shared_unsched_queue, EM_ATOMIC_GROUP_UNDEF,
113  NULL /* use default queue config */, &err_str);
114  if (queue == EM_QUEUE_UNDEF || queue != shared_unsched_queue)
115  return EM_FATAL(EM_ERR_NOT_FREE);
116 
117  /*
118  * Create static internal per-core UNSCHEDULED queues used for
119  * internal EM messaging. Cannot use em_queue_create_static()
120  * here since the requested handles lies outside of the normal
121  * static range.
122  */
123  memset(&unsch_conf, 0, sizeof(unsch_conf));
124  unsch_conf.flags |= EM_QUEUE_FLAG_DEQ_NOT_MTSAFE;
125 
126  for (i = 0; i < num_cores; i++) {
127  em_queue_t queue_req;
128 
129  queue_req = queue_id2hdl(FIRST_INTERNAL_UNSCHED_QUEUE + i);
130  snprintf(q_name, sizeof(q_name), "EMctrl-unschedQ-core%d", i);
131  q_name[EM_QUEUE_NAME_LEN - 1] = '\0';
132 
133  queue = queue_create(q_name, EM_QUEUE_TYPE_UNSCHEDULED,
134  EM_QUEUE_PRIO_UNDEF, EM_QUEUE_GROUP_UNDEF,
135  queue_req, EM_ATOMIC_GROUP_UNDEF,
136  &unsch_conf, /* request deq-not-mtsafe */
137  &err_str);
138  if (unlikely(queue == EM_QUEUE_UNDEF || queue != queue_req))
139  return EM_FATAL(EM_ERR_NOT_FREE);
140  }
141 
142  return EM_OK;
143 }

156 static em_status_t
157 daemon_eo_stop(void *eo_ctx, em_eo_t eo)
158 {
159  em_status_t stat = EM_OK;
160  eo_elem_t *const eo_elem = eo_elem_get(eo);
161 
162  (void)eo_ctx;
163 
164  EM_PRINT("%s() on EM-core %d\n", __func__, em_core_id());
165 
166  if (unlikely(eo_elem == NULL)) {
167  stat = EM_FATAL(EM_ERR_BAD_ID);
168  DAEMON_ERROR(stat, "daemon-eo handle:%" PRI_EO " invalid!", eo);
169  return stat;
170  }
171 
172  /* Cannot use API funcs - internal ctrl events might not work */
173  stat = queue_disable_all(eo_elem);
174  stat |= eo_delete_queue_all(eo_elem);
175  /* Finally delete the daemon-eo, API func is ok here */
176  stat |= em_eo_delete(eo);
177 
178  const int num_cores = em_core_count();
179  em_queue_t unsched_queue;
180  em_event_t unsched_event;
181 
182  unsched_queue = queue_id2hdl(SHARED_INTERNAL_UNSCHED_QUEUE);
183  for (;/* flush unsched queue */;) {
184  unsched_event = em_queue_dequeue(unsched_queue);
185  if (unsched_event == EM_EVENT_UNDEF)
186  break;
187  em_free(unsched_event);
188  }
189  stat = em_queue_delete(unsched_queue);
190  if (unlikely(stat != EM_OK))
191  return DAEMON_ERROR(stat, "shared unschedQ delete");
192 
193  for (int i = 0; i < num_cores; i++) {
194  unsched_queue = queue_id2hdl(FIRST_INTERNAL_UNSCHED_QUEUE + i);
195 
196  for (;/* flush unsched queue */;) {
197  unsched_event = em_queue_dequeue(unsched_queue);
198  if (unsched_event == EM_EVENT_UNDEF)
199  break;
200  em_free(unsched_event);
201  }
202 
203  stat = em_queue_delete(unsched_queue);
204  if (unlikely(stat != EM_OK))
205  return DAEMON_ERROR(stat, "core unschedQ:%d delete", i);
206  }
207 
208  return stat;
209 }

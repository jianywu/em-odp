105 /**
106  * Helper for em_event_clone().
107  *
108  * Clone an event originating from an external odp pkt-pool.
109  * Initialize the new cloned event as an EM event and return it.
110  */
111 em_event_t pkt_clone_odp(odp_packet_t pkt, odp_pool_t pkt_pool)
112 {
113  /*
114  * Alloc and copy content via ODP.
115  * Also the ev_hdr in the odp-pkt user_area is copied.
116  */
117  odp_packet_t clone_pkt = odp_packet_copy(pkt, pkt_pool);
118 
119  if (unlikely(clone_pkt == ODP_PACKET_INVALID))
120  return EM_EVENT_UNDEF;
121 
122  odp_packet_user_ptr_set(clone_pkt, PKT_USERPTR_MAGIC_NBR);
123 
124  odp_event_t odp_clone_event = odp_packet_to_event(clone_pkt);
125  event_hdr_t *clone_hdr = odp_packet_user_area(clone_pkt);
126  em_event_t clone_event = event_odp2em(odp_clone_event);
127 
128  /*
129  * Init hdr of event, also ESV init if needed.
130  * The clone_hdr is a copy of parent's, update only relevant fields.
131  */
132  if (esv_enabled())
133  clone_event = evstate_init(clone_event, clone_hdr, false);
134  else
135  clone_hdr->event = clone_event;
136 
137  /* clone_hdr->event_type = use parent's type as is */
138  clone_hdr->egrp = EM_EVENT_GROUP_UNDEF;
139 
140  if (em_shm->opt.pool.statistics_enable) {
141  /* clone_hdr->subpool = 0; */
142  clone_hdr->pool = EM_POOL_UNDEF;
143  }
144 
145  return clone_event;
146 }

161 void
162 output_queue_drain(const queue_elem_t *output_q_elem)
163 {
164  const em_queue_t output_queue = output_q_elem->queue;
165  const em_output_func_t output_fn =
166  output_q_elem->output.output_conf.output_fn;
167  void *const output_fn_args =
168  output_q_elem->output.output_conf.output_fn_args;
169 
170  const int deq_max = 32;
171 
172  em_event_t output_ev_tbl[deq_max];
173  /* use same event-tbl, dequeue odp events into the EM event-tbl */
174  odp_event_t *const odp_deq_events = (odp_event_t *)output_ev_tbl;
175 
176  const odp_queue_t odp_queue = output_q_elem->odp_queue;
177  unsigned int output_num;
178  int deq;
179  int ret;
180 
181  const bool esv_ena = esv_enabled();
182 
183  do {
184  deq = odp_queue_deq_multi(odp_queue,
185  odp_deq_events/*out=output_ev_tbl[]*/,
186  deq_max);
187  if (unlikely(deq <= 0))
188  return;
189 
190  output_num = (unsigned int)deq;
191  /* odp_deq_events[] == output_ev_tbl[], .evgen still missing */
192 
193  /* decrement pool statistics before passing events out-of-EM */
194  if (em_shm->opt.pool.statistics_enable || esv_ena) {
195  event_hdr_t *ev_hdrs[output_num];
196 
197  event_to_hdr_multi(output_ev_tbl, ev_hdrs, output_num);
198  if (em_shm->opt.pool.statistics_enable) {
199  poolstat_dec_evhdr_multi_output(ev_hdrs,
200  output_num);
201  }
202  if (esv_ena)
203  evstate_em2usr_multi(output_ev_tbl/*in/out*/,
204  ev_hdrs, output_num,
205  EVSTATE__OUTPUT_MULTI);
206  }
207 
208  ret = output_fn(output_ev_tbl, output_num,
209  output_queue, output_fn_args);
210 
211  if (unlikely((unsigned int)ret != output_num))
212  em_free_multi(&output_ev_tbl[ret], output_num - ret);
213  } while (deq > 0);
214 }

216 void
217 output_queue_buffering_drain(void)
218 {
219  output_queue_track_t *const track = &em_locm.output_queue_track;
220 
221  for (unsigned int i = 0; i < track->idx_cnt; i++) {
222  int qidx = track->idx[i];
223  queue_elem_t *output_q_elem = track->used_queues[qidx];
224  env_spinlock_t *lock = &output_q_elem->output.lock;
225 
226  /*
227  * drain if lock available, otherwise another core is already
228  * draining so no need to do anything.
229  */
230  if (env_spinlock_trylock(lock)) {
231  output_queue_drain(output_q_elem);
232  env_spinlock_unlock(lock);
233  }
234 
235  track->idx[i] = 0;
236  track->used_queues[qidx] = NULL;
237  }
238  track->idx_cnt = 0;
239 }

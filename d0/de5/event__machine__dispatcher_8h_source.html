36 /**
37  * @file
38  * @defgroup em_dispatcher Dispatcher
39  * Event Machine dispatcher related services (new to API 1.2).
40  * @{
41  *
42  * The implementation of scheduling and the dispatcher is system specific,
43  * but this file provides a placeholder for common dispatcher specific services.
44  * The dispatcher is commonly the main loop of each core (thread) that
45  * interfaces with the scheduler and does the mapping of queue->EO.
46  *
47  * API1.2 introduces EO enter-exit callback function hooks. These can be used
48  * to collect debugging information, statistics or implement new functionality.
49  * The global enter-callback is called before EO-receive on each core
50  * separately. The callback gets all the arguments the EO-receive would get and
51  * can additionally modify them. The exit-callback works in a similar way, but
52  * is instead called after the EO-receive returns and has no arguments except
53  * for the EO handle.
54  * Multiple callbacks can be registered. The calling order of multiple
55  * registered functions is the order of registration. If the same function is
56  * registered twice then it will be called twice. The amount of simultaneous
57  * callbacks is system specific (EM_CALLBACKS_MAX).
58  * If an enter-callback changes the event handle to UNDEF, the next callback
59  * will still be called with event as UNDEF, but the EO-receive won't be called
60  * with an UNDEF event.
61  *
62  * Do not include this from the application, event_machine.h will
63  * do it for you.
64  */

87 /**
88  * Dispatcher global EO-receive enter-callback.
89  *
90  * Common dispatch callback run before EO-receive functions of both the
91  * em_receive_func_t and em_receive_multi_func_t types (i.e. for EOs created
92  * with either em_eo_create() or em_eo_create_multircv()).
93  *
94  * Enter-callbacks are run just before entering EO-receive functions, they can
95  * be useful for debugging, collecting statistics, manipulating events before
96  * they reach the EO or implementing new services needing synchronization
97  * between cores.
98  * Arguments common for both types of EO receive functions are passed as
99  * references to the enter-callback (the event-type passed to the single-event
100  * receive function case is not passed, use em_event_get/set_type() instead).
101  * Arguments are references, i.e. the callback can optionally modify them.
102  * If modified, the new values will go to the next callback and eventually to
103  * the multi-event EO-receive function.
104  *
105  * Events can be dropped by changing the event-entries in the events[num]-array
106  * to EM_EVENT_UNDEF. Neither EO-receive nor any further enter-callbacks will
107  * be called if all events have been dropped by the callbacks already run, i.e.
108  * no callback will be called with 'num=0'.
109  * The callback itself needs to handle the events it drops, e.g. free them.
110  * Note: EM will remove entries of EM_EVENT_UNDEF from the events[]-array before
111  * calling the next enter-callback (if several registered) or the
112  * receive function and adjust 'num' accordingly for the call.
113  *
114  * The EO handle can be used to separate callback functionality per EO and the
115  * core id can be obtained for core specific functionality.
116  *
117  * Callback functions can be called concurrently from different cores.
118  *
119  * @see em_dispatch_register_enter_cb()
120  */
121 typedef void (*em_dispatch_enter_func_t)(em_eo_t eo, void **eo_ctx,
122  em_event_t events[/*in/out*/], int num,
123  em_queue_t *queue, void **q_ctx);

138 /**
139  * Register an EO-enter callback
140  *
141  * Register a global function to be called by the dispatcher just before calling
142  * an EO-receive function. This can be useful for debugging, collecting
143  * statistics, manipulating events before they reach the EO or implementing new
144  * services needing synchronization between cores.
145  *
146  * The function registered should be kept short since it will be run each time
147  * just before calling EO-receive. All registered callbacks will further
148  * increase the processing time.
149  *
150  * Multiple callbacks can be registered.
151  * The order of calling multiple registered functions is the order of
152  * registration. If same function is registered twice it will be called twice.
153  * The maximum number of simultaneous callbacks is system specific
154  * (EM_CALLBACKS_MAX).
155  *
156  * @param func Callback function
157  *
158  * @return EM_OK if callback registration succeeded
159  *
160  * @see em_dispatch_enter_func_t
161  */
162 em_status_t
163 em_dispatch_register_enter_cb(em_dispatch_enter_func_t func);

165 /**
166  * Unregister an EO-enter callback
167  *
168  * This can be used to unregister a previously registered enter-function.
169  *
170  * The given function is searched for and if found removed from the call list.
171  * If the same function has been registered multiple times, only one reference
172  * is removed per unregister call.
173  * Note that when this function returns, no new calls are made to the removed
174  * callback function, but it is still possible that another core could be
175  * executing the function, so care must be taken before removing anything it may
176  * still use.
177  *
178  * @param func Callback function
179  *
180  * @return EM_OK if the given function was found and removed.
181  */
182 em_status_t

159  */

162 /**
163  * @def EM_QUEUE_FLAG_NONBLOCKING_LF
164  *
165  * em_queue_flag_t value (system specific). Only combine flags with bitwise OR.
166  *

168  * Other threads can make progress while a thread is suspended.
169  * Starvation freedom is not guaranteed.
170  * Queue creation will fail if set and not supported.
171  */
172 #define EM_QUEUE_FLAG_NONBLOCKING_LF 1 /* non-blocking, lock-free */

175  * @def EM_QUEUE_FLAG_NONBLOCKING_WF
176  *
177  * em_queue_flag_t value (system specific). Only combine flags with bitwise OR.
178  *

180  * Other threads can make progress while a thread is suspended.
181  * Starvation freedom is guaranteed.
182  * Queue creation will fail if set and not supported.

320 /**
321  * 'Periodical' draining of output from EM, if needed.
322  *
323  * User provided function - EM calls this, if not NULL, in the dispatch loop on
324  * each core - set via 'em_conf.output.output_drain_fn'
325  *
326  * Draining of output events/pkts: EM will every once in a while call this
327  * user provided function to ensure that low rate buffered output is eventually
328  * sent out. Not needed if your EM output queues (EM_QUEUE_TYPE_OUTPUT) always
329  * sends all events out. Useful in situations where output is buffered and sent
330  * out in bursts when enough output has been gathered - single events or low
331  * rate flows may, without this function, never be sent out (or too late) if the
332  * buffering threshold has not been reached.
333  *
334  * @return number of events successfully drained and sent for output
335  */
336 typedef int (*em_output_drain_func_t)(void);
337 
338 /**
339  * Output function, user provided callback for queues of type
340  * EM_QUEUE_TYPE_OUTPUT.
341  *
342  * This function will be called by em_send*() when sending to a queue of type
343  * EM_QUEUE_TYPE_OUTPUT and EM will take care of correct function calling order
344  * based on the scheduling context type.
345  * The function can use em_sched_context_type_current() if it needs information
346  * about e.g. ordering requirements set by the parent scheduled queue.
347  *
348  * @param events List of events to be sent out (ptr to array of events)
349  * @param num Number of events (positive integer)
350  * @param output_queue Output queue that the events were sent to (em_send*())
351  * @param flags Output flags/options to indicate e.g. ordering
352  * requirement of the source context.
353  *
354  * @return number of events successfully sent (equal to num if all successful)
355  */
356 typedef int (*em_output_func_t)(const em_event_t events[],
357  const unsigned int num,
358  const em_queue_t output_queue,
359  void *output_fn_args);

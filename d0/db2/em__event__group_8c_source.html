40 em_status_t
41 event_group_init(event_group_tbl_t *const event_group_tbl,
42  event_group_pool_t *const event_group_pool)
43 {
44  event_group_elem_t *egrp_elem;
45  const int cores = em_core_count();
46  int ret;
47 
48  memset(event_group_tbl, 0, sizeof(event_group_tbl_t));
49  memset(event_group_pool, 0, sizeof(event_group_pool_t));
50  env_atomic32_init(&em_shm->event_group_count);
51 
52  for (int i = 0; i < EM_MAX_EVENT_GROUPS; i++) {
53  em_event_group_t egrp = egrp_idx2hdl(i);
54 
55  egrp_elem = event_group_elem_get(egrp);
56  if (unlikely(!egrp_elem))
57  return EM_ERR_BAD_POINTER;
58 
59  egrp_elem->event_group = egrp; /* store handle */
60  egrp_elem->all = 0; /* set num_notif = 0, ready = 0 */
61  env_atomic64_set(&egrp_elem->post.atomic, 0);
62  env_atomic64_set(&egrp_elem->pre.atomic, 0);
63  }
64 
65  ret = objpool_init(&event_group_pool->objpool, cores);
66  if (ret != 0)
67  return EM_ERR_LIB_FAILED;
68 
69  for (int i = 0; i < EM_MAX_EVENT_GROUPS; i++) {
70  egrp_elem = &event_group_tbl->egrp_elem[i];
71  objpool_add(&event_group_pool->objpool, i % cores,
72  &egrp_elem->event_group_pool_elem);
73  }
74 
75  return EM_OK;
76 }

78 em_event_group_t
79 event_group_alloc(void)
80 {
81  const event_group_elem_t *egrp_elem;
82  objpool_elem_t *egrp_pool_elem;
83 
84  egrp_pool_elem = objpool_rem(&em_shm->event_group_pool.objpool,
85  em_core_id());
86  if (unlikely(egrp_pool_elem == NULL))
87  return EM_EVENT_GROUP_UNDEF;
88 
89  egrp_elem = egrp_poolelem2egrp(egrp_pool_elem);
90 
91  env_atomic32_inc(&em_shm->event_group_count);
92  return egrp_elem->event_group;
93 }

95 em_status_t
96 event_group_free(em_event_group_t event_group)
97 {
98  event_group_elem_t *egrp_elem = event_group_elem_get(event_group);
99 
100  if (unlikely(egrp_elem == NULL))
101  return EM_ERR_BAD_ID;
102 
103  objpool_add(&em_shm->event_group_pool.objpool,
104  egrp_elem->event_group_pool_elem.subpool_idx,
105  &egrp_elem->event_group_pool_elem);
106 
107  env_atomic32_dec(&em_shm->event_group_count);
108  return EM_OK;
109 }

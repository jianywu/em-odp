42 /*
43  * Max supported value for the config file option 'pool.align_offset'.
44  *
45  * The limitation is set by events based on odp-bufs that include the ev-hdr at
46  * the beginning of the odp-buf payload - the alignment is adjusted into the end
47  * of the ev-hdr.
48  * Events based on odp-pkts do not have this restriction but the same limit is
49  * used for all.
50  */
51 #define ALIGN_OFFSET_MAX ((int)(sizeof(event_hdr_t) - \
52  offsetof(event_hdr_t, end_hdr_data)))

61 static int
62 read_config_file(void)
63 {
64  const char *conf_str;
65  bool val_bool = false;
66  int val = 0;
67  int ret;
68 
69  const odp_pool_capability_t *capa =
70  &em_shm->mpool_tbl.odp_pool_capability;
71 
72  EM_PRINT("EM-pool config:\n");
73 
74  /*
75  * Option: pool.statistics_enable
76  */
77  conf_str = "pool.statistics_enable";
78  ret = em_libconfig_lookup_bool(&em_shm->libconfig, conf_str, &val_bool);
79  if (unlikely(!ret)) {
80  EM_LOG(EM_LOG_ERR, "Config option '%s' not found", conf_str);
81  return -1;
82  }
83  /* store & print the value */
84  em_shm->opt.pool.statistics_enable = (int)val_bool;
85  EM_PRINT(" %s: %s(%d)\n", conf_str, val_bool ? "true" : "false",
86  val_bool);
87 
88  /*
89  * Option: pool.align_offset
90  */
91  conf_str = "pool.align_offset";
92  ret = em_libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
93  if (unlikely(!ret)) {
94  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
95  return -1;
96  }
97  if (val < 0 || val > ALIGN_OFFSET_MAX || !POWEROF2(val)) {
98  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %d'\n",
99  conf_str, val);
100  return -1;
101  }
102  /* store & print the value */
103  em_shm->opt.pool.align_offset = val;
104  EM_PRINT(" %s (default): %d (max: %d)\n",
105  conf_str, val, ALIGN_OFFSET_MAX);
106 
107  /*
108  * Option: pool.user_area_size
109  */
110  conf_str = "pool.user_area_size";
111  ret = em_libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
112  if (unlikely(!ret)) {
113  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
114  return -1;
115  }
116  if (val < 0 || (unsigned int)val > capa->pkt.max_uarea_size ||
117  val > EM_EVENT_USER_AREA_MAX_SIZE) {
118  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %d'\n",
119  conf_str, val);
120  return -1;
121  }
122  /* store & print the value */
123  em_shm->opt.pool.user_area_size = val;
124  EM_PRINT(" %s (default): %d (max: %d)\n",
125  conf_str, val,
126  MIN(EM_EVENT_USER_AREA_MAX_SIZE, capa->pkt.max_uarea_size));
127 
128  /*
129  * Option: pool.pkt_headroom
130  */
131  conf_str = "pool.pkt_headroom";
132  ret = em_libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
133  if (unlikely(!ret)) {
134  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
135  return -1;
136  }
137 
138  if (val < 0 || (unsigned int)val > capa->pkt.max_headroom) {
139  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %d'\n",
140  conf_str, val);
141  return -1;
142  }
143  /* store & print the value */
144  em_shm->opt.pool.pkt_headroom = val;
145  EM_PRINT(" %s (default): %d (max: %u)\n",
146  conf_str, val, capa->pkt.max_headroom);
147 
148  return 0;
149 }

151 em_status_t
152 pool_init(mpool_tbl_t *const mpool_tbl, mpool_pool_t *const mpool_pool,
153  const em_pool_cfg_t *default_pool_cfg)
154 {
155  em_pool_t pool;
156  int ret;
157  const int cores = em_core_count();
158 
159  memset(mpool_tbl, 0, sizeof(mpool_tbl_t));
160  memset(mpool_pool, 0, sizeof(mpool_pool_t));
161  env_atomic32_init(&em_shm->pool_count);
162 
163  ret = objpool_init(&mpool_pool->objpool, cores);
164  if (ret != 0)
165  return EM_ERR_OPERATION_FAILED;
166 
167  for (int i = 0; i < EM_CONFIG_POOLS; i++) {
168  pool = pool_idx2hdl(i);
169  mpool_elem_t *mpool_elem = pool_elem_get(pool);
170 
171  if (unlikely(!mpool_elem))
172  return EM_ERR_BAD_POINTER;
173 
174  mpool_elem->em_pool = pool;
175  mpool_elem->event_type = EM_EVENT_TYPE_UNDEF;
176  for (int j = 0; j < EM_MAX_SUBPOOLS; j++) {
177  mpool_elem->odp_pool[j] = ODP_POOL_INVALID;
178  mpool_elem->size[j] = 0;
179  }
180 
181  objpool_add(&mpool_pool->objpool, i % cores,
182  &mpool_elem->objpool_elem);
183  }
184 
185  /* Init the mapping tbl from odp-pool(=subpool) index to em-pool */
186  if (odp_pool_max_index() >= POOL_ODP2EM_TBL_LEN)
187  return EM_ERR_TOO_LARGE;
188  for (int i = 0; i < POOL_ODP2EM_TBL_LEN; i++)
189  mpool_tbl->pool_odp2em[i] = EM_POOL_UNDEF;
190 
191  /* Store common ODP pool capabilities in the mpool_tbl for easy access*/
192  if (odp_pool_capability(&mpool_tbl->odp_pool_capability) != 0)
193  return EM_ERR_LIB_FAILED;
194 
195  /* Read EM-pool related runtime config options */
196  if (read_config_file())
197  return EM_ERR_LIB_FAILED;
198 
199  /* Create the 'EM_POOL_DEFAULT' pool */
200  pool = em_pool_create(EM_POOL_DEFAULT_NAME, EM_POOL_DEFAULT,
201  default_pool_cfg);
202  if (pool == EM_POOL_UNDEF || pool != EM_POOL_DEFAULT)
203  return EM_ERR_ALLOC_FAILED;
204 
205  return EM_OK;
206 }

208 em_status_t
209 pool_term(const mpool_tbl_t *mpool_tbl)
210 {
211  em_status_t stat = EM_OK;
212  int i;
213 
214  (void)mpool_tbl;
215 
216  EM_PRINT("\n"
217  "Status before delete:\n");
218  em_pool_info_print_all();
219 
220  for (i = 0; i < EM_CONFIG_POOLS; i++) {
221  em_pool_t pool = pool_idx2hdl(i);
222  const mpool_elem_t *mpool_elem = pool_elem_get(pool);
223  em_status_t ret;
224 
225  if (pool_allocated(mpool_elem)) {
226  ret = pool_delete(pool);
227  if (ret != EM_OK)
228  stat = ret; /* save last error as return val */
229  }
230  }
231 
232  return stat;
233 }

235 static em_pool_t
236 pool_alloc(em_pool_t pool)
237 {
238  mpool_elem_t *mpool_elem;
239 
240  if (pool == EM_POOL_UNDEF) {
241  objpool_elem_t *objpool_elem =
242  objpool_rem(&em_shm->mpool_pool.objpool, em_core_id());
243 
244  if (unlikely(objpool_elem == NULL))
245  return EM_POOL_UNDEF;
246 
247  mpool_elem = mpool_poolelem2pool(objpool_elem);
248  } else {
249  int ret;
250 
251  mpool_elem = pool_elem_get(pool);
252  if (unlikely(mpool_elem == NULL))
253  return EM_POOL_UNDEF;
254 
255  ret = objpool_rem_elem(&em_shm->mpool_pool.objpool,
256  &mpool_elem->objpool_elem);
257  if (unlikely(ret != 0))
258  return EM_POOL_UNDEF;
259  }
260 
261  env_atomic32_inc(&em_shm->pool_count);
262  return mpool_elem->em_pool;
263 }

265 static em_status_t
266 pool_free(em_pool_t pool)
267 {
268  mpool_elem_t *mpool_elem = pool_elem_get(pool);
269 
270  if (unlikely(mpool_elem == NULL))
271  return EM_ERR_BAD_ID;
272 
273  objpool_add(&em_shm->mpool_pool.objpool,
274  mpool_elem->objpool_elem.subpool_idx,
275  &mpool_elem->objpool_elem);
276 
277  env_atomic32_dec(&em_shm->pool_count);
278  return EM_OK;
279 }

281 /* Helper func to invalid_pool_cfg() */
282 static int invalid_pool_cache_cfg(const em_pool_cfg_t *pool_cfg,
283  const char **err_str/*out*/)
284 {
285  const odp_pool_capability_t *capa =
286  &em_shm->mpool_tbl.odp_pool_capability;
287  uint32_t min_cache_size;
288  uint32_t cache_size;
289 
290  if (pool_cfg->event_type == EM_EVENT_TYPE_SW)
291  min_cache_size = capa->buf.min_cache_size;
292  else
293  min_cache_size = capa->pkt.min_cache_size;
294 
295  for (int i = 0; i < pool_cfg->num_subpools; i++) {
296  if (pool_cfg->subpool[i].size <= 0 ||
297  pool_cfg->subpool[i].num <= 0) {
298  *err_str = "Invalid subpool size/num";
299  return -(9 * 10 + i); /* -90, -91, ... */
300  }
301 
302  cache_size = pool_cfg->subpool[i].cache_size;
303  if (unlikely(cache_size < min_cache_size)) {
304  *err_str = "Requested cache size too small";
305  return -(10 * 10 + i); /* -100, -101, ... */
306  }
307  /*
308  * If the given cache size is larger than odp-max,
309  * then use odp-max:
310  * if (cache_size > max_cache_size)
311  * cache_size = max_cache_size;
312  * This is done later in pool_create();
313  */
314  }
315 
316  return 0;
317 }

319 int invalid_pool_cfg(const em_pool_cfg_t *pool_cfg, const char **err_str/*out*/)
320 {
321  const odp_pool_capability_t *capa =
322  &em_shm->mpool_tbl.odp_pool_capability;
323 
324  if (!pool_cfg) {
325  *err_str = "Pool config NULL";
326  return -1;
327  }
328  if (pool_cfg->__internal_check != EM_CHECK_INIT_CALLED) {
329  *err_str = "Pool config not initialized";
330  return -2;
331  }
332 
333  if (pool_cfg->num_subpools <= 0 ||
334  pool_cfg->num_subpools > EM_MAX_SUBPOOLS) {
335  *err_str = "Invalid number of subpools";
336  return -3;
337  }
338 
339  if (pool_cfg->event_type != EM_EVENT_TYPE_SW &&
340  pool_cfg->event_type != EM_EVENT_TYPE_PACKET) {
341  *err_str = "Pool event type not supported, use _SW or _PACKET";
342  return -4;
343  }
344 
345  if (pool_cfg->align_offset.in_use &&
346  (pool_cfg->align_offset.value > ALIGN_OFFSET_MAX ||
347  !POWEROF2(pool_cfg->align_offset.value))) {
348  *err_str = "Invalid align offset";
349  return -5;
350  }
351 
352  if (pool_cfg->user_area.in_use) {
353  if (pool_cfg->user_area.size > EM_EVENT_USER_AREA_MAX_SIZE) {
354  *err_str = "Event user area too large";
355  return -6;
356  }
357  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
358  size_t req_odp_uarea_sz = pool_cfg->user_area.size +
359  sizeof(event_hdr_t);
360  if (req_odp_uarea_sz > capa->pkt.max_uarea_size) {
361  *err_str = "ODP pkt max uarea not large enough";
362  return -7;
363  }
364  }
365  }
366 
367  if (pool_cfg->pkt.headroom.in_use &&
368  pool_cfg->pkt.headroom.value > capa->pkt.max_headroom) {
369  *err_str = "Requested pkt headroom size too large";
370  return -8;
371  }
372 
373  int err = invalid_pool_cache_cfg(pool_cfg, err_str/*out*/);
374 
375  return err;
376 }

378 /*
379  * Helper to pool_create() - preallocate all events in the pool for ESV to
380  * maintain event state over multiple alloc- and free-operations.
381  */
382 static void
383 pool_prealloc(const mpool_elem_t *pool_elem)
384 {
385  em_event_t event;
386  event_hdr_t *ev_hdr;
387  uint64_t num_tot = 0;
388  uint64_t num = 0;
389  const uint32_t size = pool_elem->pool_cfg.subpool[0].size;
390  list_node_t evlist;
391  list_node_t *node;
392 
393  list_init(&evlist);
394 
395  for (int i = 0; i < pool_elem->num_subpools; i++)
396  num_tot += pool_elem->pool_cfg.subpool[i].num;
397 
398  do {
399  event = event_prealloc(pool_elem, size, pool_elem->event_type);
400  if (likely(event != EM_EVENT_UNDEF)) {
401  ev_hdr = event_to_hdr(event);
402  list_add(&evlist, &ev_hdr->start_node);
403  num++;
404  }
405  } while (event != EM_EVENT_UNDEF);
406 
407  if (unlikely(num < num_tot))
408  INTERNAL_ERROR(EM_FATAL(EM_ERR_TOO_SMALL),
409  EM_ESCOPE_POOL_CREATE,
410  "events expected:%" PRIu64 " actual:%" PRIu64 "",
411  num_tot, num);
412 
413  while (!list_is_empty(&evlist)) {
414  node = list_rem_first(&evlist);
415  ev_hdr = start_node_to_event_hdr(node);
416  em_free(ev_hdr->event);
417  }
418 }

420 /*
421  * pool_create() helper: sort subpool cfg in ascending order based on buf size
422  */
423 static void
424 sort_pool_cfg(const em_pool_cfg_t *pool_cfg, em_pool_cfg_t *sorted_cfg /*out*/)
425 {
426  const int num_subpools = pool_cfg->num_subpools;
427 
428  *sorted_cfg = *pool_cfg;
429 
430  for (int i = 0; i < num_subpools - 1; i++) {
431  int idx = i; /* array index containing smallest size */
432 
433  for (int j = i + 1; j < num_subpools; j++) {
434  if (sorted_cfg->subpool[j].size <
435  sorted_cfg->subpool[idx].size)
436  idx = j; /* store idx to smallest */
437  }
438 
439  /* min size at [idx], swap with [i] */
440  if (idx != i) {
441  uint32_t size = sorted_cfg->subpool[i].size;
442  uint32_t num = sorted_cfg->subpool[i].num;
443  uint32_t cache_size = sorted_cfg->subpool[i].cache_size;
444 
445  sorted_cfg->subpool[i] = sorted_cfg->subpool[idx];
446 
447  sorted_cfg->subpool[idx].size = size;
448  sorted_cfg->subpool[idx].num = num;
449  sorted_cfg->subpool[idx].cache_size = cache_size;
450  }
451  }
452 }

454 /*
455  * pool_create() helper: set pool event-cache size.
456  *
457  * Set the requested subpool cache-size based on user provided value and
458  * limit set by odp-pool-capability.
459  * Requested value can be larger than odp-max, use odp--max in this
460  * case.
461  * Verification against odp-min value done in invalid_pool_cfg().
462  */
463 static void
464 set_poolcache_size(em_pool_cfg_t *pool_cfg)
465 {
466  const odp_pool_capability_t *capa =
467  &em_shm->mpool_tbl.odp_pool_capability;
468  int num_subpools = pool_cfg->num_subpools;
469  uint32_t max_cache_size;
470 
471  if (pool_cfg->event_type == EM_EVENT_TYPE_SW)
472  max_cache_size = capa->buf.max_cache_size;
473  else
474  max_cache_size = capa->pkt.max_cache_size;
475 
476  for (int i = 0; i < num_subpools; i++) {
477  if (max_cache_size < pool_cfg->subpool[i].cache_size)
478  pool_cfg->subpool[i].cache_size = max_cache_size;
479  }
480 }

500 /*
501  * pool_create() helper: determine payload alignment.
502  */
503 static int
504 set_align(const em_pool_cfg_t *pool_cfg,
505  uint32_t *align_offset /*out*/, uint32_t *odp_align /*out*/)
506 {
507  const odp_pool_capability_t *capa =
508  &em_shm->mpool_tbl.odp_pool_capability;
509  uint32_t offset = 0;
510  uint32_t align = ODP_CACHE_LINE_SIZE;
511 
512  /* Pool-specific param overrides config file 'align_offset' value */
513  if (pool_cfg->align_offset.in_use)
514  offset = pool_cfg->align_offset.value; /* pool cfg */
515  else
516  offset = em_shm->opt.pool.align_offset; /* cfg file */
517 
518  /* Set subpool minimum alignment */
519  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
520  if (align > capa->pkt.max_align)
521  align = capa->pkt.max_align;
522  } else {
523  if (align > capa->buf.max_align)
524  align = capa->buf.max_align;
525  }
526 
527  *align_offset = offset;
528  *odp_align = align;
529 
530  /* verify alignment requirements */
531  if (!POWEROF2(align) || align <= offset)
532  return -1;
533 
534  return 0;
535 }

537 /*
538  * pool_create() helper: determine user area size.
539  */
540 static int
541 set_uarea_size(const em_pool_cfg_t *pool_cfg, uint32_t align_offset,
542  size_t *uarea_req_size/*out*/, size_t *uarea_pad_size/*out*/)
543 {
544  size_t req_size = 0;
545  size_t pad_size = 0;
546  size_t max_size = 0;
547  const odp_pool_capability_t *capa =
548  &em_shm->mpool_tbl.odp_pool_capability;
549 
550  if (pool_cfg->user_area.in_use) /* use pool-cfg */
551  req_size = pool_cfg->user_area.size;
552  else /* use cfg-file */
553  req_size = em_shm->opt.pool.user_area_size;
554 
555  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
556  pad_size = req_size;
557  max_size = MIN(capa->pkt.max_uarea_size,
558  EM_EVENT_USER_AREA_MAX_SIZE);
559  } else if (req_size > 0) {
560  /* EM_EVENT_TYPE_SW: bufs */
561  /* Note: contains align_offset extra space for adjustment */
562  pad_size = ROUND_UP(req_size + align_offset, 32);
563  max_size = EM_EVENT_USER_AREA_MAX_SIZE;
564  }
565 
566  if (req_size > max_size)
567  return -1;
568 
569  *uarea_req_size = req_size;
570  *uarea_pad_size = pad_size;
571  return 0;
572 }

600 static int
601 create_subpools(const em_pool_cfg_t *pool_cfg,
602  uint32_t align_offset, uint32_t odp_align,
603  uint32_t uarea_size, uint32_t pkt_headroom,
604  mpool_elem_t *mpool_elem /*out*/)
605 {
606  const int num_subpools = pool_cfg->num_subpools;
607  mpool_tbl_t *const mpool_tbl = &em_shm->mpool_tbl;
608 
609  for (int i = 0; i < num_subpools; i++) {
610  char pool_name[ODP_POOL_NAME_LEN];
611  odp_pool_param_t pool_params;
612  uint32_t size = pool_cfg->subpool[i].size;
613  uint32_t cache_size = pool_cfg->subpool[i].cache_size;
614 
615  odp_pool_param_init(&pool_params);
616 
617  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
618  pool_params.type = ODP_POOL_PACKET;
619  /* num == max_num */
620  pool_params.pkt.num = pool_cfg->subpool[i].num;
621  pool_params.pkt.max_num = pool_cfg->subpool[i].num;
622 
623  if (size > align_offset)
624  size = size - align_offset;
625  else
626  size = 1; /* 0:default, can be big => use 1 */
627  /* len == max_len */
628  pool_params.pkt.len = size;
629  pool_params.pkt.max_len = size;
630  pool_params.pkt.seg_len = size;
631  pool_params.pkt.align = odp_align;
632  /*
633  * Reserve space for the event header in each packet's
634  * ODP-user-area:
635  */
636  pool_params.pkt.uarea_size = sizeof(event_hdr_t) +
637  uarea_size;
638  /*
639  * Set the pkt headroom.
640  * Make sure the alloc-alignment fits into the headroom.
641  */
642  pool_params.pkt.headroom = pkt_headroom;
643  if (pkt_headroom < align_offset)
644  pool_params.pkt.headroom = align_offset;
645 
646  pool_params.pkt.cache_size = cache_size;
647  } else { /* pool_cfg->event_type == EM_EVENT_TYPE_SW */
648  pool_params.type = ODP_POOL_BUFFER;
649  pool_params.buf.num = pool_cfg->subpool[i].num;
650  pool_params.buf.size = size + sizeof(event_hdr_t) +
651  uarea_size;
652  pool_params.buf.align = odp_align;
653  pool_params.buf.cache_size = cache_size;
654  }
655 
656  snprintf(pool_name, sizeof(pool_name), "%" PRI_POOL ":%d-%s",
657  mpool_elem->em_pool, i, mpool_elem->name);
658  pool_name[sizeof(pool_name) - 1] = '\0';
659 
660  odp_pool_t odp_pool = odp_pool_create(pool_name, &pool_params);
661 
662  if (unlikely(odp_pool == ODP_POOL_INVALID))
663  return -1;
664 
665  int odp_pool_idx = odp_pool_index(odp_pool);
666 
667  if (unlikely(odp_pool_idx < 0))
668  return -2;
669 
670  /* Store mapping from odp-pool (idx) to em-pool */
671  mpool_tbl->pool_odp2em[odp_pool_idx] = mpool_elem->em_pool;
672 
673  mpool_elem->odp_pool[i] = odp_pool;
674  mpool_elem->size[i] = pool_cfg->subpool[i].size;
675  mpool_elem->num_subpools++; /* created subpools for delete */
676 
677  /*odp_pool_print(odp_pool);*/
678  }
679 
680  return 0;
681 }

683 em_pool_t
684 pool_create(const char *name, em_pool_t req_pool, const em_pool_cfg_t *pool_cfg)
685 {
686  /* Allocate a free EM pool */
687  const em_pool_t pool = pool_alloc(req_pool/* requested or undef*/);
688 
689  if (unlikely(pool == EM_POOL_UNDEF))
690  return EM_POOL_UNDEF;
691 
692  mpool_elem_t *mpool_elem = pool_elem_get(pool);
693 
694  /* Sanity check */
695  if (!mpool_elem || mpool_elem->em_pool != pool)
696  return EM_POOL_UNDEF;
697 
698  mpool_elem->event_type = pool_cfg->event_type;
699  /* Store successfully created subpools later */
700  mpool_elem->num_subpools = 0;
701  /* Store the event pool name, if given */
702  if (name && *name) {
703  strncpy(mpool_elem->name, name, sizeof(mpool_elem->name));
704  mpool_elem->name[sizeof(mpool_elem->name) - 1] = '\0';
705  } else {
706  mpool_elem->name[0] = '\0';
707  }
708 
709  em_pool_cfg_t sorted_cfg;
710 
711  /*
712  * Sort the subpool cfg in ascending order based on the buffer size
713  */
714  sort_pool_cfg(pool_cfg, &sorted_cfg/*out*/);
715  /* Use sorted_cfg instead of pool_cfg from here on */
716 
717  /*
718  * Set the cache-size of each subpool in the EM-pool
719  */
720  set_poolcache_size(&sorted_cfg);
721 
722  /* Store the sorted config */
723  mpool_elem->pool_cfg = sorted_cfg;
724 
725  /*
726  * Initialize pool-statistics for this pool
727  */
728  init_pool_stat(pool);
729 
730  /*
731  * Event payload alignment requirement for the pool
732  */
733  uint32_t align_offset = 0;
734  uint32_t odp_align = 0;
735  int err = set_align(&sorted_cfg, &align_offset/*out*/,
736  &odp_align/*out*/);
737  if (unlikely(err)) {
738  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_POOL_CREATE,
739  "EM-pool:\"%s\" align mismatch:\n"
740  "align:%u cfg:align_offset:%u",
741  name, odp_align, align_offset);
742  goto error;
743  }
744  /* store the align offset, needed in pkt-alloc */
745  mpool_elem->align_offset = align_offset;
746 
747  /*
748  * Event user area size.
749  * Pool-specific param overrides config file 'user_area_size' value
750  */
751  size_t uarea_req_size = 0;
752  size_t uarea_pad_size = 0;
753 
754  err = set_uarea_size(&sorted_cfg, align_offset,
755  &uarea_req_size/*out*/, &uarea_pad_size/*out*/);
756  if (unlikely(err)) {
757  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_POOL_CREATE,
758  "EM-pool:\"%s\" invalid uarea config:\n"
759  "req.size:%zu => padded uarea size:%zu",
760  name, uarea_req_size, uarea_pad_size);
761  goto error;
762  }
763 
764  /* store the user_area sizes, needed in alloc */
765  mpool_elem->user_area.req_size = uarea_req_size & UINT16_MAX;
766  mpool_elem->user_area.pad_size = uarea_pad_size & UINT16_MAX;
767 
768  EM_DBG("EM-pool:\"%s\":\n"
769  " user_area: .req_size=%zu .pad_size=%zu align_offset=%u\n",
770  name, uarea_req_size, uarea_pad_size, align_offset);
771 
772  /*
773  * Set the headroom for events in EM packet pools
774  */
775  uint32_t pkt_headroom = 0;
776  uint32_t max_headroom = 0;
777 
778  if (sorted_cfg.event_type == EM_EVENT_TYPE_PACKET) {
779  err = set_pkt_headroom(&sorted_cfg, &pkt_headroom/*out*/,
780  &max_headroom/*out*/);
781  if (unlikely(err)) {
782  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_POOL_CREATE,
783  "EM-pool:\"%s\" invalid pkt headroom:\n"
784  "headroom:%u vs. max:headroom:%u",
785  name, pkt_headroom, max_headroom);
786  goto error;
787  }
788  }
789 
790  /*
791  * Create the subpools for the EM event-pool.
792  * Each EM subpool is an ODP pool.
793  */
794  err = create_subpools(&sorted_cfg, align_offset, odp_align,
795  uarea_pad_size, pkt_headroom, mpool_elem /*out*/);
796  if (unlikely(err)) {
797  INTERNAL_ERROR(EM_FATAL(EM_ERR_ALLOC_FAILED),
798  EM_ESCOPE_POOL_CREATE,
799  "EM-pool:\"%s\" create fails:%d\n"
800  "subpools req:%d vs. subpools created:%d",
801  name, err, sorted_cfg.num_subpools,
802  mpool_elem->num_subpools);
803  goto error;
804  }
805 
806  /*
807  * ESV: preallocate all events in the pool
808  */
809  if (esv_enabled() && em_shm->opt.esv.prealloc_pools)
810  pool_prealloc(mpool_elem);
811 
812  /* Success! */
813  return mpool_elem->em_pool;
814 
815 error:
816  (void)pool_delete(pool);
817  return EM_POOL_UNDEF;
818 }

820 em_status_t
821 pool_delete(em_pool_t pool)
822 {
823  mpool_tbl_t *const mpool_tbl = &em_shm->mpool_tbl;
824  mpool_elem_t *const mpool_elem = pool_elem_get(pool);
825  int i;
826 
827  if (unlikely(mpool_elem == NULL || !pool_allocated(mpool_elem)))
828  return EM_ERR_BAD_STATE;
829 
830  for (i = 0; i < mpool_elem->num_subpools; i++) {
831  odp_pool_t odp_pool = mpool_elem->odp_pool[i];
832  int odp_pool_idx;
833  int ret;
834 
835  if (odp_pool == ODP_POOL_INVALID)
836  return EM_ERR_NOT_FOUND;
837 
838  odp_pool_idx = odp_pool_index(odp_pool);
839  if (unlikely(odp_pool_idx < 0))
840  return EM_ERR_BAD_ID;
841 
842  ret = odp_pool_destroy(odp_pool);
843  if (unlikely(ret))
844  return EM_ERR_LIB_FAILED;
845 
846  /* Clear mapping from odp-pool (idx) to em-pool */
847  mpool_tbl->pool_odp2em[odp_pool_idx] = EM_POOL_UNDEF;
848 
849  mpool_elem->odp_pool[i] = ODP_POOL_INVALID;
850  mpool_elem->size[i] = 0;
851  }
852 
853  mpool_elem->name[0] = '\0';
854  mpool_elem->event_type = EM_EVENT_TYPE_UNDEF;
855  mpool_elem->num_subpools = 0;
856 
857  return pool_free(pool);
858 }

124 /**
125  * @brief Get the ODP pools used as subpools in a given EM event pool.
126  *
127  * An EM event pool consists of 1 to 'EM_MAX_SUBPOOLS' subpools. Each subpool
128  * is an ODP pool. This function outputs the ODP pool handles of these subpools
129  * into a user-provided array and returns the number of handles written.
130  *
131  * The obtained ODP pools must not be deleted or alterede outside of EM,
132  * e.g. these ODP pools must only be deleted as part of an EM event pool
133  * using em_pool_delete().
134  *
135  * ODP pool handles obtained through this function can be used to
136  * - configure ODP pktio to use an ODP pool created via EM (allows for
137  * better ESV tracking)
138  * - print ODP-level pool statistics with ODP APIs etc.
139  *
140  * Note that direct allocations and free:s via ODP APIs will bypass
141  * EM checks (e.g. ESV) and might cause errors unless properely handled:
142  * - use em_odp_event2em() to initialize as an EM event
143  * - use em_event_mark_free() before ODP-free operations (SW- or HW-free)
144  *
145  * @param pool EM event pool handle.
146  * @param[out] odp_pools Output array to be filled with the ODP pools used as
147  * subpools in the given EM event pool. The array must
148  * fit 'num' entries.
149  * @param num Number of entries in the 'odp_pools[]' array.
150  * Using 'num=EM_MAX_SUBPOOLS' will always be large
151  * enough to fit all subpools in the EM event pool.
152  *
153  * @return The number of ODP pools filled into 'odp_pools[]'
154  */
155 int em_odp_pool2odp(em_pool_t pool, odp_pool_t odp_pools[/*out*/], int num);

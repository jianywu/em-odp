105 {
106  const char *conf_str;
107  int val = 0;
108  int ret;
109 
110  EM_PRINT("EM-queue config:\n");

128  EM_PRINT(" %s: %d\n", conf_str, val);

130  /*
131  * Option: queue.prio_map_mode
132  */
133  conf_str = "queue.priority.map_mode";
134  ret = em_libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
135  if (unlikely(!ret)) {
136  EM_LOG(EM_LOG_ERR, "Config option '%s' not found\n", conf_str);
137  return -1;

159  EM_PRINT(",");

161  EM_PRINT("]\n");
162  }

166 /**
167  * Helper: initialize a queue pool (populate pool with q_elems)
168  */
169 static int
170 queue_pool_init(queue_tbl_t *const queue_tbl,
171  queue_pool_t *const queue_pool,
172  int min_qidx, int max_qidx)
173 {
174  const int cores = em_core_count();
175  const int qs_per_pool = (max_qidx - min_qidx + 1);
176  int qs_per_subpool = qs_per_pool / cores;
177  int qs_leftover = qs_per_pool % cores;
178  int subpool_idx = 0;
179  int add_cnt = 0;
180  int i;
181 
182  if (objpool_init(&queue_pool->objpool, cores) != 0)
183  return -1;
184 
185  for (i = min_qidx; i <= max_qidx; i++) {
186  objpool_add(&queue_pool->objpool, subpool_idx,
187  &queue_tbl->queue_elem[i].queue_pool_elem);
188  add_cnt++;
189  if (add_cnt == qs_per_subpool + qs_leftover) {
190  subpool_idx++; /* add to next subpool */
191  qs_leftover = 0; /* added leftovers to subpool 0 */
192  add_cnt = 0;
193  }
194  }
195 
196  return 0;
197 }

199 /**
200  * Initialize the EM queues
201  */
202 em_status_t
203 queue_init(queue_tbl_t *const queue_tbl,
204  queue_pool_t *const queue_pool,
205  queue_pool_t *const queue_pool_static)
206 {
207  odp_queue_capability_t *const odp_queue_capa =
208  &queue_tbl->odp_queue_capability;
209  odp_schedule_capability_t *const odp_sched_capa =
210  &queue_tbl->odp_schedule_capability;
211  int min;
212  int max;
213  int ret;
214 
215  memset(queue_tbl, 0, sizeof(queue_tbl_t));
216  memset(queue_pool, 0, sizeof(queue_pool_t));
217  memset(queue_pool_static, 0, sizeof(queue_pool_t));
218  env_atomic32_init(&em_shm->queue_count);
219 
220  if (read_config_file())
221  return EM_ERR_LIB_FAILED;
222 
223  /* Retieve and store the ODP queue capabilities into 'queue_tbl' */
224  ret = odp_queue_capability(odp_queue_capa);
225  RETURN_ERROR_IF(ret != 0, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
226  "odp_queue_capability():%d failed", ret);
227 
228  /* Retieve and store the ODP schedule capabilities into 'queue_tbl' */
229  ret = odp_schedule_capability(odp_sched_capa);
230  RETURN_ERROR_IF(ret != 0, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
231  "odp_schedule_capability():%d failed", ret);
232 
233  RETURN_ERROR_IF(odp_queue_capa->max_queues < EM_MAX_QUEUES,
234  EM_ERR_TOO_LARGE, EM_ESCOPE_INIT,
235  "EM_MAX_QUEUES:%i > odp-max-queues:%u",
236  EM_MAX_QUEUES, odp_queue_capa->max_queues);
237 
238  /* Initialize the queue element table */
239  for (int i = 0; i < EM_MAX_QUEUES; i++)
240  queue_tbl->queue_elem[i].queue = queue_idx2hdl(i);
241 
242  /* Initialize the static queue pool */
243  min = queue_id2idx(EM_QUEUE_STATIC_MIN);
244  max = queue_id2idx(LAST_INTERNAL_QUEUE);
245  if (queue_pool_init(queue_tbl, queue_pool_static, min, max) != 0)
246  return EM_ERR_LIB_FAILED;
247 
248  /* Initialize the dynamic queue pool */
249  min = queue_id2idx(FIRST_DYN_QUEUE);
250  max = queue_id2idx(LAST_DYN_QUEUE);
251  if (queue_pool_init(queue_tbl, queue_pool, min, max) != 0)
252  return EM_ERR_LIB_FAILED;
253 
254  /* Initialize priority mapping, adapt to values from ODP */
255  min = odp_schedule_min_prio();
256  max = odp_schedule_max_prio();
257  em_shm->queue_prio.num_runtime = max - min + 1;
258  ret = queue_init_prio_map(min, max, em_shm->queue_prio.num_runtime);
259  RETURN_ERROR_IF(ret != 0, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
260  "mapping odp priorities failed: %d", ret);
261  return EM_OK;
262 }

265  * Queue inits done during EM core local init (once at startup on each core).
266  *
267  * Initialize event storage for queues of type 'EM_QUEUE_TYPE_LOCAL'.
268  */
269 em_status_t
270 queue_init_local(void)
271 {
272  int prio;
273  int core;
274  char name[20];
275  odp_queue_param_t param;
276  em_locm_t *const locm = &em_locm;
277 
278  core = em_core_id();
279 
280  odp_queue_param_init(&param);
281  param.type = ODP_QUEUE_TYPE_PLAIN;
282  param.enq_mode = ODP_QUEUE_OP_MT_UNSAFE;
283  param.deq_mode = ODP_QUEUE_OP_MT_UNSAFE;
284  param.order = ODP_QUEUE_ORDER_IGNORE;
285  param.size = 512;
286 
287  locm->local_queues.empty = 1;
288 
289  for (prio = 0; prio < EM_QUEUE_PRIO_NUM; prio++) {
290  snprintf(name, sizeof(name),
291  "local-q:c%02d:prio%d", core, prio);
292  name[sizeof(name) - 1] = '\0';
293 
294  locm->local_queues.prio[prio].empty_prio = 1;
295  locm->local_queues.prio[prio].queue =
296  odp_queue_create(name, &param);
297  if (unlikely(locm->local_queues.prio[prio].queue ==
298  ODP_QUEUE_INVALID))
299  return EM_ERR_ALLOC_FAILED;
300  }
301 
302  memset(&locm->output_queue_track, 0,
303  sizeof(locm->output_queue_track));
304 
305  return EM_OK;
306 }

308 /**
309  * Queue termination done during em_term_core().
310  *
311  * Flush & destroy event storage for queues of type 'EM_QUEUE_TYPE_LOCAL'.
312  */
313 em_status_t
314 queue_term_local(void)
315 {
316  em_event_t event;
317  event_hdr_t *ev_hdr;
318  em_status_t stat = EM_OK;
319  int ret;
320  bool esv_ena = esv_enabled();
321 
322  /* flush all events */
323  while ((ev_hdr = local_queue_dequeue()) != NULL) {
324  event = event_hdr_to_event(ev_hdr);
325  if (esv_ena)
326  event = evstate_em2usr(event, ev_hdr,
327  EVSTATE__TERM_CORE__QUEUE_LOCAL);
328  em_free(event);
329  }
330 
331  for (int prio = 0; prio < EM_QUEUE_PRIO_NUM; prio++) {
332  ret = odp_queue_destroy(em_locm.local_queues.prio[prio].queue);
333  if (unlikely(ret != 0))
334  stat = EM_ERR_LIB_FAILED;
335  }
336 
337  return stat;
338 }

340 /**
341  * Allocate a new EM queue
342  *
343  * @param queue EM queue handle if a specific EM queue is requested,
344  * EM_QUEUE_UNDEF if any EM queue will do.
345  *
346  * @return EM queue handle
347  * @retval EM_QUEUE_UNDEF on failure
348  */
349 static inline em_queue_t
350 queue_alloc(em_queue_t queue, const char **err_str)
351 {
352  queue_elem_t *queue_elem;
353  objpool_elem_t *queue_pool_elem;
354 
355  if (queue == EM_QUEUE_UNDEF) {
356  /*
357  * Allocate a dynamic queue, i.e. take next available
358  */
359  queue_pool_elem = objpool_rem(&em_shm->queue_pool.objpool,
360  em_core_id());
361  if (unlikely(queue_pool_elem == NULL)) {
362  *err_str = "queue pool element alloc failed!";
363  return EM_QUEUE_UNDEF;
364  }
365  queue_elem = queue_poolelem2queue(queue_pool_elem);
366  } else {
367  /*
368  * Allocate a specific static-handle queue, handle given
369  */
370  internal_queue_t iq;
371 
372  iq.queue = queue;
373  if (iq.queue_id < EM_QUEUE_STATIC_MIN ||
374  iq.queue_id > LAST_INTERNAL_QUEUE) {
375  *err_str = "queue handle not from static range!";
376  return EM_QUEUE_UNDEF;
377  }
378 
379  queue_elem = queue_elem_get(queue);
380  if (unlikely(queue_elem == NULL)) {
381  *err_str = "queue_elem ptr NULL!";
382  return EM_QUEUE_UNDEF;
383  }
384  /* Verify that the queue is not allocated */
385  if (queue_allocated(queue_elem)) {
386  *err_str = "queue already allocated!";
387  return EM_QUEUE_UNDEF;
388  }
389  /* Remove the queue from the pool */
390  int ret = objpool_rem_elem(&em_shm->queue_pool_static.objpool,
391  &queue_elem->queue_pool_elem);
392  if (unlikely(ret != 0)) {
393  *err_str = "static queue pool element alloc failed!";
394  return EM_QUEUE_UNDEF;
395  }
396  }
397 
398  env_atomic32_inc(&em_shm->queue_count);
399  return queue_elem->queue;
400 }

402 static inline em_status_t
403 queue_free(em_queue_t queue)
404 {
405  queue_elem_t *const queue_elem = queue_elem_get(queue);
406  objpool_t *objpool;
407  internal_queue_t iq;
408 
409  iq.queue = queue;
410 
411  if (unlikely(queue_elem == NULL))
412  return EM_ERR_BAD_ID;
413 
414  if (iq.queue_id >= EM_QUEUE_STATIC_MIN &&
415  iq.queue_id <= LAST_INTERNAL_QUEUE)
416  objpool = &em_shm->queue_pool_static.objpool;
417  else
418  objpool = &em_shm->queue_pool.objpool;
419 
420  queue_elem->state = EM_QUEUE_STATE_INVALID;
421 
422  objpool_add(objpool,
423  queue_elem->queue_pool_elem.subpool_idx,
424  &queue_elem->queue_pool_elem);
425 
426  env_atomic32_dec(&em_shm->queue_count);
427  return EM_OK;
428 }

430 static int
431 queue_create_check_sched(const queue_setup_t *setup, const char **err_str)
432 {
433  const queue_group_elem_t *queue_group_elem = NULL;
434  const atomic_group_elem_t *ag_elem = NULL;
435 
436  queue_group_elem = queue_group_elem_get(setup->queue_group);
437  /* scheduled queues are always associated with a queue group */
438  if (unlikely(queue_group_elem == NULL || !queue_group_allocated(queue_group_elem))) {
439  *err_str = "Invalid queue group!";
440  return -1;
441  }
442 
443  if (setup->atomic_group != EM_ATOMIC_GROUP_UNDEF) {
444  ag_elem = atomic_group_elem_get(setup->atomic_group);
445  if (unlikely(ag_elem == NULL || !atomic_group_allocated(ag_elem))) {
446  *err_str = "Invalid atomic group!";
447  return -1;
448  }
449  }
450 
451  if (unlikely(setup->prio >= EM_QUEUE_PRIO_NUM)) {
452  *err_str = "Invalid queue priority!";
453  return -1;
454  }
455  return 0;
456 }

458 static int
459 queue_create_check_args(const queue_setup_t *setup, const char **err_str)
460 {
461  /* scheduled queue */
462  if (setup->type == EM_QUEUE_TYPE_ATOMIC ||
463  setup->type == EM_QUEUE_TYPE_PARALLEL ||
464  setup->type == EM_QUEUE_TYPE_PARALLEL_ORDERED)
465  return queue_create_check_sched(setup, err_str);
466 
467  /* other queue types */
468  switch (setup->type) {
469  case EM_QUEUE_TYPE_UNSCHEDULED:
470  /* API arg checks for unscheduled queues */
471  if (unlikely(setup->prio != EM_QUEUE_PRIO_UNDEF)) {
472  *err_str = "Invalid priority for unsched queue!";
473  return -1;
474  }
475  if (unlikely(setup->queue_group != EM_QUEUE_GROUP_UNDEF)) {
476  *err_str = "Queue group not used with unsched queues!";
477  return -1;
478  }
479  if (unlikely(setup->atomic_group != EM_ATOMIC_GROUP_UNDEF)) {
480  *err_str = "Atomic group not used with unsched queues!";
481  return -1;
482  }
483  break;
484 
485  case EM_QUEUE_TYPE_LOCAL:
486  /* API arg checks for local queues */
487  if (unlikely(setup->queue_group != EM_QUEUE_GROUP_UNDEF)) {
488  *err_str = "Queue group not used with local queues!";
489  return -1;
490  }
491  if (unlikely(setup->atomic_group != EM_ATOMIC_GROUP_UNDEF)) {
492  *err_str = "Atomic group not used with local queues!";
493  return -1;
494  }
495  if (unlikely(setup->prio >= EM_QUEUE_PRIO_NUM)) {
496  *err_str = "Invalid queue priority!";
497  return -1;
498  }
499  break;
500 
501  case EM_QUEUE_TYPE_OUTPUT:
502  /* API arg checks for output queues */
503  if (unlikely(setup->queue_group != EM_QUEUE_GROUP_UNDEF)) {
504  *err_str = "Queue group not used with output queues!";
505  return -1;
506  }
507  if (unlikely(setup->atomic_group != EM_ATOMIC_GROUP_UNDEF)) {
508  *err_str = "Atomic group not used with output queues!";
509  return -1;
510  }
511  if (unlikely(setup->conf == NULL ||
512  setup->conf->conf_len < sizeof(em_output_queue_conf_t) ||
513  setup->conf->conf == NULL)) {
514  *err_str = "Invalid output queue conf";
515  return -1;
516  }
517  break;
518 
519  default:
520  *err_str = "Unknown queue type";
521  return -1;
522  }
523 
524  return 0;
525 }

527 /**
528  * Create an EM queue: alloc, setup and add to queue group list
529  */
530 em_queue_t
531 queue_create(const char *name, em_queue_type_t type, em_queue_prio_t prio,
532  em_queue_group_t queue_group, em_queue_t queue_req,
533  em_atomic_group_t atomic_group, const em_queue_conf_t *conf,
534  const char **err_str)
535 {
536  int err;
537 
538  /* Use default EM queue conf if none given */
539  if (conf == NULL)
540  conf = &default_queue_conf;
541 
542  queue_setup_t setup = {.name = name, .type = type, .prio = prio,
543  .atomic_group = atomic_group,
544  .queue_group = queue_group, .conf = conf};
545 
546  err = queue_create_check_args(&setup, err_str);
547  if (err) {
548  /* 'err_str' set by queue_create_check_args() */
549  return EM_QUEUE_UNDEF;
550  }
551 
552  /*
553  * Allocate the queue handle and obtain the corresponding queue-element
554  */
555  const char *alloc_err_str = "";
556 
557  em_queue_t queue = queue_alloc(queue_req, &alloc_err_str);
558 
559  if (unlikely(queue == EM_QUEUE_UNDEF)) {
560  *err_str = alloc_err_str;
561  return EM_QUEUE_UNDEF;
562  }
563  if (unlikely(queue_req != EM_QUEUE_UNDEF && queue_req != queue)) {
564  queue_free(queue);
565  *err_str = "Failed to allocate the requested queue!";
566  return EM_QUEUE_UNDEF;
567  }
568 
569  queue_elem_t *queue_elem = queue_elem_get(queue);
570 
571  if (unlikely(!queue_elem)) {
572  queue_free(queue);
573  *err_str = "Queue elem NULL!";
574  return EM_QUEUE_UNDEF;
575  }
576 
577  /*
578  * Setup/configure the queue
579  */
580  err = queue_setup(queue_elem, &setup, err_str);
581  if (unlikely(err)) {
582  queue_free(queue);
583  /* 'err_str' set by queue_setup() */
584  return EM_QUEUE_UNDEF;
585  }
586 
587  return queue;
588 }

590 em_status_t
591 queue_delete(queue_elem_t *const queue_elem)
592 {
593  queue_state_t old_state;
594  queue_state_t new_state;
595  em_status_t ret;
596  em_queue_t queue = queue_elem->queue;
597  em_queue_type_t type = queue_elem->type;
598 
599  if (unlikely(!queue_allocated(queue_elem)))
600  return EM_ERR_BAD_STATE;
601 
602  old_state = queue_elem->state;
603  new_state = EM_QUEUE_STATE_INVALID;
604 
605  if (type != EM_QUEUE_TYPE_UNSCHEDULED &&
606  type != EM_QUEUE_TYPE_OUTPUT) {
607  /* verify scheduled queue state transition */
608  ret = queue_state_change__check(old_state, new_state,
609  0/*!is_setup*/);
610  RETURN_ERROR_IF(ret != EM_OK, ret, EM_ESCOPE_QUEUE_DELETE,
611  "EM-Q:%" PRI_QUEUE " inv. state change:%d=>%d",
612  queue, old_state, new_state);
613  }
614 
615  if (type != EM_QUEUE_TYPE_UNSCHEDULED &&
616  type != EM_QUEUE_TYPE_LOCAL &&
617  type != EM_QUEUE_TYPE_OUTPUT) {
618  queue_group_elem_t *const queue_group_elem =
619  queue_group_elem_get(queue_elem->queue_group);
620 
621  RETURN_ERROR_IF(queue_group_elem == NULL ||
622  !queue_group_allocated(queue_group_elem),
623  EM_ERR_BAD_ID, EM_ESCOPE_QUEUE_DELETE,
624  "Invalid queue group: %" PRI_QGRP "",
625  queue_elem->queue_group);
626 
627  /* Remove the queue from the queue group list */
628  queue_group_rem_queue_list(queue_group_elem, queue_elem);
629  }
630 
631  if (type == EM_QUEUE_TYPE_OUTPUT) {
632  env_spinlock_t *const lock = &queue_elem->output.lock;
633  q_elem_output_t *const q_out = &queue_elem->output;
634 
635  env_spinlock_lock(lock);
636  /* Drain any remaining events from the output queue */
637  output_queue_drain(queue_elem);
638  env_spinlock_unlock(lock);
639 
640  /* delete the fn-args storage if allocated in create */
641  if (q_out->output_fn_args_event != EM_EVENT_UNDEF) {
642  em_free(q_out->output_fn_args_event);
643  q_out->output_fn_args_event = EM_EVENT_UNDEF;
644  }
645  }
646 
647  if (queue_elem->odp_queue != ODP_QUEUE_INVALID) {
648  int err = odp_queue_destroy(queue_elem->odp_queue);
649 
650  RETURN_ERROR_IF(err, EM_ERR_LIB_FAILED, EM_ESCOPE_QUEUE_DELETE,
651  "EM-Q:%" PRI_QUEUE ":odp_queue_destroy(" PRIu64 "):%d",
652  queue, odp_queue_to_u64(queue_elem->odp_queue),
653  err);
654  }
655 
656  queue_elem->odp_queue = ODP_QUEUE_INVALID;
657 
658  /* Zero queue name */
659  em_shm->queue_tbl.name[queue_hdl2idx(queue)][0] = '\0';
660 
661  /* Remove the queue from the atomic group it belongs to, if any */
662  atomic_group_remove_queue(queue_elem);
663 
664  return queue_free(queue);
665 }

1115  * |old_state|is_setup |is_teardown|

1117  * |INVALID | INIT | (NULL) |

1130

1132  state_diff = new_state - old_state;
1133  else
1134  state_diff = old_state - new_state;

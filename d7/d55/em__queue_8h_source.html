100 /**
101  * Enqueue multiple events into an unscheduled queue.
102  * Internal func, application should use em_send_multi() instead.
103  */
104 static inline unsigned int
105 queue_unsched_enqueue_multi(const em_event_t events[], int num,
106  const queue_elem_t *const q_elem)
107 {
108  odp_event_t odp_events[num];
109  odp_queue_t odp_queue = q_elem->odp_queue;
110  int ret;
111 
112  if (unlikely(EM_CHECK_LEVEL > 1 && odp_queue == ODP_QUEUE_INVALID))
113  return 0;
114 
115  if (unlikely(EM_CHECK_LEVEL > 0 &&
116  q_elem->state != EM_QUEUE_STATE_UNSCHEDULED))
117  return 0;
118 
119  events_em2odp(events, odp_events, num);
120 
121  ret = odp_queue_enq_multi(odp_queue, odp_events, num);
122  if (unlikely(ret < 0))
123  return 0;
124 
125  return ret;
126 }

128 /**
129  * Enqueue en event into an unscheduled queue.
130  * Internal func, application should use em_send() instead.
131  */
132 static inline em_status_t
133 queue_unsched_enqueue(em_event_t event, const queue_elem_t *const q_elem)
134 {
135  odp_event_t odp_event = event_em2odp(event);
136  odp_queue_t odp_queue = q_elem->odp_queue;
137  int ret;
138 
139  if (unlikely(EM_CHECK_LEVEL > 1 &&
140  (odp_event == ODP_EVENT_INVALID ||
141  odp_queue == ODP_QUEUE_INVALID)))
142  return EM_ERR_NOT_FOUND;
143 
144  if (unlikely(EM_CHECK_LEVEL > 0 &&
145  q_elem->state != EM_QUEUE_STATE_UNSCHEDULED))
146  return EM_ERR_BAD_STATE;
147 
148  ret = odp_queue_enq(odp_queue, odp_event);
149  if (unlikely(EM_CHECK_LEVEL > 0 && ret != 0))
150  return EM_ERR_LIB_FAILED;
151 
152  return EM_OK;
153 }

222 /** Returns queue element associated with queued id 'queue' */
223 static inline queue_elem_t *
224 queue_elem_get(const em_queue_t queue)
225 {
226  int queue_idx;
227  internal_queue_t iq;
228  queue_elem_t *queue_elem;
229 
230  iq.queue = queue;

235  return NULL;
236 
237  queue_elem = &em_shm->queue_tbl.queue_elem[queue_idx];

311 static inline event_hdr_t *
312 local_queue_dequeue(void)
313 {
314  em_locm_t *const locm = &em_locm;
315  odp_queue_t local_queue;
316  odp_event_t odp_event;
317  em_event_t event;
318  event_hdr_t *ev_hdr;
319  em_queue_prio_t prio;
320  int i;
321 
322  if (locm->local_queues.empty)
323  return NULL;
324 
325  prio = EM_QUEUE_PRIO_NUM - 1;
326  for (i = 0; i < EM_QUEUE_PRIO_NUM; i++) {
327  /* from hi to lo prio: next prio if local queue is empty */
328  if (locm->local_queues.prio[prio].empty_prio) {
329  prio--;
330  continue;
331  }
332 
333  local_queue = locm->local_queues.prio[prio].queue;
334  odp_event = odp_queue_deq(local_queue);
335 
336  if (odp_event != ODP_EVENT_INVALID) {
337  event = event_odp2em(odp_event); /* .evgen not set */
338  ev_hdr = event_to_hdr(event);
339  return ev_hdr;
340  }
341 
342  locm->local_queues.prio[prio].empty_prio = 1;
343  prio--;
344  }
345 
346  locm->local_queues.empty = 1;
347  return NULL;
348 }

350 static inline int
351 next_local_queue_events(em_event_t ev_tbl[/*out*/], int num_events)
352 {
353  em_locm_t *const locm = &em_locm;
354 
355  if (locm->local_queues.empty)
356  return 0;
357 
358  /* use same output-array: odp_evtbl[] = ev_tbl[] */
359  odp_event_t *const odp_evtbl = (odp_event_t *)ev_tbl;
360 
361  em_queue_prio_t prio;
362  odp_queue_t local_queue;
363  int num;
364 
365  prio = EM_QUEUE_PRIO_NUM - 1;
366  for (int i = 0; i < EM_QUEUE_PRIO_NUM; i++) {
367  /* from hi to lo prio: next prio if local queue is empty */
368  if (locm->local_queues.prio[prio].empty_prio) {
369  prio--;
370  continue;
371  }
372 
373  local_queue = locm->local_queues.prio[prio].queue;
374  num = odp_queue_deq_multi(local_queue, odp_evtbl/*out=ev_tbl*/,
375  num_events);
376  if (num > 0)
377  return num; /* odp_evtbl[] = ev_tbl[], .evgen not set */
378 
379  locm->local_queues.prio[prio].empty_prio = 1;
380  prio--;
381  }
382 
383  locm->local_queues.empty = 1;
384  return 0;
385 }

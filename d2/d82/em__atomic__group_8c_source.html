33 /**
34  * Atomic group inits done at global init (once at startup on one core)
35  */
36 em_status_t
37 atomic_group_init(atomic_group_tbl_t *const atomic_group_tbl,
38  atomic_group_pool_t *const atomic_group_pool)
39 {
40  atomic_group_elem_t *atomic_group_elem;
41  const int cores = em_core_count();
42  int ret;
43 
44  memset(atomic_group_tbl, 0, sizeof(atomic_group_tbl_t));
45  memset(atomic_group_pool, 0, sizeof(atomic_group_pool_t));
46  env_atomic32_init(&em_shm->atomic_group_count);
47 
48  for (int i = 0; i < EM_MAX_ATOMIC_GROUPS; i++) {
49  em_atomic_group_t agrp = agrp_idx2hdl(i);
50  atomic_group_elem_t *const agrp_elem =
51  atomic_group_elem_get(agrp);
52 
53  if (unlikely(!agrp_elem))
54  return EM_ERR_BAD_POINTER;
55 
56  agrp_elem->atomic_group = agrp; /* store handle */
57 
58  /* Init list and lock */
59  env_spinlock_init(&agrp_elem->lock);
60  list_init(&agrp_elem->qlist_head);
61  env_atomic32_init(&agrp_elem->num_queues);
62  }
63 
64  ret = objpool_init(&atomic_group_pool->objpool, cores);
65  if (ret != 0)
66  return EM_ERR_LIB_FAILED;
67 
68  for (int i = 0; i < EM_MAX_ATOMIC_GROUPS; i++) {
69  atomic_group_elem = &atomic_group_tbl->ag_elem[i];
70  objpool_add(&atomic_group_pool->objpool, i % cores,
71  &atomic_group_elem->atomic_group_pool_elem);
72  }
73 
74  return EM_OK;
75 }

84 /**
85  * Dynamic atomic group allocation
86  */
87 em_atomic_group_t
88 atomic_group_alloc(void)
89 {
90  const atomic_group_elem_t *ag_elem;
91  const objpool_elem_t *ag_p_elem;
92 
93  ag_p_elem = objpool_rem(&em_shm->atomic_group_pool.objpool,
94  em_core_id());
95 
96  if (unlikely(ag_p_elem == NULL))
97  return EM_ATOMIC_GROUP_UNDEF;
98 
99  ag_elem = ag_pool_elem2ag_elem(ag_p_elem);
100 
101  env_atomic32_inc(&em_shm->atomic_group_count);
102  return ag_elem->atomic_group;
103 }

142 static inline int
143 ag_local_processing_ended(atomic_group_elem_t *const ag_elem)
144 {
145  em_locm_t *const locm = &em_locm;
146 
147  /*
148  * Check if atomic group processing has ended for this core, meaning
149  * the application called em_atomic_processing_end()
150  */
151  if (locm->atomic_group_released) {
152  locm->atomic_group_released = 0;
153  /*
154  * Try to acquire the atomic group lock and continue processing.
155  * It is possible that another core has acquired the lock
156  */
157  if (env_spinlock_trylock(&ag_elem->lock))
158  return 0;
159  else
160  return 1;
161  }
162 
163  return 0;
164 }

166 static inline int
167 ag_internal_enq(const atomic_group_elem_t *ag_elem, const em_event_t ev_tbl[],
168  const int num_events, const em_queue_prio_t priority)
169 {
170  odp_event_t odp_ev_tbl[num_events];
171  odp_queue_t plain_q;
172  int ret;
173 
174  if (priority == EM_QUEUE_PRIO_HIGHEST)
175  plain_q = ag_elem->internal_queue.hi_prio;
176  else
177  plain_q = ag_elem->internal_queue.lo_prio;
178 
179  events_em2odp(ev_tbl, odp_ev_tbl, num_events);
180 
181  /* Enqueue events to internal queue */
182  ret = odp_queue_enq_multi(plain_q, odp_ev_tbl, num_events);
183  if (unlikely(ret != num_events))
184  return ret > 0 ? ret : 0;
185 
186  return num_events;
187 }

189 static inline int
190 ag_internal_deq(const atomic_group_elem_t *ag_elem, em_event_t ev_tbl[/*out*/],
191  const int num_events)
192 {
193  /*
194  * Dequeue odp events directly into ev_tbl[].
195  * The function call_eo_receive_fn/multi() will convert to
196  * EM events with event-generation counts, if ESV is enabled,
197  * before passing the events to the user EO.
198  */
199  odp_event_t *const ag_ev_tbl = (odp_event_t *const)ev_tbl;
200  int hi_cnt;
201  int lo_cnt;
202 
203  /* hi-prio events */
204  hi_cnt = odp_queue_deq_multi(ag_elem->internal_queue.hi_prio,
205  ag_ev_tbl/*out*/, num_events);
206  if (hi_cnt == num_events || hi_cnt < 0)
207  return hi_cnt;
208 
209  /* ...then lo-prio events */
210  lo_cnt = odp_queue_deq_multi(ag_elem->internal_queue.lo_prio,
211  &ag_ev_tbl[hi_cnt]/*out*/,
212  num_events - hi_cnt);
213  if (unlikely(lo_cnt < 0))
214  return hi_cnt;
215 
216  return hi_cnt + lo_cnt;
217 }

219 void
220 atomic_group_dispatch(em_event_t ev_tbl[], event_hdr_t *const ev_hdr_tbl[],
221  const int num_events, queue_elem_t *const q_elem)
222 {
223  atomic_group_elem_t *const ag_elem =
224  atomic_group_elem_get(q_elem->atomic_group);
225  const em_queue_prio_t priority = q_elem->priority;
226  int enq_cnt;
227 
228  /* Insert the original q_elem pointer into the event header */
229  for (int i = 0; i < num_events; i++)
230  ev_hdr_tbl[i]->q_elem = q_elem;
231 
232  /* Enqueue the scheduled events into the atomic group internal queue */
233  enq_cnt = ag_internal_enq(ag_elem, ev_tbl, num_events, priority);
234 
235  if (unlikely(enq_cnt < num_events)) {
236  em_free_multi(&ev_tbl[enq_cnt], num_events - enq_cnt);
237  /*
238  * Use dispatch escope since this func is called only from
239  * dispatch_round() => atomic_group_dispatch()
240  */
241  INTERNAL_ERROR(EM_ERR_OPERATION_FAILED, EM_ESCOPE_DISPATCH,
242  "Atomic group:%" PRI_AGRP " internal enqueue:\n"
243  " num_events:%d enq_cnt:%d",
244  ag_elem->atomic_group, num_events, enq_cnt);
245  }
246 
247  /*
248  * Try to acquire the atomic group lock - if not available then some
249  * other core is already handling the same atomic group.
250  */
251  if (!env_spinlock_trylock(&ag_elem->lock))
252  return;
253 
254  em_locm_t *const locm = &em_locm;
255 
256  /* hint */
257  odp_schedule_release_atomic();
258 
259  locm->atomic_group_released = 0;
260  /*
261  * Loop until no more events or until atomic processing end.
262  * Events in the ag_elem->internal_queue:s have been scheduled
263  * already once and should be dispatched asap.
264  */
265  em_event_t deq_ev_tbl[EM_SCHED_AG_MULTI_MAX_BURST];
266  event_hdr_t *deq_hdr_tbl[EM_SCHED_AG_MULTI_MAX_BURST];
267 
268  do {
269  int deq_cnt = ag_internal_deq(ag_elem, deq_ev_tbl /*out*/,
270  EM_SCHED_AG_MULTI_MAX_BURST);
271 
272  if (unlikely(deq_cnt <= 0)) {
273  env_spinlock_unlock(&ag_elem->lock);
274  /* return if no more events available */
275  return;
276  }
277 
278  locm->event_burst_cnt = deq_cnt;
279  event_to_hdr_multi(deq_ev_tbl, deq_hdr_tbl /*out*/, deq_cnt);
280  int tbl_idx = 0; /* index into 'deq_hdr_tbl[]' */
281 
282  /*
283  * Dispatch in batches of 'batch_cnt' events.
284  * Each batch contains events from the same atomic queue.
285  */
286  do {
287  queue_elem_t *const batch_qelem =
288  deq_hdr_tbl[tbl_idx]->q_elem;
289  int batch_cnt = 1;
290 
291  for (int i = tbl_idx + 1; i < deq_cnt &&
292  deq_hdr_tbl[i]->q_elem == batch_qelem; i++) {
293  batch_cnt++;
294  }
295 
296  dispatch_events(&deq_ev_tbl[tbl_idx],
297  &deq_hdr_tbl[tbl_idx],
298  batch_cnt, batch_qelem);
299  tbl_idx += batch_cnt;
300  } while (tbl_idx < deq_cnt);
301 
302  } while (!ag_local_processing_ended(ag_elem));
303 }

106 static inline int
107 pool_find_subpool(const mpool_elem_t *const pool_elem, size_t size)
108 {
109  int subpool = 0;
110 
111  if (EM_MAX_SUBPOOLS > 1) { /* Compile time option */
112  int i;
113  /* Find the optimal subpool to allocate the event from */
114  for (i = 0; i < pool_elem->num_subpools &&
115  size > pool_elem->size[i]; i++)
116  ;
117 
118  if (unlikely(i >= pool_elem->num_subpools))
119  return -1;
120 
121  subpool = i;
122  }
123 
124  return subpool;
125 }

130 /**
131  * Get the EM event-pool that an odp-pool belongs to.
132  *
133  * An EM event-pool consists of up to EM_MAX_SUBPOOLS subpools (that are
134  * odp-pools) - a table (em_shm->mpool_tbl.pool_odp2em[]) contains the
135  * mapping and is populated during em_pool_create() calls.
136  */
137 static inline em_pool_t
138 pool_odp2em(odp_pool_t odp_pool)
139 {
140  /*
141  * 'idx' is in the range: 0 to odp_pool_max_index(), which is smaller
142  * than the length of the em_shm->mpool_tbl.pool_odp2em[] array
143  * (verified at startup in pool_init()).
144  */
145  int idx = odp_pool_index(odp_pool);
146 
147  if (unlikely(idx < 0))
148  return EM_POOL_UNDEF;
149 
150  return em_shm->mpool_tbl.pool_odp2em[idx];
151 }

153 /**
154  * Increment event-pool statistics counter for 'subpool'.
155  *
156  * Increment the alloc-count.

158 static inline void
159 poolstat_inc(em_pool_t pool, int subpool, uint64_t cnt)
160 {
161  const int pool_idx = pool_hdl2idx(pool);
162  mpool_statistics_t *const pstat =
163  &em_shm->mpool_tbl.pool_stat_core[em_locm.core_id];
164 
165  pstat->stat[pool_idx][subpool].alloc += cnt;

184 /**
185  * Decrement event-pool statistics counter for 'subpool'
186  *
187  * Instead of decrementing the alloc-count, a separate free-count is maintained

195  &em_shm->mpool_tbl.pool_stat_core[em_locm.core_id];

197  pstat->stat[pool_idx][subpool].free += cnt;

216 /**
217  * Decrement pool-stats for the subpool the events are allocated from.
218  */
219 static inline void
220 poolstat_dec_evhdr_multi(event_hdr_t *ev_hdrs[], const int num)
221 {
222  /* Update pool statistcs */
223  int idx = 0; /* index into ev_hdrs[] */
224  int dec;
225  int i;
226 
227  do {
228  for (; idx < num &&
229  ev_hdrs[idx]->pool == EM_POOL_UNDEF; idx++)
230  ; /* skip events from external pools */
231  if (idx >= num)
232  break;
233 
234  em_pool_t pool = ev_hdrs[idx]->pool;
235  int subpool = ev_hdrs[idx]->subpool;
236 
237  for (i = idx + 1; i < num &&
238  ev_hdrs[i]->pool == pool &&
239  ev_hdrs[i]->subpool == subpool; i++)
240  ; /* count events from same pool */
241  dec = i - idx;
242  idx = i;
243  if (likely(valid_pool(pool)))
244  poolstat_dec(pool, subpool, dec);
245  } while (idx < num);
246 }

280 /**
281  * Decrement pool-stats, on output from EM, for the subpool the event is
282  * allocated from.
283  *
284  * Mark ev_hdr->pool = EM_POOL_UNDEF to avoid further pool-stat decrements
285  * should the event be sent back into EM from 'output'.
286  *
287  * Here 'output' means sending to en external queue (event chaining) or
288  * sending out-of-EM to a queue of type EM_QUEUE_TYPE_OUTPUT.
289  */
290 static inline void
291 poolstat_dec_evhdr_output(event_hdr_t *const ev_hdr)
292 {
293  em_pool_t pool = ev_hdr->pool;
294  int subpool;
295 
296  if (pool != EM_POOL_UNDEF) {
297  subpool = ev_hdr->subpool;
298  if (likely(valid_pool(pool)))
299  poolstat_dec(pool, subpool, 1);
300  /* No further pool-statistics for this event */
301  ev_hdr->pool = EM_POOL_UNDEF;
302  }
303 }

305 /**
306  * Decrement pool-stats, on output from EM, for the subpool the events are
307  * allocated from.
308  *
309  * Mark ev_hdr->pool = EM_POOL_UNDEF for each event to avoid further pool-stat
310  * decrements should any event(s) be sent back into EM from 'output'.
311  *
312  * Here 'output' means sending to an external queue (event chaining) or
313  * sending out-of-EM to a queue of type EM_QUEUE_TYPE_OUTPUT.
314  */
315 static inline void
316 poolstat_dec_evhdr_multi_output(event_hdr_t *const ev_hdrs[], const int num)
317 {
318  /* Update pool statistcs */
319  int idx = 0; /* index into ev_hdrs[] */
320  int dec;
321  int i;
322 
323  do {
324  for (; idx < num &&
325  ev_hdrs[idx]->pool == EM_POOL_UNDEF; idx++)
326  ; /* skip events from external pools */
327  if (idx >= num)
328  break;
329 
330  em_pool_t pool = ev_hdrs[idx]->pool;
331  int subpool = ev_hdrs[idx]->subpool;
332 
333  /*
334  * Count events from same pool and
335  * mark no further pool-statistics for output events (=UNDEF)
336  */
337  ev_hdrs[idx]->pool = EM_POOL_UNDEF;
338  for (i = idx + 1; i < num &&
339  ev_hdrs[i]->pool == pool &&
340  ev_hdrs[i]->subpool == subpool; i++)
341  ev_hdrs[i]->pool = EM_POOL_UNDEF;
342 
343  dec = i - idx;
344  idx = i;
345  if (likely(valid_pool(pool)))
346  poolstat_dec(pool, subpool, dec);
347  } while (idx < num);
348 }

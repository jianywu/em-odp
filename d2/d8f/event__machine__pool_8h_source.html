36 /**
37  * @file
38  * @defgroup em_pool Event Pool
39  * Event Machine event pool related services
40  * @{
41  *
42  * EM events are allocated from event pools with em_alloc() and freed back into
43  * them with em_free(). The event pools to be allocated from must first be
44  * created with em_pool_create().
45  *
46  * Note that EM should always provide at least one pool, i.e. 'EM_POOL_DEFAULT'
47  * that can be used for event allocation. The default pool creation is platform
48  * specific: it can e.g. be done in 'em_init(conf)' with an appropriate
49  * default pool config passed via the 'conf' (em_conf_t) parameter.
50  * Further event pools should be created explicitly with em_pool_create().
51  *
52  * Event pool APIs for pool deletion, lookup, iteration etc. are listed below.
53  *
54  * The 'em_pool_cfg_t' type given to em_pool_create() is HW/platform specific
55  * and is defined in event_machine_hw_types.h
56  *
57  * Do not include this from the application, event_machine.h will
58  * do it for you.
59  */

65 /**
66  * EM pool configuration
67  *
68  * Configuration of an EM event pool consisting of up to 'EM_MAX_SUBPOOLS'
69  * subpools, each supporting a specific event payload size. Event allocation,
70  * i.e. em_alloc(), will use the subpool that provides the best fit for the
71  * requested size.
72  *
73  * Example usage:
74  * @code
75  * em_pool_cfg_t pool_cfg;
76  *
77  * em_pool_cfg_init(&pool_cfg); // init with default values (mandatory)
78  * pool_cfg.event_type = EM_EVENT_TYPE_PACKET;
79  * ...
80  * pool_cfg.num_subpools = 4;
81  * pool_cfg.subpool[0].size = X;
82  * pool_cfg.subpool[0].num = Y;
83  * pool_cfg.subpool[0].cache_size = Z;
84  * ...
85  * pool = em_pool_create(..., &pool_cfg);
86  * @endcode
87  */
88 typedef struct {
89  /**
90  * Event type determines the pool type used:
91  * - EM_EVENT_TYPE_SW creates subpools of type 'ODP_POOL_BUFFER'
92  * This kind of EM pool CANNOT be used to create events of major
93  * type EM_EVENT_TYPE_PACKET.
94  * - EM_EVENT_TYPE_PACKET creates subpools of type 'ODP_POOL_PACKET'
95  * This kind of EM pool can be used for events of all kinds.
96  * @note Only major types are considered here, setting minor is error
97  */
98  em_event_type_t event_type;
99  /**
100  * Alignment offset in bytes for the event payload start address
101  * (for all events allocated from this EM pool).
102  *
103  * The default EM event payload start address alignment is a
104  * power-of-two that is at minimum 32 bytes (i.e. 32 B, 64 B, 128 B etc.
105  * depending on e.g. target cache-line size).
106  * The 'align_offset.value' option can be used to fine-tune the
107  * start-address by a small offset to e.g. make room for a small
108  * SW header before the rest of the payload that might need a specific
109  * alignment for direct HW-access.
110  * Example: setting 'align_offset.value = 8' makes sure that the payload
111  * _after_ 8 bytes will be aligned at minimum (2^x) 32 bytes.
112  *
113  * This option conserns all events allocated from the pool and overrides
114  * the global config file option 'pool.align_offset' for this pool.
115  */
116  struct {
117  /**
118  * Select: Use pool-specific align-offset 'value' from below or
119  * use the global default value from the config file.
120  * false: Use value from the config file (default).
121  * true: Use pool-specific value set below.
122  */
123  bool in_use;
124  /**
125  * Pool-specific event payload alignment offset value in bytes
126  * (only evaluated if 'in_use=true').
127  * Overrides the config file value for this pool.
128  * The given 'value' must be a small power-of-two: 2, 4, or 8
129  * 0: Explicitly set 'No align offset' for the pool.
130  */
131  uint32_t value;
132  } align_offset;
133 
134  /**
135  * Event user area size in bytes.
136  * (for all events allocated from this EM pool).
137  *
138  * The user area is located within the event metadata (hdr) and is not
139  * part of the event payload. The event user area can e.g. be used to
140  * store additional state data related to the payload contents. EM does
141  * not initialize the contents of the user area.
142  *
143  * This option concerns all events allocated from the pool and overrides
144  * the global config file option 'pool.user_area_size' for this pool.
145  */
146  struct {
147  /**
148  * Select: Use pool-specific event user area 'size' from below
149  * or use the global default value from the config file.
150  * false: Use user area size from the config file (default).
151  * true: Use pool-specific size set below.
152  */
153  bool in_use;
154  /**
155  * Pool-specific event user area size in bytes (only evaluated
156  * if 'in_use=true').
157  * Overrides the config file default size for this pool.
158  * 0: Explicitly set 'No user area' for the pool.
159  */
160  size_t size;
161  } user_area;
162 
163  /**
164  * Parameters for an EM-pool with '.event_type = EM_EVENT_TYPE_PACKET'
165  * Ignored for other pool types.
166  */
167  struct {
168  /**
169  * Pool-specific packet minimum headroom
170  *
171  * This option conserns all events allocated from the pool and
172  * overrides the global config file option 'pool.pkt_headroom'
173  * for this pool.
174  */
175  struct {
176  /**
177  * Select: Use pool-specific packet headroom value from
178  * below or use the global default value from
179  * the config file.
180  * false: Use value from the config file (default).
181  * true: Use pool-specific value set below.
182  */
183  bool in_use;
184  /**
185  * Pool-specific packet minimum headroom in bytes,
186  * each packet must have at least this much headroom.
187  * (only evaluated if 'in_use=true').
188  * Overrides the config file value for this pool.
189  * 0: Explicitly set 'No align offset' for the pool.
190  */
191  uint32_t value;
192  } headroom;
193  } pkt;

195  /**
196  * Number of subpools within one EM pool, max=EM_MAX_SUBPOOLS
197  */
198  int num_subpools;
199  struct {
200  /**
201  * Event payload size of the subpool (size > 0).
202  * EM does not initialize the payload data.
203  */
204  uint32_t size;
205  /** Number of events in the subpool (num > 0) */
206  uint32_t num;
207  /**
208  * Maximum number of locally cached subpool events per EM-core.
209  *
210  * Allocating or freeing events from a core-local event-cache
211  * can be faster than using the global event subpool. Cached
212  * events are only available on the local core and can reduce
213  * the number of globally free events in the subpool, thus
214  * consider setting 'num > EM-core-count * cache_size'.
215  * The actual used cache_size will be smaller than or equal to
216  * the requested value, depending on the implementation.
217  */
218  uint32_t cache_size;
219  } subpool[EM_MAX_SUBPOOLS];

222  * Internal check - don't touch!
223  *
224  * EM will verify that em_pool_cfg_init(pool_cfg) has been called before
225  * creating a pool with em_pool_create(..., pool_cfg)
226  */
227  uint32_t __internal_check;
228 } em_pool_cfg_t;

350 /**
351  * Initialize event pool iteration and return the first event pool handle.
352  *
353  * Can be used to initialize the iteration to retrieve all created event pools
354  * for debugging or management purposes. Use em_pool_get_next() after this call
355  * until it returns EM_POOL_UNDEF.
356  * A new call to em_pool_get_first() resets the iteration, which is maintained
357  * per core (thread). The operation should be completed in one go before
358  * returning from the EO's event receive function (or start/stop).
359  *
360  * The number of event pools (output arg 'num') may not match the amount of
361  * event pools actually returned by iterating using em_pool_get_next()
362  * if event pools are added or removed in parallel by another core. The order
363  * of the returned event pool handles is undefined.
364  *
365  * @code
366  * unsigned int num;
367  * em_pool_t pool = em_pool_get_first(&num);
368  * while (pool != EM_POOL_UNDEF) {
369  * pool = em_pool_get_next();
370  * }
371  * @endcode
372  *
373  * @param[out] num Pointer to an unsigned int to store the amount of
374  * event pools into
375  * @return The first event pool handle or EM_POOL_UNDEF if none exist
376  *
377  * @see em_pool_get_next()
378  */
379 em_pool_t
380 em_pool_get_first(unsigned int *num);

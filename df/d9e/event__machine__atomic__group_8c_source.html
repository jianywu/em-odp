39 em_atomic_group_t
40 em_atomic_group_create(const char *name, em_queue_group_t queue_group)
41 {
42  em_atomic_group_t atomic_group = EM_ATOMIC_GROUP_UNDEF;
43  atomic_group_elem_t *ag_elem = NULL;
44  const char *err_str = "";
45  em_status_t error = EM_OK;
46  odp_queue_param_t queue_param;
47  unsigned int size;
48 
49  if (unlikely(invalid_qgrp(queue_group))) {
50  error = EM_ERR_BAD_ID;
51  err_str = "Invalid queue group!";
52  goto error;
53  }
54 
55  /* New Atomic group */
56  atomic_group = atomic_group_alloc();
57 
58  if (unlikely(atomic_group == EM_ATOMIC_GROUP_UNDEF)) {
59  error = EM_ERR_ALLOC_FAILED;
60  err_str = "Atomic group allocation failed!";
61  goto error;
62  }
63 
64  /* Initialize the atomic group */
65  ag_elem = atomic_group_elem_get(atomic_group);
66  if (unlikely(!ag_elem)) {
67  error = EM_ERR_BAD_POINTER;
68  err_str = "Atomic group allocation failed: ag_elem NULL!";
69  goto error;
70  }
71 
72  env_atomic32_init(&ag_elem->num_queues);
73 
74  /* Store the related queue group */
75  ag_elem->queue_group = queue_group;
76 
77  if (name != NULL) {
78  strncpy(ag_elem->name, name, sizeof(ag_elem->name));
79  ag_elem->name[sizeof(ag_elem->name) - 1] = '\0';
80  } else {
81  ag_elem->name[0] = '\0';
82  }
83 
84  odp_queue_param_init(&queue_param);
85 
86  queue_param.type = ODP_QUEUE_TYPE_PLAIN;
87  queue_param.enq_mode = ODP_QUEUE_OP_MT;
88  /* dequeueing protected by ag_elem->lock */
89  queue_param.deq_mode = ODP_QUEUE_OP_MT_UNSAFE;
90  /* Queue size: use EM default value from config file: */
91  size = em_shm->opt.queue.min_events_default;
92  if (size != 0)
93  queue_param.size = size;
94  /* else: use odp default as set by odp_queue_param_init() */
95 
96  ag_elem->internal_queue.hi_prio = odp_queue_create(ag_elem->name,
97  &queue_param);
98  ag_elem->internal_queue.lo_prio = odp_queue_create(ag_elem->name,
99  &queue_param);
100  if (unlikely(ag_elem->internal_queue.hi_prio == ODP_QUEUE_INVALID ||
101  ag_elem->internal_queue.lo_prio == ODP_QUEUE_INVALID))
102  goto error;
103 
104  return atomic_group;
105 
106 error:
107  INTERNAL_ERROR(error, EM_ESCOPE_ATOMIC_GROUP_CREATE, err_str);
108  if (atomic_group != EM_ATOMIC_GROUP_UNDEF)
109  em_atomic_group_delete(atomic_group);
110 
111  return EM_ATOMIC_GROUP_UNDEF;
112 }

114 /*
115  * Helper for em_atomic_group_delete()
116  * Flush the atomic group's internal queues and then destroy them.
117  */
118 static int
119 ag_internal_queue_destroy(odp_queue_t plain_q)
120 {
121  odp_event_t odp_deq_tbl[EM_SCHED_AG_MULTI_MAX_BURST];
122  event_hdr_t *ev_hdr_tbl[EM_SCHED_MULTI_MAX_BURST];
123  em_event_t ev_tbl[EM_SCHED_AG_MULTI_MAX_BURST];
124  int ev_cnt = 0;
125  bool esv_ena = esv_enabled();
126 
127  if (plain_q == ODP_QUEUE_INVALID)
128  return -1;
129 
130  do {
131  ev_cnt = odp_queue_deq_multi(plain_q, odp_deq_tbl,
132  EM_SCHED_AG_MULTI_MAX_BURST);
133  if (ev_cnt <= 0)
134  break;
135 
136  events_odp2em(odp_deq_tbl, ev_tbl/*out*/, ev_cnt);
137 
138  if (esv_ena) {
139  event_to_hdr_multi(ev_tbl, ev_hdr_tbl/*out*/, ev_cnt);
140  evstate_em2usr_multi(ev_tbl/*in/out*/, ev_hdr_tbl,
141  ev_cnt, EVSTATE__AG_DELETE);
142  }
143 
144  em_free_multi(ev_tbl, ev_cnt);
145  } while (ev_cnt > 0);
146 
147  return odp_queue_destroy(plain_q);
148 }

150 em_status_t
151 em_atomic_group_delete(em_atomic_group_t atomic_group)
152 {
153  atomic_group_elem_t *const ag_elem =
154  atomic_group_elem_get(atomic_group);
155  em_status_t error = EM_OK;
156  int err = 0;
157 
158  RETURN_ERROR_IF(ag_elem == NULL,
159  EM_ERR_BAD_ID, EM_ESCOPE_ATOMIC_GROUP_DELETE,
160  "Invalid atomic group - cannot delete!");
161 
162  env_spinlock_lock(&ag_elem->lock);
163 
164  /* Error checks */
165  err = !list_is_empty(&ag_elem->qlist_head);
166  err |= !atomic_group_allocated(ag_elem);
167 
168  if (unlikely(err)) {
169  env_spinlock_unlock(&ag_elem->lock);
170  return INTERNAL_ERROR(EM_ERR_BAD_STATE,
171  EM_ESCOPE_ATOMIC_GROUP_DELETE,
172  "Atomic group in bad state - cannot delete!");
173  }
174 
175  /* Flush the atomic group's internal queues and destroy them */
176  err = ag_internal_queue_destroy(ag_elem->internal_queue.hi_prio);
177  err |= ag_internal_queue_destroy(ag_elem->internal_queue.lo_prio);
178 
179  ag_elem->queue_group = EM_QUEUE_GROUP_UNDEF;
180  ag_elem->name[0] = '\0';
181 
182  env_spinlock_unlock(&ag_elem->lock);
183 
184  /* Free the atomic group (elem) back into the AG-pool */
185  error = atomic_group_free(atomic_group);
186  RETURN_ERROR_IF(error != EM_OK || err != 0,
187  error, EM_ESCOPE_ATOMIC_GROUP_DELETE,
188  "Atomic group free failed(%d)!", err);
189 
190  return EM_OK;
191 }

193 em_queue_t
194 em_queue_create_ag(const char *name, em_queue_prio_t prio,
195  em_atomic_group_t atomic_group, const em_queue_conf_t *conf)
196 {
197  em_queue_t queue;
198  queue_elem_t *q_elem;
199  em_queue_group_t queue_group;
200  atomic_group_elem_t *const ag_elem =
201  atomic_group_elem_get(atomic_group);
202  const char *err_str = "";
203 
204  if (unlikely(ag_elem == NULL || !atomic_group_allocated(ag_elem))) {
205  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_QUEUE_CREATE_AG,
206  "Invalid Atomic Group:%" PRI_AGRP "",
207  atomic_group);
208  return EM_QUEUE_UNDEF;
209  }
210 
211  queue_group = ag_elem->queue_group;
212 
213  queue = queue_create(name, EM_QUEUE_TYPE_ATOMIC, prio, queue_group,
214  EM_QUEUE_UNDEF, atomic_group, conf, &err_str);
215 
216  if (unlikely(queue == EM_QUEUE_UNDEF)) {
217  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_QUEUE_CREATE_AG,
218  "Atomic Group queue creation failed! (%s)",
219  err_str);
220  return EM_QUEUE_UNDEF;
221  }
222 
223  q_elem = queue_elem_get(queue);
224  /* Add queue to atomic group list */
225  atomic_group_add_queue_list(ag_elem, q_elem);
226 
227  return queue;
228 }

280 em_atomic_group_get_name(em_atomic_group_t atomic_group,
281  char *name, size_t maxlen)
282 {
283  const atomic_group_elem_t *ag_elem =
284  atomic_group_elem_get(atomic_group);
285  size_t len = 0;
286 
287  if (unlikely(name == NULL || maxlen == 0)) {
288  INTERNAL_ERROR(EM_ERR_BAD_POINTER,
289  EM_ESCOPE_ATOMIC_GROUP_GET_NAME,
290  "Invalid args: name=0x%" PRIx64 ", maxlen=%zu",
291  name, maxlen);
292  return 0;
293  }
294 
295  if (unlikely(ag_elem == NULL || !atomic_group_allocated(ag_elem))) {
296  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_ATOMIC_GROUP_GET_NAME,
297  "Invalid Atomic Group:%" PRI_AGRP "",
298  atomic_group);
299  name[0] = '\0';
300  return 0;
301  }
302 
303  len = strnlen(ag_elem->name, sizeof(ag_elem->name) - 1);
304  if (maxlen - 1 < len)
305  len = maxlen - 1;
306 
307  memcpy(name, ag_elem->name, len);
308  name[len] = '\0';
309 
310  return len;
311 }

330 em_atomic_group_t
331 em_atomic_group_get_first(unsigned int *num)
332 {
333  const atomic_group_elem_t *const agrp_elem_tbl =
334  em_shm->atomic_group_tbl.ag_elem;
335  const atomic_group_elem_t *ag_elem = &agrp_elem_tbl[0];
336  const unsigned int agrp_count = atomic_group_count();
337 
338  _agrp_tbl_iter_idx = 0; /* reset iteration */
339 
340  if (num)
341  *num = agrp_count;
342 
343  if (agrp_count == 0) {
344  _agrp_tbl_iter_idx = EM_MAX_ATOMIC_GROUPS; /*UNDEF=_get_next()*/
345  return EM_ATOMIC_GROUP_UNDEF;
346  }
347 
348  /* find first */
349  while (!atomic_group_allocated(ag_elem)) {
350  _agrp_tbl_iter_idx++;
351  if (_agrp_tbl_iter_idx >= EM_MAX_ATOMIC_GROUPS)
352  return EM_ATOMIC_GROUP_UNDEF;
353  ag_elem = &agrp_elem_tbl[_agrp_tbl_iter_idx];
354  }
355 
356  return agrp_idx2hdl(_agrp_tbl_iter_idx);
357 }

359 em_atomic_group_t
360 em_atomic_group_get_next(void)
361 {
362  if (_agrp_tbl_iter_idx >= EM_MAX_ATOMIC_GROUPS - 1)
363  return EM_ATOMIC_GROUP_UNDEF;
364 
365  _agrp_tbl_iter_idx++;
366 
367  const atomic_group_elem_t *const agrp_elem_tbl =
368  em_shm->atomic_group_tbl.ag_elem;
369  const atomic_group_elem_t *ag_elem = &agrp_elem_tbl[_agrp_tbl_iter_idx];
370 
371  /* find next */
372  while (!atomic_group_allocated(ag_elem)) {
373  _agrp_tbl_iter_idx++;
374  if (_agrp_tbl_iter_idx >= EM_MAX_ATOMIC_GROUPS)
375  return EM_ATOMIC_GROUP_UNDEF;
376  ag_elem = &agrp_elem_tbl[_agrp_tbl_iter_idx];
377  }
378 
379  return agrp_idx2hdl(_agrp_tbl_iter_idx);

382 em_queue_t
383 em_atomic_group_queue_get_first(unsigned int *num,
384  em_atomic_group_t atomic_group)
385 {
386  const atomic_group_elem_t *const agrp_elem =
387  atomic_group_elem_get(atomic_group);
388 
389  if (unlikely(agrp_elem == NULL || !atomic_group_allocated(agrp_elem))) {
390  INTERNAL_ERROR(EM_ERR_BAD_ID,
391  EM_ESCOPE_ATOMIC_GROUP_QUEUE_GET_FIRST,
392  "Invalid atomic group:%" PRI_AGRP "",
393  atomic_group);
394  if (num)
395  *num = 0;
396  return EM_QUEUE_UNDEF;
397  }
398 
399  const unsigned int num_queues =
400  env_atomic32_get(&agrp_elem->num_queues);
401 
402  if (num)
403  *num = num_queues;
404 
405  if (num_queues == 0) {
406  _agrp_q_iter_idx = EM_MAX_QUEUES; /* UNDEF = _get_next() */
407  return EM_QUEUE_UNDEF;
408  }
409 
410  /*
411  * A 'agrp_elem' contains a linked list with all it's queues. That list
412  * might be modified while processing this iteration, so instead we just
413  * go through the whole queue table.
414  * This is potentially a slow implementation and perhaps worth
415  * re-thinking?
416  */
417  const queue_elem_t *const q_elem_tbl = em_shm->queue_tbl.queue_elem;
418  const queue_elem_t *q_elem = &q_elem_tbl[0];
419 
420  _agrp_q_iter_idx = 0; /* reset list */
421  _agrp_q_iter_agrp = atomic_group;
422 
423  /* find first */
424  while (!queue_allocated(q_elem) ||
425  q_elem->atomic_group != _agrp_q_iter_agrp) {
426  _agrp_q_iter_idx++;
427  if (_agrp_q_iter_idx >= EM_MAX_QUEUES)
428  return EM_QUEUE_UNDEF;
429  q_elem = &q_elem_tbl[_agrp_q_iter_idx];
430  }
431 
432  return queue_idx2hdl(_agrp_q_iter_idx);
433 }
